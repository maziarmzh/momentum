34template <typename T>
35void VertexPositionErrorFunctionT<T>::evalFunction(
36 size_t constrIndex,
37 const SkeletonStateT<T>& /*state*/,
38 const MeshStateT<T>& /*meshState*/,
39 std::span<const Eigen::Vector3<T>> worldVecs,
40 FuncType& f,
41 std::span<DfdvType> dfdv) const {
42 MT_CHECK(constrIndex < this->constraints_.size(), "Constraint index out of range");
43 MT_CHECK(worldVecs.size() == 1, "Expected exactly one world vector for vertex position");
44
45 const auto& constraint = this->constraints_[constrIndex];
46
47 // Residual: f = vertexPosition - target
48 f = worldVecs[0] - constraint.target;
49
50 // Derivative: df/dv = I (identity matrix)
51 if (!dfdv.empty()) {
52 MT_CHECK(dfdv.size() == 1, "Expected exactly one dfdv for vertex position");
53 dfdv[0] = Eigen::Matrix<T, 3, 3>::Identity();
54 }
55}

57template <typename T>
58double VertexPositionErrorFunctionT<T>::getGradient(
59 const ModelParametersT<T>& /*params*/,
60 const SkeletonStateT<T>& state,
61 const MeshStateT<T>& meshState,
62 Eigen::Ref<Eigen::VectorX<T>> gradient) {
63 const auto& constraints = this->constraints_;
64 const size_t numConstraints = constraints.size();
65 if (numConstraints == 0) {
66 return 0.0;
67 }
68
69 auto dispensoOptions = dispenso::ParForOptions();
70 dispensoOptions.maxThreads = static_cast<uint32_t>(this->maxThreads_);
71 const SkeletonDerivativeT<T> skeletonDerivative(
72 this->skeleton_,
73 this->parameterTransform_,
74 this->activeJointParams_,
75 this->enabledParameters_);
76
77 const bool isL2 = this->loss_.isL2();
78 const T lossInvC2 = this->loss_.invC2();
79
80 std::vector<std::tuple<double, VectorX<T>>> errorGradThread;
81 dispenso::parallel_for(
82 errorGradThread,
83 [&]() -> std::tuple<double, VectorX<T>> {
84 return {0.0, VectorX<T>::Zero(this->parameterTransform_.numAllModelParameters())};
85 },
86 size_t(0),
87 numConstraints,
88 [&](std::tuple<double, VectorX<T>>& errorGradLocal, const size_t i) {
89 double& errorLocal = std::get<0>(errorGradLocal);
90 auto& gradLocal = std::get<1>(errorGradLocal);
91
92 const T constrWeight = static_cast<T>(constraints[i].weight);
93 if (constrWeight == T(0)) {
94 return;
95 }
96
97 const size_t vertexIndex = constraints[i].vertexIndex;
98 const Eigen::Vector3<T> worldPos =
99 meshState.posedMesh_->vertices[vertexIndex].template cast<T>();
100
101 // Inline the position residual: f = worldPos - target (no virtual call)
102 const Eigen::Vector3<T> f = worldPos - constraints[i].target;
103
104 if (f.isZero()) {
105 return;
106 }
107
108 const T sqrError = f.squaredNorm();
109 const T w = constrWeight * this->weight_ * this->legacyWeight_;
110
111 Eigen::Vector3<T> weightedResidual;
112 if (isL2) {
113 errorLocal += w * sqrError * lossInvC2;
114 weightedResidual = T(2) * w * lossInvC2 * f;
115 } else {
116 errorLocal += w * this->loss_.value(sqrError);
117 weightedResidual = T(2) * w * this->loss_.deriv(sqrError) * f;
118 }
119
120 // Use IdentityDfdv specialization — eliminates dfdv*derivative matrix multiply
121 skeletonDerivative.accumulateVertexGradientIdentityDfdv(
122 vertexIndex, weightedResidual, state, meshState, this->character_, gradLocal);
123 },
124 dispensoOptions);
125
126 if (!errorGradThread.empty()) {
127 errorGradThread[0] = std::accumulate(
128 errorGradThread.begin() + 1,
129 errorGradThread.end(),
130 errorGradThread[0],
131 [](const auto& a, const auto& b) -> std::tuple<double, VectorX<T>> {
132 return {std::get<0>(a) + std::get<0>(b), std::get<1>(a) + std::get<1>(b)};
133 });
134
135 gradient += std::get<1>(errorGradThread[0]);
136 return std::get<0>(errorGradThread[0]);
137 }
138
139 return 0.0;
140}

142template <typename T>
143double VertexPositionErrorFunctionT<T>::getJacobian(
144 const ModelParametersT<T>& /*params*/,
145 const SkeletonStateT<T>& state,
146 const MeshStateT<T>& meshState,
147 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
148 Eigen::Ref<Eigen::VectorX<T>> residual,
149 int& usedRows) {
150 const auto& constraints = this->constraints_;
151 const size_t numConstraints = constraints.size();
152 usedRows = 0;
153 if (numConstraints == 0) {
154 return 0.0;
155 }
156
157 auto dispensoOptions = dispenso::ParForOptions();
158 dispensoOptions.maxThreads = static_cast<uint32_t>(this->maxThreads_);
159 const SkeletonDerivativeT<T> skeletonDerivative(
160 this->skeleton_,
161 this->parameterTransform_,
162 this->activeJointParams_,
163 this->enabledParameters_);
164
165 constexpr size_t FuncDim = 3;
166 const bool isL2 = this->loss_.isL2();
167 const T lossInvC2 = this->loss_.invC2();
168
169 std::vector<double> errorThread;
170 dispenso::parallel_for(
171 errorThread,
172 [&]() -> double { return 0.0; },
173 size_t(0),
174 numConstraints,
175 [&](double& errorLocal, const size_t i) {
176 const T constrWeight = static_cast<T>(constraints[i].weight);
177 if (constrWeight == T(0)) {
178 return;
179 }
180
181 const size_t vertexIndex = constraints[i].vertexIndex;
182 const Eigen::Vector3<T> worldPos =
183 meshState.posedMesh_->vertices[vertexIndex].template cast<T>();
184
185 const size_t rowIndex = i * FuncDim;
186
187 // Inline the position residual: f = worldPos - target (no virtual call)
188 const Eigen::Vector3<T> f = worldPos - constraints[i].target;
189
190 const T sqrError = f.squaredNorm();
191 const T w = constrWeight * this->weight_ * this->legacyWeight_;
192
193 T deriv;
194 if (isL2) {
195 errorLocal += w * sqrError * lossInvC2;
196 deriv = std::sqrt(w * lossInvC2);
197 } else {
198 errorLocal += w * this->loss_.value(sqrError);
199 deriv = std::sqrt(w * this->loss_.deriv(sqrError));
200 }
201
202 residual.template segment<FuncDim>(rowIndex).noalias() = deriv * f;
203
204 if (deriv == T(0)) {
205 return;
206 }
207
208 // Use IdentityDfdv specialization — eliminates dfdv*derivative matrix multiply
209 skeletonDerivative.accumulateVertexJacobianIdentityDfdv(
210 vertexIndex, deriv, state, meshState, this->character_, jacobian, rowIndex);
211 },
212 dispensoOptions);
213
214 usedRows = static_cast<int>(numConstraints * FuncDim);
215
216 if (!errorThread.empty()) {
217 return std::accumulate(errorThread.begin() + 1, errorThread.end(), errorThread[0]);
218 }
219
220 return 0.0;
221}
