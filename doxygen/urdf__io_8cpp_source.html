33struct MimicData {
34 std::string mimickedJointName; // URDF joint name to mimic
35 float multiplier{};
36 float offset{}; // radians for revolute, meters for prismatic
37 Eigen::Index jointParamsBaseIndex{}; // this joint's parameter base index
38 int axisIdx{}; // principal axis index (0=X, 1=Y, 2=Z)

107[[nodiscard]] std::pair<int, float> getPrincipalAxis(const urdf::Vector3& axis) {
108 const auto ax = static_cast<float>(axis.x);
109 const auto ay = static_cast<float>(axis.y);
110 const auto az = static_cast<float>(axis.z);
111 constexpr float kTol = 1e-4f;
112
113 if (std::abs(std::abs(ax) - 1.0f) < kTol && std::abs(ay) < kTol && std::abs(az) < kTol) {
114 return {0, ax > 0 ? 1.0f : -1.0f};
115 }
116 if (std::abs(ax) < kTol && std::abs(std::abs(ay) - 1.0f) < kTol && std::abs(az) < kTol) {
117 return {1, ay > 0 ? 1.0f : -1.0f};
118 }
119 if (std::abs(ax) < kTol && std::abs(ay) < kTol && std::abs(std::abs(az) - 1.0f) < kTol) {
120 return {2, az > 0 ? 1.0f : -1.0f};
121 }
122
123 MT_THROW(
124 "Non-principal joint axis ({}, {}, {}) is not supported. "
125 "Only axes aligned with X, Y, or Z are supported.",
126 ax,
127 ay,
128 az);
129}

136std::optional<Mesh> loadVisualMesh(
137 const urdf::Geometry& geometry,
138 const filesystem::path& urdfDir) {
139 switch (geometry.type) {
140 case urdf::Geometry::MESH: {
141 const auto& urdfMesh = dynamic_cast<const urdf::Mesh&>(geometry);
142 if (urdfDir.empty()) {
143 MT_LOGW("Cannot resolve mesh path without URDF directory: {}", urdfMesh.filename);
144 return std::nullopt;
145 }
146 const auto resolvedPath = resolveMeshPath(urdfMesh.filename, urdfDir);
147 if (!resolvedPath) {
148 return std::nullopt;
149 }
150
151 const auto ext = resolvedPath->extension().string();
152 std::optional<Mesh> mesh;
153 if (ext == ".stl" || ext == ".STL") {
154 mesh = loadStlMesh(*resolvedPath);
155 } else if (ext == ".obj" || ext == ".OBJ") {
156 mesh = loadObjMesh(*resolvedPath);
157 } else {
158 MT_LOGW("Unsupported mesh format '{}': {}", ext, resolvedPath->string());
159 return std::nullopt;
160 }
161
162 if (!mesh) {
163 return std::nullopt;
164 }
165
166 // Apply scale from URDF mesh element
167 const Eigen::Vector3f scale(
168 static_cast<float>(urdfMesh.scale.x),
169 static_cast<float>(urdfMesh.scale.y),
170 static_cast<float>(urdfMesh.scale.z));
171 if (!scale.isApprox(Eigen::Vector3f::Ones())) {
172 for (auto& v : mesh->vertices) {
173 v = v.cwiseProduct(scale);
174 }
175 }
176 return mesh;
177 }
178 case urdf::Geometry::BOX: {
179 const auto& box = dynamic_cast<const urdf::Box&>(geometry);
180 return generateBoxMesh(
181 static_cast<float>(box.dim.x),
182 static_cast<float>(box.dim.y),
183 static_cast<float>(box.dim.z));
184 }
185 case urdf::Geometry::CYLINDER: {
186 const auto& cyl = dynamic_cast<const urdf::Cylinder&>(geometry);
187 return generateCylinderMesh(static_cast<float>(cyl.radius), static_cast<float>(cyl.length));
188 }
189 case urdf::Geometry::SPHERE: {
190 const auto& sphere = dynamic_cast<const urdf::Sphere&>(geometry);
191 return generateSphereMesh(static_cast<float>(sphere.radius));
192 }
193 default:
194 MT_LOGW("Unknown URDF geometry type: {}", static_cast<int>(geometry.type));
195 return std::nullopt;
196 }
197}

202template <typename T>
203bool loadUrdfSkeletonRecursive(
204 ParsingData<T>& data,
205 size_t parentJointId,
206 const urdf::ModelInterface* urdfModel,
207 const urdf::Link* urdfLink) {
208 MT_THROW_IF(urdfLink == nullptr, "URDF link is null.");
209
210 //-------------
211 // Parse joint
212 //-------------
213
214 auto* urdfJoint = urdfLink->parent_joint.get();
215 MT_THROW_IF(
216 parentJointId != kInvalidIndex && urdfJoint == nullptr,
217 "URDF parent joint is null for a non root link ({}).",
218 urdfLink->name);
219
220 Joint joint;
221 joint.name = urdfLink->name;
222 joint.parent = parentJointId;
223
224 const Eigen::Index jointId = data.skeleton.joints.size();
225 const Eigen::Index jointParamsBaseIndex = jointId * kParametersPerJoint;
226 const Eigen::Index modelParamsBaseIndex = data.totalDoFs;
227
228 //---------------------------
229 // Parse Parameter transform
230 //---------------------------
231 //
232 // URDF joints specify an arbitrary axis direction for their DOF. Instead of aligning that
233 // axis to Momentum's X axis via preRotation (which would create a rotated internal frame),
234 // we map the DOF directly to the corresponding principal axis parameter (rx/ry/rz for
235 // revolute, tx/ty/tz for prismatic). This preserves the URDF link frame as the joint's
236 // local frame, making visual meshes and animation data directly compatible.
237
238 if (urdfJoint != nullptr) {
239 // Check if this joint mimics another joint. Mimic joints don't create their own model
240 // parameter — they reuse the mimicked joint's parameter with a multiplier and offset.
241 const bool isMimic = urdfJoint->mimic != nullptr;
242
243 switch (urdfJoint->type) {
244 case urdf::Joint::PRISMATIC: {
245 const auto [axisIdx, sign] = getPrincipalAxis(urdfJoint->axis);
246 if (isMimic) {
247 MimicData mimic;
248 mimic.mimickedJointName = urdfJoint->mimic->joint_name;
249 mimic.multiplier = static_cast<float>(urdfJoint->mimic->multiplier);
250 mimic.offset = static_cast<float>(urdfJoint->mimic->offset);
251 mimic.jointParamsBaseIndex = jointParamsBaseIndex;
252 mimic.axisIdx = axisIdx;
253 mimic.sign = sign;
254 mimic.isRotation = false;
255 data.mimicJoints.push_back(mimic);
256 } else {
257 data.parameterTransform.name.push_back(urdfJoint->name);
258 data.triplets.emplace_back(jointParamsBaseIndex + axisIdx, modelParamsBaseIndex, sign);
259 data.urdfJointNameToModelParamIdx[urdfJoint->name] = modelParamsBaseIndex;
260 data.totalDoFs += 1;
261 }
262 break;
263 }
264 case urdf::Joint::REVOLUTE:
265 case urdf::Joint::CONTINUOUS: {
266 const auto [axisIdx, sign] = getPrincipalAxis(urdfJoint->axis);
267 if (isMimic) {
268 MimicData mimic;
269 mimic.mimickedJointName = urdfJoint->mimic->joint_name;
270 mimic.multiplier = static_cast<float>(urdfJoint->mimic->multiplier);
271 mimic.offset = static_cast<float>(urdfJoint->mimic->offset);
272 mimic.jointParamsBaseIndex = jointParamsBaseIndex;
273 mimic.axisIdx = axisIdx;
274 mimic.sign = sign;
275 mimic.isRotation = true;
276 data.mimicJoints.push_back(mimic);
277 } else {
278 data.parameterTransform.name.push_back(urdfJoint->name);
279 data.triplets.emplace_back(
280 jointParamsBaseIndex + 3 + axisIdx, modelParamsBaseIndex, sign);
281 data.urdfJointNameToModelParamIdx[urdfJoint->name] = modelParamsBaseIndex;
282 data.totalDoFs += 1;
283 }
284 break;
285 }
286 case urdf::Joint::FLOATING: {
287 // Floating joints have 6 DOFs; use the URDF joint name with axis suffixes.
288 // Translation parameters use a 10x scaling factor so that a unit change in
289 // model parameter space produces 10 cm of translation. This makes translation
290 // easier to change relative to rotation in optimization.
291 constexpr float kFloatingTranslationScale = 10.0f;
292 const auto& jn = urdfJoint->name;
293 data.parameterTransform.name.push_back(fmt::format("{}_tx", jn));
294 data.triplets.emplace_back(
295 jointParamsBaseIndex + 0, modelParamsBaseIndex + 0, kFloatingTranslationScale);
296 data.parameterTransform.name.push_back(fmt::format("{}_ty", jn));
297 data.triplets.emplace_back(
298 jointParamsBaseIndex + 1, modelParamsBaseIndex + 1, kFloatingTranslationScale);
299 data.parameterTransform.name.push_back(fmt::format("{}_tz", jn));
300 data.triplets.emplace_back(
301 jointParamsBaseIndex + 2, modelParamsBaseIndex + 2, kFloatingTranslationScale);
302 data.parameterTransform.name.push_back(fmt::format("{}_rx", jn));
303 data.triplets.emplace_back(jointParamsBaseIndex + 3, modelParamsBaseIndex + 3, 1.0f);
304 data.parameterTransform.name.push_back(fmt::format("{}_ry", jn));
305 data.triplets.emplace_back(jointParamsBaseIndex + 4, modelParamsBaseIndex + 4, 1.0f);
306 data.parameterTransform.name.push_back(fmt::format("{}_rz", jn));
307 data.triplets.emplace_back(jointParamsBaseIndex + 5, modelParamsBaseIndex + 5, 1.0f);
308 data.totalDoFs += 6;
309 break;
310 }
311 case urdf::Joint::PLANAR: {
312 // The axis defines the plane normal. DOFs are translation along the two in-plane
313 // axes and rotation around the normal.
314 const auto [normalIdx, sign] = getPrincipalAxis(urdfJoint->axis);
315 const int planeAxis0 = (normalIdx + 1) % 3;
316 const int planeAxis1 = (normalIdx + 2) % 3;
317 const auto& jn = urdfJoint->name;
318 data.parameterTransform.name.push_back(
319 fmt::format("{}_{}", jn, kTranslationNames[planeAxis0]));
320 data.triplets.emplace_back(
321 jointParamsBaseIndex + planeAxis0, modelParamsBaseIndex + 0, 1.0f);
322 data.parameterTransform.name.push_back(
323 fmt::format("{}_{}", jn, kTranslationNames[planeAxis1]));
324 data.triplets.emplace_back(
325 jointParamsBaseIndex + planeAxis1, modelParamsBaseIndex + 1, 1.0f);
326 data.parameterTransform.name.push_back(fmt::format("{}_{}", jn, kRotationNames[normalIdx]));
327 data.triplets.emplace_back(
328 jointParamsBaseIndex + 3 + normalIdx, modelParamsBaseIndex + 2, sign);
329 data.totalDoFs += 3;
330 break;
331 }
332 case urdf::Joint::FIXED: {
333 // No degrees of freedom
334 break;
335 }
336 default: {
337 MT_THROW("Unsupported joint type: {}", toString(urdfJoint->type));
338 }
339 }
340
341 // Parse joint limits (required only for non-mimic revolute and prismatic joints).
342 // Mimic joints are driven by the mimicked joint's parameter, so their limits don't
343 // correspond to an independent model parameter.
344 if (!isMimic && urdfJoint->limits) {
345 auto urdfJointLimits = urdfJoint->limits;
346 switch (urdfJoint->type) {
347 case urdf::Joint::REVOLUTE: {
348 const auto [axisIdx, sign] = getPrincipalAxis(urdfJoint->axis);
349 ParameterLimit jointLimits;
350 jointLimits.type = MinMax;
351 jointLimits.data.minMax.parameterIndex = modelParamsBaseIndex;
352 if (sign > 0) {
353 jointLimits.data.minMax.limits[0] = urdfJointLimits->lower; // rad
354 jointLimits.data.minMax.limits[1] = urdfJointLimits->upper;
355 } else {
356 // Negative axis: model param = -urdf_angle, so limits are negated and swapped
357 jointLimits.data.minMax.limits[0] = -urdfJointLimits->upper;
358 jointLimits.data.minMax.limits[1] = -urdfJointLimits->lower;
359 }
360 jointLimits.weight = 1.0f;
361 data.limits.push_back(jointLimits);
362 break;
363 }
364 case urdf::Joint::PRISMATIC: {
365 const auto [axisIdx, sign] = getPrincipalAxis(urdfJoint->axis);
366 ParameterLimit jointLimits;
367 jointLimits.type = MinMax;
368 jointLimits.data.minMax.parameterIndex = modelParamsBaseIndex;
369 if (sign > 0) {
370 jointLimits.data.minMax.limits[0] = toCm<float>(urdfJointLimits->lower);
371 jointLimits.data.minMax.limits[1] = toCm<float>(urdfJointLimits->upper);
372 } else {
373 jointLimits.data.minMax.limits[0] = -toCm<float>(urdfJointLimits->upper);
374 jointLimits.data.minMax.limits[1] = -toCm<float>(urdfJointLimits->lower);
375 }
376 jointLimits.weight = 1.0f;
377 data.limits.push_back(jointLimits);
378 break;
379 }
380 default: {
381 // Do nothing
382 break;
383 }
384 }
385 }
386 }
387
388 // Set Joint Offset
389 //
390 // The preRotation is simply the URDF joint origin rotation — the transform from the parent
391 // link frame to the child link frame at bind pose. No axis alignment is applied because DOFs
392 // are mapped directly to the corresponding principal axis parameter (rx/ry/rz).
393 //
394 // This means the Momentum joint frame coincides with the URDF link frame, which makes visual
395 // meshes and animation data directly compatible without any frame correction.
396 if (urdfJoint != nullptr) {
397 const urdf::Pose& urdfPose = urdfJoint->parent_to_joint_origin_transform;
398 joint.preRotation = toMomentumQuaternion<float>(urdfPose.rotation);
399 joint.translationOffset = toMomentumVector3<float>(urdfPose.position) * toCm<float>();
400 } else {
401 joint.preRotation = Quaternionf::Identity();
402 joint.translationOffset.setZero();
403 }
404
405 data.skeleton.joints.push_back(joint);
406
407 // Collect visual elements for mesh loading
408 if (!urdfLink->visual_array.empty()) {
409 LinkVisualData visualData;
410 visualData.jointIndex = static_cast<size_t>(jointId);
411 visualData.visuals = urdfLink->visual_array;
412 data.linkVisuals.push_back(std::move(visualData));
413 }
414
415 // Continue parsing child links and joints
416 for (const auto& childLink : urdfLink->child_links) {
417 if (!loadUrdfSkeletonRecursive(data, jointId, urdfModel, childLink.get())) {
418 return false;
419 }
420 }
421
422 return true;
423}

427template <typename T>
428void resolveMimicJoints(ParsingData<T>& data) {
429 for (const auto& mimic : data.mimicJoints) {
430 auto it = data.urdfJointNameToModelParamIdx.find(mimic.mimickedJointName);
431 if (it == data.urdfJointNameToModelParamIdx.end()) {
432 MT_LOGW("Mimic joint references unknown joint '{}', skipping.", mimic.mimickedJointName);
433 continue;
434 }
435 const Eigen::Index mimickedModelParamIdx = it->second;
436 const float coefficient = mimic.sign * mimic.multiplier;
437
438 if (mimic.isRotation) {
439 data.triplets.emplace_back(
440 mimic.jointParamsBaseIndex + 3 + mimic.axisIdx, mimickedModelParamIdx, coefficient);
441 data.parameterTransform.offsets[mimic.jointParamsBaseIndex + 3 + mimic.axisIdx] =
442 mimic.sign * mimic.offset;
443 } else {
444 data.triplets.emplace_back(
445 mimic.jointParamsBaseIndex + mimic.axisIdx, mimickedModelParamIdx, coefficient);
446 data.parameterTransform.offsets[mimic.jointParamsBaseIndex + mimic.axisIdx] =
447 mimic.sign * toCm<float>(mimic.offset);
448 }
449 }
450}

454Eigen::Vector3b extractVertexColor(const urdf::Visual& visual, bool& hasColor) {
455 if (visual.material &&
456 (visual.material->color.r > 0 || visual.material->color.g > 0 ||
457 visual.material->color.b > 0)) {
458 const auto& c = visual.material->color;
459 hasColor = true;
460 return {
461 static_cast<uint8_t>(std::clamp(c.r, 0.0f, 1.0f) * 255.0f),
462 static_cast<uint8_t>(std::clamp(c.g, 0.0f, 1.0f) * 255.0f),
463 static_cast<uint8_t>(std::clamp(c.b, 0.0f, 1.0f) * 255.0f)};
464 }
465 return {255, 255, 255};
466}

470template <typename T>
471std::optional<std::pair<Mesh, SkinWeights>> buildCombinedMesh(
472 const ParsingData<T>& data,
473 const filesystem::path& urdfDir) {
474 if (data.linkVisuals.empty()) {
475 return std::nullopt;
476 }
477
478 const SkeletonStateT<float> bindState(data.parameterTransform.bindPose(), data.skeleton, false);
479
480 Mesh combinedMesh;
481 SkinWeights skinWeights;
482 bool hasAnyColor = false;
483
484 for (const auto& linkVisual : data.linkVisuals) {
485 const size_t jointIdx = linkVisual.jointIndex;
486 const auto& jointWorldTransform = bindState.jointState[jointIdx].transform;
487
488 for (const auto& visual : linkVisual.visuals) {
489 if (!visual || !visual->geometry) {
490 continue;
491 }
492
493 auto meshOpt = loadVisualMesh(*visual->geometry, urdfDir);
494 if (!meshOpt) {
495 continue;
496 }
497
498 // Convert from meters (URDF) to centimeters (Momentum) and apply transforms
499 for (auto& v : meshOpt->vertices) {
500 v *= toCm<float>();
501 }
502 const auto visualOrigin = toMomentumTransform<float>(visual->origin);
503 const Eigen::Vector3f visualTranslation = visualOrigin.translation * toCm<float>();
504 for (auto& v : meshOpt->vertices) {
505 v = visualOrigin.rotation * v + visualTranslation;
506 }
507 for (auto& v : meshOpt->vertices) {
508 v = jointWorldTransform.rotation * v + jointWorldTransform.translation;
509 }
510
511 const Eigen::Vector3b vertexColor = extractVertexColor(*visual, hasAnyColor);
512
513 // Merge into the combined mesh
514 const auto vertexOffset = static_cast<int32_t>(combinedMesh.vertices.size());
515 combinedMesh.vertices.insert(
516 combinedMesh.vertices.end(), meshOpt->vertices.begin(), meshOpt->vertices.end());
517 for (const auto& face : meshOpt->faces) {
518 combinedMesh.faces.push_back(
519 {face[0] + vertexOffset, face[1] + vertexOffset, face[2] + vertexOffset});
520 }
521 combinedMesh.colors.resize(combinedMesh.vertices.size(), vertexColor);
522
523 // Extend skin weights: all new vertices are rigidly bound to this joint
524 const auto prevVertexCount = skinWeights.index.rows();
525 const auto newVertexCount =
526 prevVertexCount + static_cast<Eigen::Index>(meshOpt->vertices.size());
527 skinWeights.index.conservativeResize(newVertexCount, Eigen::NoChange);
528 skinWeights.weight.conservativeResize(newVertexCount, Eigen::NoChange);
529 skinWeights.index.bottomRows(newVertexCount - prevVertexCount).setZero();
530 skinWeights.weight.bottomRows(newVertexCount - prevVertexCount).setZero();
531 for (auto vi = prevVertexCount; vi < newVertexCount; ++vi) {
532 skinWeights.index(vi, 0) = static_cast<uint32_t>(jointIdx);
533 skinWeights.weight(vi, 0) = 1.0f;
534 }
535 }
536 }
537
538 if (!hasAnyColor) {
539 combinedMesh.colors.clear();
540 }
541
542 if (combinedMesh.vertices.empty()) {
543 return std::nullopt;
544 }
545
546 combinedMesh.updateNormals();
547 return std::make_pair(std::move(combinedMesh), std::move(skinWeights));
548}

550template <typename T>
551CharacterT<T> loadUrdfCharacterFromUrdfModel(
552 urdf::ModelInterfaceSharedPtr urdfModel,
553 const filesystem::path& urdfDir) {
554 const urdf::Link* root = urdfModel->getRoot().get();
555 MT_THROW_IF(!root, "Failed to parse URDF file from. No root link found.");
556
557 ParsingData<T> data;
558
559 // Special Case: If the root link is named "world", it is treated as a world link. In this case,
560 // the actual root link is the first child link. Otherwise, the root link itself is considered the
561 // actual root link, and it is assumed to have a fixed joint (hence, no degrees of freedom).
562 if (root->name == "world") {
563 if (root->child_links.empty()) {
564 MT_THROW(
565 "Failed to parse URDF. The world link must have at least one child link, but it has none.");
566 } else if (root->child_links.size() > 1) {
567 MT_THROW(
568 "Failed to parse URDF. The world link must have only one child link, but it has {}.",
569
570 root->child_links.size());
571 }
572 root = root->child_links[0].get();
573 }
574
575 if (!loadUrdfSkeletonRecursive(data, kInvalidIndex, urdfModel.get(), root)) {
576 MT_THROW("Failed to parse URDF.");
577 }
578
579 const size_t numJoints = data.skeleton.joints.size();
580 const size_t numJointParameters = numJoints * kParametersPerJoint;
581 const size_t numModelParameters = data.totalDoFs;
582 data.parameterTransform.offsets.setZero(numJointParameters);
583
584 resolveMimicJoints(data);
585
586 data.parameterTransform.transform.resize(numJointParameters, numModelParameters);
587 data.parameterTransform.transform.setFromTriplets(data.triplets.begin(), data.triplets.end());
588 data.parameterTransform.activeJointParams = data.parameterTransform.computeActiveJointParams();
589
590 const std::string robotName = urdfModel->getName();
591
592 auto meshResult = buildCombinedMesh(data, urdfDir);
593 if (meshResult) {
594 auto& [mesh, skinWeights] = *meshResult;
595 return CharacterT<T>(
596 data.skeleton,
597 data.parameterTransform,
598 data.limits,
599 LocatorList{},
600 &mesh,
601 &skinWeights,
602 nullptr, // collision
603 nullptr, // poseShapes
604 {}, // blendShapes
605 {}, // faceExpressionBlendShapes
606 robotName);
607 }
608
609 return CharacterT<T>(
610 data.skeleton,
611 data.parameterTransform,
612 data.limits,
613 LocatorList{},
614 nullptr, // mesh
615 nullptr, // skinWeights
616 nullptr, // collision
617 nullptr, // poseShapes
618 {}, // blendShapes
619 {}, // faceExpressionBlendShapes
620 robotName);
621}

626CharacterT<T> loadUrdfCharacter(const filesystem::path& filepath) {
627 urdf::ModelInterfaceSharedPtr urdfModel;
628
629 try {
630 urdfModel = urdf::parseURDFFile(filepath.string());
631 } catch (const std::exception& e) {
632 MT_THROW("Failed to parse URDF file from: {}. Error: {}", filepath.string(), e.what());

636
637 try {
638 return loadUrdfCharacterFromUrdfModel<T>(urdfModel, filepath.parent_path());

640 MT_THROW(
641 "Failed to create Character from URDF file: {}. Error: {}", filepath.string(), e.what());
642 } catch (...) {
643 MT_THROW("Failed to create Character from URDF file: {}", filepath.string());
644 }
645}
