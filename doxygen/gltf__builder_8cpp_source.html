58fx::gltf::Animation& addMomentumAnimationToModel(
59 fx::gltf::Document& model,
60 const std::string& nameIn = "default") {
61 auto animationIter = std::find_if(
62 model.animations.begin(), model.animations.end(), [nameIn](const fx::gltf::Animation& anim) {
63 return nameIn == anim.name;
64 });
65 if (animationIter == model.animations.end()) {
66 model.animations.emplace_back();
67 auto& animation = model.animations.at(model.animations.size() - 1);
68 animation.name = nameIn;
69 return animation;
70 }
71 return *animationIter;

86// Add a mesh node with only one primitive (i.e. combined mesh)
87// Use a flag to decide if we should save the vertex color. If a mesh only has default color, we may
88// not want to save it.
89size_t addMeshToModel(fx::gltf::Document& model, const Mesh& mesh, const bool addColor = false) {
90 if (mesh.vertices.empty()) {
91 return std::numeric_limits<size_t>::max();
92 }
93
94 model.meshes.emplace_back();
95 auto& m = model.meshes.back();
96 m.primitives.emplace_back();
97 auto& prim = m.primitives.back();
98
99 // Add faces to mesh
100 if ((!mesh.faces.empty()) && (mesh.faces[0].size() > 0)) {
101 prim.mode = fx::gltf::Primitive::Mode::Triangles;
102 const std::span<const int32_t> faces(&mesh.faces[0][0], mesh.faces.size() * 3);
103 prim.indices = createAccessorBuffer<const int32_t>(model, faces);
104 setBufferView(model, prim.indices, fx::gltf::BufferView::TargetType::ElementArrayBuffer);
105 } else {
106 prim.mode = fx::gltf::Primitive::Mode::Points;
107 }
108
109 // Add vertices to mesh
110 std::vector<Vector3f> verts = mesh.vertices;
111 fromMomentumVec3f(verts);
112 MT_CHECK((!verts.empty()) && (verts[0].size() > 0));
113 prim.attributes["POSITION"] = createAccessorBuffer<const Vector3f>(model, verts, true);
114 setBufferView(model, prim.attributes["POSITION"], fx::gltf::BufferView::TargetType::ArrayBuffer);
115
116 // need min/max for position accessor
117 Box3f bbox;
118 for (const auto& v : verts) {
119 bbox.extend(v);
120 }
121 for (size_t d = 0; d < 3; d++) {
122 model.accessors.at(prim.attributes["POSITION"]).min.emplace_back(bbox.min()[d]);
123 model.accessors.at(prim.attributes["POSITION"]).max.emplace_back(bbox.max()[d]);
124 }
125
126 // add normal attribute
127 if (mesh.normals.size() == mesh.vertices.size()) {
128 prim.attributes["NORMAL"] = createAccessorBuffer<const Vector3f>(model, mesh.normals, true);
129 setBufferView(model, prim.attributes["NORMAL"], fx::gltf::BufferView::TargetType::ArrayBuffer);
130 }
131
132 // add texcoord attribute
133 if (!mesh.texcoords.empty()) {
134 // NOTE: GLTF doesn't support storing multiple texture coordinates per vertex, so instead of
135 // using the standard GLTF texture arrays we store the data in a custom location
136 auto& def = addMomentumExtension(prim.extensionsAndExtras);
137 def["texcoords"] = createAccessorBuffer<const Vector2f>(model, mesh.texcoords, true);
138 setBufferView(model, def["texcoords"], fx::gltf::BufferView::TargetType::ArrayBuffer);
139
140 MT_THROW_IF(mesh.texcoord_faces.empty(), "Mesh has texcoords but no texcoord_faces");
141 const std::span<const int32_t> texFaces(
142 &mesh.texcoord_faces[0][0], mesh.texcoord_faces.size() * 3);
143 def["texfaces"] = createAccessorBuffer<const int32_t>(model, texFaces);
144 setBufferView(model, def["texfaces"], fx::gltf::BufferView::TargetType::ArrayBuffer);
145 }
146
147 if (mesh.colors.size() == mesh.vertices.size() && addColor) {
148 prim.attributes["COLOR_0"] =
149 createAccessorBuffer<const Vector3b>(model, mesh.colors, true, true);
150 setBufferView(model, prim.attributes["COLOR_0"], fx::gltf::BufferView::TargetType::ArrayBuffer);
151 }
152
153 return model.meshes.size() - 1;
154}

156void addMorphTargetsToModel(
157 fx::gltf::Document& model,
158 const BlendShape& shapes,
159 size_t numShapes,
160 size_t meshIndex) {
161 if (model.meshes.empty() || meshIndex == kInvalidIndex) {
162 MT_LOGW("No mesh is saved in the file when adding blendshapes");
163 return;
164 }
165
166 // assume there is only one primitive on the character mesh.
167 auto& prim = model.meshes.at(model.nodes.at(meshIndex).mesh).primitives.at(0);
168 size_t numShapesAdded = prim.targets.size();
169 if (numShapesAdded >= numShapes) {
170 // enough shapes already exist; nothing to be done.
171 return;
172 }
173 auto& def = addMomentumExtension(prim.extensionsAndExtras);
174 def["shapeNames"] = shapes.getShapeNames();
175
176 const MatrixXf& shapeVecs = shapes.getShapeVectors();
177 for (size_t iShape = numShapesAdded; iShape < numShapes; ++iShape) {
178 // make a copy so we can scale it
179 VectorXf deltas = shapeVecs.col(iShape);
180 fromMomentumVec3f(deltas);
181 prim.targets.emplace_back();
182 auto& attr = prim.targets.back();
183
184 std::span<const Eigen::Vector3f> span(
185 reinterpret_cast<Eigen::Vector3f*>(deltas.data()), static_cast<size_t>(deltas.size() / 3));
186 attr["POSITION"] = createAccessorBuffer<const Vector3f>(model, span, true);
187 setBufferView(model, attr["POSITION"], fx::gltf::BufferView::TargetType::ArrayBuffer);
188
189 // set bbox
190 Box3f bbox;
191 for (const Vector3f& v : span) {
192 bbox.extend(v);
193 }
194 for (size_t d = 0; d < 3; ++d) {
195 model.accessors.at(attr["POSITION"]).min.emplace_back(bbox.min()[d]);
196 model.accessors.at(attr["POSITION"]).max.emplace_back(bbox.max()[d]);
197 }
198 }

201void addMorphWeightsToModel(
202 fx::gltf::Document& model,
203 const Character& character,
204 const float fps,
205 const MatrixXf& modelParams,
206 const size_t meshIndex,
207 const std::string& motionName = "default") {
208 const size_t numFrames = modelParams.cols();
209 if (numFrames == 0) {
210 return;
211 }
212
213 const auto& pt = character.parameterTransform;
214 const size_t numWeights = pt.numBlendShapeParameters();
215 if (numWeights == 0) {
216 return;
217 }
218
219 // validate the mesh node - bounds checking
220 MT_THROW_IF(
221 meshIndex >= model.nodes.size() || model.nodes[meshIndex].mesh < 0 ||
222 static_cast<size_t>(model.nodes[meshIndex].mesh) >= model.meshes.size(),
223 "Invalid mesh index for blendshape animation (meshIndex: {}, nodes size: {})",
224 meshIndex,
225 model.nodes.size());
226
227 // validate the mesh node - original logic
228 if (model.meshes[model.nodes[meshIndex].mesh].primitives.size() != 1 ||
229 model.meshes[model.nodes[meshIndex].mesh].primitives.at(0).targets.size() != numWeights) {
230 MT_LOGW("No valid mesh to add blendshape animation for");
231 return;
232 }
233
234 // extract weight values
235 std::vector<float> timestamps(numFrames, 0);
236 bool useChannel = false;
237 // Important to make sure row vs col is what createSampler expects
238 MatrixXf weights(numWeights, numFrames);
239
240 for (size_t iFrame = 0; iFrame < numFrames; ++iFrame) {
241 const VectorXf& pose = modelParams.col(iFrame);
242 const VectorXf w = extractBlendWeights(pt, ModelParameters(pose)).v;
243 if (w.norm() > 1e-5f) {
244 useChannel = true;
245 }
246 weights.col(iFrame) = w;
247 timestamps[iFrame] = static_cast<float>(iFrame) / fps;
248 }
249 if (!useChannel) {
250 return;
251 }
252
253 // store weight values in gltf document
254 auto& animation = addMomentumAnimationToModel(model, motionName);
255 const auto timestampIdx = createAccessorBuffer<const float>(model, timestamps);
256 if (!timestamps.empty() && !model.accessors.empty() && timestampIdx < model.accessors.size()) {
257 model.accessors[timestampIdx].min.emplace_back(timestamps.front());
258 model.accessors[timestampIdx].max.emplace_back(timestamps.back());
259 }
260
261 animation.channels.emplace_back();
262 auto& channel = animation.channels.back();
263 channel.sampler = createSampler<const float>(
264 model,
265 animation,
266 std::span<const float>(weights.data(), numWeights * numFrames),
267 timestampIdx);
268 channel.target.node = static_cast<int32_t>(meshIndex);
269 channel.target.path = "weights";
270}

302void addActorAnimationToModel(
303 fx::gltf::Document& model,
304 const float fps,
305 std::span<const std::vector<momentum::Marker>> markerSequence,
306 const GltfBuilder::MarkerMesh markerMesh,
307 const std::string& animName) {
308 if (markerSequence.empty()) {
309 return;
310 }
311
312 // create arrays for data
313 const float kLowConfidence = 1e-3;
314 std::map<std::string, size_t> markerNameToIndex;
315 std::vector<std::string> markerNames;
316 std::vector<std::vector<float>> timestamps;
317 std::vector<std::vector<Vector3f>> markerPositions;
318 std::vector<std::vector<Vector3f>> markerConf;
319 for (size_t i = 0; i < markerSequence.size(); i++) {
320 const float timestamp = static_cast<float>(i) / fps;
321 for (const auto& marker : markerSequence[i]) {
322 if (marker.occluded || marker.confidence < kLowConfidence) {
323 continue;
324 }
325
326 // create new array if marker is unknown
327 if (markerNameToIndex.count(marker.name) == 0) {
328 const auto index = timestamps.size();
329 timestamps.emplace_back();
330 markerPositions.emplace_back();
331 markerConf.emplace_back();
332 markerNameToIndex[marker.name] = index;
333 markerNames.emplace_back(marker.name);
334 }
335
336 // get index
337 const auto& index = markerNameToIndex.at(marker.name);
338 MT_THROW_IF(
339 index >= timestamps.size() || index >= markerPositions.size() ||
340 index >= markerConf.size(),
341 "Marker index {} exceeds container size",
342 index);
343 timestamps[index].push_back(timestamp);
344 markerPositions[index].push_back(fromMomentumVec3f(marker.pos.cast<float>()));
345 markerConf[index].push_back(Vector3f::Ones() * marker.confidence);
346 }
347 }
348
349 // add data to gltf document
350 auto& scene = getDefaultScene(model);
351 auto& animation = addMomentumAnimationToModel(model, animName);
352
353 MT_CHECK(
354 (markerNames.size() == timestamps.size()) && (markerNames.size() == markerPositions.size()) &&
355 (markerNames.size() == markerNameToIndex.size()) &&
356 (markerNames.size() == markerConf.size()));
357
358 for (size_t j = 0; j < markerConf.size(); j++) {
359 // create marker
360 const auto nodeIndex = static_cast<int32_t>(model.nodes.size());
361 model.nodes.emplace_back();
362 auto& node = model.nodes.back();
363 scene.nodes.push_back(nodeIndex);
364 node.name = markerNames[j];
365 node.translation = {0.0f, 0.0f, 0.0f};
366 node.scale = {1.0f, 1.0f, 1.0f};
367 node.extensionsAndExtras["extensions"]["FB_momentum"]["type"] = "marker";
368 // for convenience, save confidence as a flat float array so we don't need to parse animation
369 // channels
370 {
371 std::vector<float> confValues;
372 confValues.reserve(markerConf[j].size());
373 for (const auto& v : markerConf[j]) {
374 confValues.push_back(v[0]);
375 }
376 node.extensionsAndExtras["extensions"]["FB_momentum"]["confidence"] = confValues;
377 }
378 switch (markerMesh) {
379 case GltfBuilder::MarkerMesh::UnitCube: {
380 auto newMeshIdx = addMeshToModel(model, kUnitCubeRed, true);
381 if (newMeshIdx < model.meshes.size()) {
382 node.mesh = static_cast<int32_t>(newMeshIdx);
383 }
384 break;
385 }
386 case GltfBuilder::MarkerMesh::None:
387 break;
388 }
389
390 // create animation channel
391 const auto timestampIdx = createAccessorBuffer<const float>(model, timestamps[j]);
392 if (!timestamps[j].empty() && !model.accessors.empty() &&
393 timestampIdx < model.accessors.size()) {
394 model.accessors[timestampIdx].min.emplace_back(timestamps[j].front());
395 model.accessors[timestampIdx].max.emplace_back(timestamps[j].back());
396 }
397
398 // translation
399 animation.channels.emplace_back();
400 auto& tchannel = animation.channels.back();
401 tchannel.sampler =
402 // NOLINTNEXTLINE(facebook-hte-ParameterUncheckedArrayBounds)
403 createSampler<const Vector3f>(model, animation, markerPositions[j], timestampIdx);
404 tchannel.target.node = nodeIndex;
405 tchannel.target.path = "translation";
406
407 // scale -- for visualization purpose only
408 animation.channels.emplace_back();
409 auto& schannel = animation.channels.back();
410 schannel.sampler = createSampler<const Vector3f>(model, animation, markerConf[j], timestampIdx);
411 schannel.target.node = nodeIndex;
412 schannel.target.path = "scale";
413 }
414}

416size_t addMeshToModel(
417 fx::gltf::Document& model,
418 const Character& character,
419 const std::vector<size_t>& jointToNodeMap,
420 const size_t rootNodeIdx) {
421 // write mesh node/skinning data
422 if (character.mesh && character.skinWeights) {
423 // create new mesh node
424 const auto nodeIdx = model.nodes.size();
425 model.nodes.emplace_back();
426 auto& node = model.nodes.back();
427 MT_THROW_IF(
428 rootNodeIdx >= model.nodes.size(),
429 "Root node index {} exceeds model.nodes size {}",
430 rootNodeIdx,
431 model.nodes.size());
432 model.nodes[rootNodeIdx].children.push_back(static_cast<int32_t>(nodeIdx));
433
434 // create model mesh
435 const auto& mesh = *character.mesh;
436 auto newMeshIdx = addMeshToModel(model, mesh);
437 if (newMeshIdx >= model.meshes.size()) {
438 // Mesh not valid, skip
439 return kInvalidIndex;
440 }
441 node.mesh = static_cast<int32_t>(newMeshIdx);
442 auto& m = model.meshes[node.mesh];
443 MT_CHECK(m.primitives.size() == 1);
444 auto& prim = m.primitives.back();
445
446 const auto& skin = *character.skinWeights;
447 model.skins.emplace_back();
448 auto& sk = model.skins.back();
449
450 // add skin node
451 for (const auto& ji : jointToNodeMap) {
452 sk.joints.push_back(static_cast<uint32_t>(ji));
453 }
454 sk.skeleton = sk.joints[0];
455
456 node.name = "mesh";
457 node.skin = static_cast<int32_t>(model.skins.size() - 1);
458
459 // add inverse bind matrix
460 std::vector<Matrix4f> ibm;
461 for (const auto& mat : character.inverseBindPose) {
462 auto data = mat;
463 data.translation() = fromMomentumVec3f(data.translation());
464 ibm.emplace_back(data.matrix());
465 }
466 sk.inverseBindMatrices = createAccessorBuffer<const Matrix4f>(model, ibm);
467
468 // add skinning data
469 int weightSets = 1;
470 if (skin.weight.block(0, 4, skin.weight.rows(), 4).maxCoeff() > 0.001f) {
471 // only export two sets of weights if more than one set is being used
472 // only one vertex on the second set would have to have more than 0.1% influence to qualify
473 weightSets = 2;
474 }
475 for (size_t b = 0; b < weightSets; b++) {
476 std::vector<Vector4s> indices(skin.index.rows());
477 std::vector<Vector4f> weights(skin.index.rows());
478 for (size_t i = 0; i < indices.size(); i++) {
479 indices[i] = Vector4s(
480 skin.index(i, b * 4 + 0),
481 skin.index(i, b * 4 + 1),
482 skin.index(i, b * 4 + 2),
483 skin.index(i, b * 4 + 3));
484 // NOLINTNEXTLINE(facebook-hte-LocalUncheckedArrayBounds)
485 weights[i] = Vector4f(
486 skin.weight(i, b * 4 + 0),
487 skin.weight(i, b * 4 + 1),
488 skin.weight(i, b * 4 + 2),
489 skin.weight(i, b * 4 + 3));
490 }
491
492 const auto kIndicesAttribute = std::string("JOINTS_") + std::to_string(b);
493 prim.attributes[kIndicesAttribute] = createAccessorBuffer<const Vector4s>(model, indices);
494 setBufferView(
495 model, prim.attributes[kIndicesAttribute], fx::gltf::BufferView::TargetType::ArrayBuffer);
496
497 const auto kWeightsAttribute = std::string("WEIGHTS_") + std::to_string(b);
498 prim.attributes[kWeightsAttribute] = createAccessorBuffer<const Vector4f>(model, weights);
499 setBufferView(
500 model, prim.attributes[kWeightsAttribute], fx::gltf::BufferView::TargetType::ArrayBuffer);
501 }
502 return nodeIdx;
503 }
504 return kInvalidIndex;
505}

507void addSkeletonStatesToModel(
508 fx::gltf::Document& model,
509 const Character& character,
510 const float fps,
511 std::span<const SkeletonState> skeletonStates,
512 std::span<const size_t> jointToNodeMap,
513 const std::string& motionName = "default") {
514 const auto numFrames = skeletonStates.size();
515 if (numFrames == 0) {
516 return;
517 }
518
519 // check for valid character and stuff before adding things
520 MT_THROW_IF(
521 character.skeleton.joints.empty() || character.parameterTransform.numJointParameters() == 0,
522 "Character is not valid when trying to add skeleton states to model");
523
524 // store parameterized motion into joints for actual animation
525 const auto numJoints = character.skeleton.joints.size();
526 std::vector<float> timestamps(numFrames);
527 std::vector<Vector3<bool>> useChannel(numJoints, Vector3<bool>::Constant(false));
528 std::vector<std::vector<Vector3f>> translation(numJoints, std::vector<Vector3f>(numFrames));
529 std::vector<std::vector<Vector4f>> rotation(numJoints, std::vector<Vector4f>(numFrames));
530 std::vector<std::vector<Vector3f>> scale(numJoints, std::vector<Vector3f>(numFrames));
531
532 // create motion data using the mapped data
533 bool foundAnyChannel = false;
534 for (Eigen::Index i = 0; i < static_cast<Eigen::Index>(numFrames); i++) {
535 for (size_t j = 0; j < numJoints; j++) {
536 auto localT = skeletonStates[i].jointState[j].localTranslation().cast<float>();
537 if (!(localT - character.skeleton.joints[j].translationOffset).isZero(1e-5f)) {
538 useChannel[j][0] = true;
539 foundAnyChannel = true;
540 }
541 translation[j][i] = fromMomentumVec3f(localT);
542
543 auto localR = skeletonStates[i].jointState[j].localRotation().cast<float>();
544 rotation[j][i] = {localR.x(), localR.y(), localR.z(), localR.w()};
545 if (!(localR.toRotationMatrix() - character.skeleton.joints[j].preRotation.toRotationMatrix())
546 .isZero(1e-5f)) {
547 useChannel[j][1] = true;
548 foundAnyChannel = true;
549 }
550
551 float localScale = skeletonStates[i].jointState[j].localScale();
552 scale[j][i] = Vector3f::Ones() * localScale;
553 if (std::fabs(localScale - 1.0f) > 1e-5f) {
554 useChannel[j][2] = true;
555 foundAnyChannel = true;
556 }
557 }
558 timestamps[i] = static_cast<float>(i) / fps;
559 }
560
561 // fx-gltf has a bug where if you write an animation that has no actual
562 // animated channels, it throws an error "Required field not found : channels".
563 // So we'll just make sure there is at least one channel in this case.
564 if (!foundAnyChannel && !useChannel.empty()) {
565 useChannel[0][0] = true;
566 }
567
568 // add data to gltf document
569 auto& animation = addMomentumAnimationToModel(model, motionName);
570 const auto timestampIdx = createAccessorBuffer<const float>(model, timestamps);
571 if (!timestamps.empty() && !model.accessors.empty() && timestampIdx < model.accessors.size()) {
572 model.accessors[timestampIdx].min.emplace_back(timestamps.front());
573 model.accessors[timestampIdx].max.emplace_back(timestamps.back());
574 }
575
576 // only save the animation channels that actually have any useful data
577 for (size_t j = 0; j < numJoints; j++) {
578 if (useChannel[j][0]) {
579 animation.channels.emplace_back();
580 auto& tchannel = animation.channels.back();
581 tchannel.sampler =
582 createSampler<const Vector3f>(model, animation, translation[j], timestampIdx);
583 tchannel.target.node = static_cast<int32_t>(jointToNodeMap[j]);
584 tchannel.target.path = "translation";
585 }
586
587 if (useChannel[j][1]) {
588 animation.channels.emplace_back();
589 auto& rchannel = animation.channels.back();
590 rchannel.sampler = createSampler<const Vector4f>(model, animation, rotation[j], timestampIdx);
591 rchannel.target.node = static_cast<int32_t>(jointToNodeMap[j]);
592 rchannel.target.path = "rotation";

605void addMotionToModel(
606 fx::gltf::Document& model,
607 const Character& character,
608 const float fps,
609 const MotionParameters& inputMotion,
610 const IdentityParameters& inputIdentity,
611 const std::vector<size_t>& jointToNodeMap,
612 const size_t meshIndex,
613 const bool addExtension = true,
614 const std::string& motionName = "default") {
615 // disentangle input values for storage
616 const auto& parameterNames = std::get<0>(inputMotion);
617 const auto& motion = std::get<1>(inputMotion);
618 const auto& jointNames = std::get<0>(inputIdentity);
619 const auto& identity = std::get<1>(inputIdentity);
620
621 const auto numFrames = motion.cols();
622 if (numFrames == 0) {
623 return;
624 }
625
626 // **** Save animation to Momentum extension ****
627 // add motion if we have poses
628 bool fullAnimation = false;
629 if (addExtension) {
630 auto& def = addMomentumExtension(model.extensionsAndExtras);
631 // store motion array and offsets as metadata
632 def["motion"]["nframes"] = numFrames;
633 if (!parameterNames.empty() && motion.size() > 0) {
634 def["motion"]["parameterNames"] = parameterNames;
635 def["motion"]["poses"] = createAccessorBuffer<const float>(
636 model, std::span<const float>(motion.data(), motion.size()));
637 fullAnimation = true;
638 }
639
640 if (!jointNames.empty() && identity.v.size() > 0) {
641 def["motion"]["jointNames"] = jointNames;
642 def["motion"]["offsets"] = createAccessorBuffer<const float>(
643 model, std::span<const float>(identity.v.data(), identity.v.size()));
644 fullAnimation = true;
645 }
646 } else {
647 // There may be animation outside of the momentum machinery
648 fullAnimation = true;
649 }
650
651 // **** Save animation to gltf ****
652 // add blenshapes (that are used) if available
653 if (meshIndex != kInvalidIndex && character.blendShape != nullptr) {
654 addMorphTargetsToModel(
655 model,
656 *character.blendShape.get(),
657 character.parameterTransform.numBlendShapeParameters(),
658 meshIndex);
659
660 addMorphWeightsToModel(model, character, fps, motion, meshIndex, motionName);
661 }
662
663 // check for valid character and motion before adding joint animation
664 if (!fullAnimation || character.skeleton.joints.empty()) {
665 return;
666 }
667
668 // map data to character
669 const auto inputPoses = mapMotionToCharacter(inputMotion, character);
670 const auto inputOffset = mapIdentityToCharacter(inputIdentity, character);
671
672 // create skeleton states from model params
673 std::vector<momentum::SkeletonState> skeletonStates(numFrames);
674 CharacterParameters params;
675 params.offsets = inputOffset;
676 for (Eigen::Index i = 0; i < numFrames; i++) {
677 params.pose = inputPoses.col(i);
678 const CharacterState state(params, character, false, false, false);
679 skeletonStates[i] = state.skeletonState;
680 }
681
682 // add skeleton states to model
683 addSkeletonStatesToModel(model, character, fps, skeletonStates, jointToNodeMap, motionName);
684}

694std::vector<size_t> addSkeletonToModel(
695 fx::gltf::Document& model,
696 const Character& character,
697 const bool updateExtension = true,
698 size_t modelRootNodeIndex = kInvalidIndex) {
699 // add all joints to node list
700 std::vector<size_t> jointToNodeMap;
701
702 for (size_t i = 0; i < character.skeleton.joints.size(); i++) {
703 const auto& joint = character.skeleton.joints[i];
704
705 // create new node
706 const auto nodeIndex = appendNode(model, joint.name);
707 auto& node = model.nodes[nodeIndex];
708 if (i == 0) {
709 if (modelRootNodeIndex != kInvalidIndex && modelRootNodeIndex < model.nodes.size()) {
710 model.nodes[modelRootNodeIndex].children.push_back(static_cast<int32_t>(nodeIndex));
711 } else {
712 auto& scene = getDefaultScene(model);
713 scene.nodes.push_back(static_cast<uint32_t>(nodeIndex));
714 }
715 }
716
717 // set joint values in node
718 node.rotation = fromMomentumQuaternionf(joint.preRotation);
719 const auto translation = fromMomentumVec3f(joint.translationOffset);
720 node.translation = {translation[0], translation[1], translation[2]};
721
722 if (updateExtension) {
723 auto& extension = addMomentumExtension(node.extensionsAndExtras);
724 extension["type"] = std::string("skeleton_joint");
725 }
726
727 // add node as child to parent joint
728 if (joint.parent != kInvalidIndex) {
729 model.nodes.at(jointToNodeMap.at(joint.parent))
730 .children.push_back(static_cast<int32_t>(nodeIndex));
731 }
732
733 // add node to map
734 jointToNodeMap.push_back(nodeIndex);
735 }
736
737 return jointToNodeMap;
738}

740void addCollisionsToModel(
741 fx::gltf::Document& model,
742 const Character& character,
743 const std::vector<size_t>& jointToNodeMap,
744 bool addExtensions) {
745 // add all collision objects to node list
746 if (character.collision) {
747 for (size_t i = 0; i < character.collision->size(); i++) {
748 const auto& col = (*character.collision)[i];
749
750 MT_CHECK(col.parent != kInvalidIndex);
751
752 // create new node
753 const auto nodeIndex = model.nodes.size();
754 model.nodes.emplace_back();
755 auto& node = model.nodes.back();
756
757 // add node as child to parent joint
758 model.nodes.at(jointToNodeMap.at(col.parent))
759 .children.push_back(static_cast<int32_t>(nodeIndex));
760
761 // add values for the tapered capsule
762 node.name = character.skeleton.joints.at(col.parent).name + "_col";
763
764 node.rotation = fromMomentumQuaternionf(col.transformation.rotation);
765 const auto translation = fromMomentumVec3f(col.transformation.translation);
766 node.translation = {translation[0], translation[1], translation[2]};
767
768 // add extra properties
769 if (addExtensions) {
770 auto& extension = addMomentumExtension(node.extensionsAndExtras);
771 extension["type"] = "collision_capsule";
772 extension["length"] = col.length;
773 extension["radius"] = col.radius;
774 }
775 }
776 }

780 fx::gltf::Document& model,
781 const Character& character,
782 const std::vector<size_t>& jointToNodeMap,
783 bool addExtensions) {

788 // create new node

815 extension["attachedToSkin"] = loc.attachedToSkin;

817 if (loc.skinOffset != 0.0f) {
818 extension["skinOffset"] = loc.skinOffset;

829void saveDocument(
830 const filesystem::path& filename,
831 fx::gltf::Document& model,
832 const GltfFileFormat fileFormat) {
833 // save model
834 auto deducedFileFormat = fileFormat;
835 if (fileFormat == GltfFileFormat::Auto && filename.extension() == filesystem::path(".gltf")) {
836 deducedFileFormat = GltfFileFormat::Ascii;
837 }
838
839 try {
840 // Check if the parent directory of the file exists. If the parent directory does not exist
841 // and is not an empty string (which means the file is in the current working directory),
842 // create it. This is necessary because fx::gltf::Save throws an exception if the directory
843 // does not exist.
844 const filesystem::path parentDir = filename.parent_path();
845 if (!parentDir.empty() && !filesystem::exists(parentDir)) {
846 filesystem::create_directories(parentDir);
847 }
848
849 switch (deducedFileFormat) {
850 case GltfFileFormat::Ascii: {
851 // we can't embed binary if we writing to ascii format.
852 // so we need to provide exported with a uri name relative to json root.
853 filesystem::path new_filename = filename;
854 new_filename.replace_extension(filesystem::path("glbin"));
855 if (!model.buffers.empty()) {
856 model.buffers.front().uri = new_filename.filename().string();
857 }
858
859 MT_LOGE(
860 "There are some compatibility issues with .gltf files. Not all extension data might be exported correctly.");
861 fx::gltf::Save(model, filename.string(), false);
862 break;
863 }
864 case GltfFileFormat::Auto:
865 [[fallthrough]];
866 case GltfFileFormat::Binary:
867 fx::gltf::Save(model, filename.string(), true);
868 break;
869 }
870 } catch (const std::exception& e) {
871 MT_THROW("Failed to save file to: {}. Error: {}", filename.string(), e.what());
872 } catch (...) {
873 std::rethrow_exception(std::current_exception());
874 }
875
876 MT_THROW_IF(!filesystem::exists(filename), "Unable to save: {}", filename.string());
877}

913void GltfBuilder::addCharacter(
914 const Character& character,
915 const Vector3f& positionOffset /*= Vector3f::Zero()*/,
916 const Quaternionf& rotationOffset /*= Quaternionf::Identity()*/,
917 const FileSaveOptions& options) {
918 if (impl_->characterData.find(character.name) != impl_->characterData.end()) {
919 // Character already exist. Doesn't allow character with the same name to be saved.
920 // #TODO: proper warning
921 return;
922 }
923 auto& scene = getDefaultScene(impl_->document);
924
925 // Add character root node
926 const auto rootNodeIdx = appendNode(impl_->document, character.name);
927 MT_CHECK(
928 rootNodeIdx >= 0 && rootNodeIdx < impl_->document.nodes.size(),
929 "{}: {}",
930 rootNodeIdx,
931 impl_->document.nodes.size());
932 auto& node = impl_->document.nodes[rootNodeIdx];
933 const auto translation = fromMomentumVec3f(positionOffset);
934 node.translation = {translation[0], translation[1], translation[2]};
935 node.rotation = fromMomentumQuaternionf(rotationOffset);
936 scene.nodes.push_back(static_cast<uint32_t>(rootNodeIdx));
937 auto& characterData = impl_->characterData[character.name];
938 characterData.rootIndex = rootNodeIdx;
939
940 // fill with data
941 characterData.nodeMap =
942 addSkeletonToModel(impl_->document, character, options.extensions, rootNodeIdx);
943 if (options.collisions) {
944 addCollisionsToModel(impl_->document, character, characterData.nodeMap, options.extensions);
945 }
946 if (options.locators) {
947 addLocatorsToModel(impl_->document, character, characterData.nodeMap, options.extensions);
948 }
949 if (options.mesh) {
950 characterData.meshIndex =
951 addMeshToModel(impl_->document, character, characterData.nodeMap, rootNodeIdx);
952
953 // Add blendshapes if both addMesh and addBlendShapes are true
954 if (options.blendShapes && character.blendShape != nullptr) {
955 addMorphTargetsToModel(
956 impl_->document,
957 *character.blendShape,
958 character.blendShape->shapeSize(),
959 characterData.meshIndex);
960 }
961 }
962 if (options.extensions) {
963 auto& def = addMomentumExtension(impl_->document.extensionsAndExtras);
964 parameterTransformToJson(character, def["transform"]);
965 parameterLimitsToJson(character, def["parameterLimits"]);
966 parameterSetsToJson(character, def["parameterSet"]);
967 poseConstraintsToJson(character, def["poseConstraints"]);
968 // add metadata if it exists
969 if (!character.metadata.empty()) {
970 def["metadata"] = character.metadata;
971 }
972 }
973}
