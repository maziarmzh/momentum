20// SIMD mirrors of the scalar helpers in generalized_loss.cpp. Each takes a Packet of squared
21// errors and broadcasts the scalar `invC2` over the lanes via DrJit's overloaded operators.
22// Math intrinsics are dispatched through `drjit::sqrt/log/exp/pow` to vectorize across the packet
23// instead of falling back to per-lane `std::*` calls.
24
25// L2: returns invC2 * sqrError (no 0.5 factor — kept separate so the result can be reused as the
26// inner term of the L1/Cauchy/Welsch/General formulas).
27template <typename T>
28Packet<T> L2Loss(const Packet<T>& sqrError, T invC2) {
29 return sqrError * invC2;
30}
31
32// d/dsqrError of L2Loss is the constant invC2 — broadcast implicitly to a Packet at the call site.
33template <typename T>
34Packet<T> derivL2Loss(const Packet<T>& /* sqrError */, T invC2) {
35 return invC2;
36}
37
38// L1 (pseudo-Huber, alpha = 1): sqrt(invC2 * sqrError + 1) - 1.
39// The trailing `- 1` is intentionally kept so value() at sqrError=0 is exactly 0.
40template <typename T>
41Packet<T> L1Loss(const Packet<T>& sqrError, T invC2) {
42 return drjit::sqrt(L2Loss(sqrError, invC2) + T(1)) - T(1);
43}
44
45template <typename T>
46Packet<T> derivL1Loss(const Packet<T>& sqrError, T invC2) {
47 return T(0.5) * invC2 / drjit::sqrt(L2Loss(sqrError, invC2) + T(1));
48}
49
50// Cauchy / Lorentzian (alpha = 0): log(0.5 * invC2 * sqrError + 1).
51template <typename T>
52Packet<T> CauchyLoss(const Packet<T>& sqrError, T invC2) {
53 return drjit::log(T(0.5) * L2Loss(sqrError, invC2) + T(1));
54}
55
56template <typename T>
57Packet<T> derivCauchyLoss(const Packet<T>& sqrError, T invC2) {
58 return invC2 / (invC2 * sqrError + T(2));
59}
60
61// Welsch (alpha -> -inf limit): 1 - exp(-0.5 * invC2 * sqrError). Bounded in [0, 1).
62template <typename T>
63Packet<T> WelschLoss(const Packet<T>& sqrError, T invC2) {
64 return T(1) - drjit::exp(T(-0.5) * L2Loss(sqrError, invC2));
65}
66
67template <typename T>
68Packet<T> derivWelschLoss(const Packet<T>& sqrError, T invC2) {
69 return (T(0.5) * invC2) * drjit::exp(T(-0.5) * L2Loss(sqrError, invC2));
70}

75SimdGeneralizedLossT<T>::SimdGeneralizedLossT(const T& a, const T& c) : Base(a, c) {}

77// Branching is on the scalar `lossType_` cached at construction (Base classifies alpha into one
78// of L2/L1/Cauchy/Welsch/General). All packet lanes follow the same branch — no per-lane mask or
79// `drjit::select` is needed because alpha is uniform across the packet.
80template <typename T>
81Packet<T> SimdGeneralizedLossT<T>::value(const Packet<T>& sqrError) const {
82 switch (this->lossType_) {
83 case Base::LossType::L2:
84 return L2Loss(sqrError, this->invC2_);
85 case Base::LossType::L1:
86 return L1Loss(sqrError, this->invC2_);
87 case Base::LossType::Cauchy:
88 return CauchyLoss(sqrError, this->invC2_);
89 case Base::LossType::Welsch:
90 // Welsch limit is taken when alpha <= kWelsch (alpha == -inf is handled by the same branch
91 // since Base classifies any sufficiently negative alpha as Welsch).
92 return WelschLoss(sqrError, this->invC2_);
93 case Base::LossType::General:
94 // General Barron loss for alpha not near 0, 1, 2, or -inf. The packet `pow` is the slow
95 // path; the special cases above exist precisely to avoid it. The scalar prefactor
96 // `|alpha - 2| / alpha` is computed once outside the packet expression.
97 return (drjit::pow(
98 L2Loss(sqrError, this->invC2_) / std::fabs(this->alpha_ - T(2)) + T(1),
99 T(0.5) * this->alpha_) -
100 T(1)) *
101 (std::fabs(this->alpha_ - T(2)) / this->alpha_);
102 default: {
103 // TODO: returning a default-constructed Packet here may yield uninitialized lanes; consider
104 // returning drjit::zeros<Packet<T>>() or asserting in debug builds so callers don't silently
105 // propagate garbage on a corrupted lossType_.
106 MT_LOGE("Invalid lossType_ ({})", static_cast<int>(this->lossType_));
107 return {};
108 }
109 }
110}

112// Same lane-uniform switch structure as value(); see notes there. Returns d(loss)/d(sqrError) so
113// callers can chain-rule into the Jacobian of the underlying residual.
114template <typename T>
115Packet<T> SimdGeneralizedLossT<T>::deriv(const Packet<T>& sqrError) const {
116 switch (this->lossType_) {
117 case Base::LossType::L2:
118 return derivL2Loss(sqrError, this->invC2_);
119 case Base::LossType::L1:
120 return derivL1Loss(sqrError, this->invC2_);
121 case Base::LossType::Cauchy:
122 return derivCauchyLoss(sqrError, this->invC2_);
123 case Base::LossType::Welsch:
124 return derivWelschLoss(sqrError, this->invC2_);
125 case Base::LossType::General:
126 // General Barron derivative; uses packet `pow` and is the slow path.
127 return (T(0.5) * this->invC2_) *
128 drjit::pow(
129 L2Loss(sqrError, this->invC2_) / std::fabs(this->alpha_ - T(2)) + T(1),
130 T(0.5) * this->alpha_ - T(1));
131 default: {
132 // TODO: see value() — same uninitialized-lane concern on the default branch.
133 MT_LOGE("Invalid lossType_ ({})", static_cast<int>(this->lossType_));
134 return {};
135 }
136 }
