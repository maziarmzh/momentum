49template <typename T>
50void GaussNewtonSolverQRT<T>::doIteration() {
51 MT_PROFILE_FUNCTION();
52
53 // Initialize Jacobian computation (sets up SkeletonState, MeshState, etc.)
54 {
55 MT_PROFILE_EVENT("Initialize Jacobian computation");
56 this->solverFunction_->initializeJacobianComputation(this->parameters_);
57 }
58
59 const auto nFullParams = static_cast<Eigen::Index>(this->parameters_.size());
60
61 std::vector<Eigen::Index> enabledParameters;
62 enabledParameters.reserve(nFullParams);
63 for (int iFullParam = 0; iFullParam < nFullParams; ++iFullParam) {
64 if (this->activeParameters_.test(iFullParam)) {
65 enabledParameters.push_back(iFullParam);
66 }
67 }
68
69 const auto nSubsetParams = static_cast<int>(enabledParameters.size());
70
71 if (enabledParameters.empty()) {
72 return;
73 }
74
75 // momentum solves the problem (J^T*J + lambda*I) x = J^T*r;
76 // the QR solver wants the square root of that lambda.
77 qrSolver_.reset(nSubsetParams, std::sqrt(regularization_));
78
79 double error_orig = 0.0;
80 const size_t numBlocks = this->solverFunction_->getJacobianBlockCount();
81
82 for (size_t blockIdx = 0; blockIdx < numBlocks; ++blockIdx) {
83 const size_t blockSize = this->solverFunction_->getJacobianBlockSize(blockIdx);
84 if (blockSize == 0) {
85 continue;
86 }
87
88 const auto paddedBlockSize = padToSimdAlignment(blockSize);
89 const auto rows = static_cast<Eigen::Index>(paddedBlockSize);
90
91 jacobian_.resizeAndSetZero(rows, nFullParams);
92 residual_.resizeAndSetZero(rows);
93
94 size_t usedRows = 0;
95 error_orig += this->solverFunction_->computeJacobianBlock(
96 this->parameters_, blockIdx, jacobian_.mat(), residual_.mat(), usedRows);
97
98 if (usedRows == 0) {
99 continue;
100 }
101
102 qrSolver_.addMutating(
103 ColumnIndexedMatrix<Eigen::MatrixX<T>>(
104 jacobian_.mat().topRows(usedRows), enabledParameters),
105 residual_.mat().topRows(usedRows));
106 }
107
108 // Finalize Jacobian computation
109 {
110 MT_PROFILE_EVENT("Finalize Jacobian computation");
111 this->solverFunction_->finalizeJacobianComputation();
112 }
113
114 this->error_ = error_orig;
115
116 const Eigen::VectorX<T> subsetDelta = qrSolver_.result();
117
118 Eigen::VectorX<T> searchDir = Eigen::VectorX<T>::Zero(nFullParams);
119 for (int iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
120 const auto iFullParam = enabledParameters[iSubsetParam];
121 searchDir(iFullParam) = subsetDelta(iSubsetParam);
122 }
123
124 if (doLineSearch_) {
125 const double innerProd = -qrSolver_.At_times_b().dot(subsetDelta);
126
127 // Line search:
128 const float c_1 = 1e-4f;
129 const float tau = 0.5f;
130 float alpha = 1.0f;
131
132 const Eigen::VectorX<T> parameters_orig = this->parameters_;
133 for (size_t kStep = 0; kStep < 10 && std::fpclassify(alpha) == FP_NORMAL; ++kStep) {
134 // update the this->parameters_
135 this->parameters_ = parameters_orig;
136 this->solverFunction_->updateParameters(this->parameters_, alpha * searchDir);
137
138 const double error_new = this->solverFunction_->getError(this->parameters_);
139
140 if ((error_orig - error_new) >= c_1 * alpha * -innerProd) {
141 break;
142 }
143
144 // Reduce step size:
145 alpha = alpha * tau;
146 }
147 } else {
148 this->solverFunction_->updateParameters(this->parameters_, searchDir);
149 }
150}
