22template <typename ErrorFunctionType>
23size_t computeJacobianSize(const std::vector<std::shared_ptr<ErrorFunctionType>>& functions) {
24 size_t result = 0;
25 for (const auto& f : functions) {
26 if (f->getWeight() <= 0) {
27 continue;
28 }
29 result += f->getJacobianSize();
30 }
31 return result;
32}

34template <typename T>
35std::vector<Eigen::Index> buildSequenceColumnIndices(
36 const SequenceSolverFunctionT<T>* fn,
37 size_t bandwidth) {
38 std::vector<Eigen::Index> result;
39 result.reserve(bandwidth * fn->getPerFrameParameterIndices().size());
40 for (size_t iSubFrame = 0; iSubFrame < bandwidth; ++iSubFrame) {
41 std::transform(
42 fn->getPerFrameParameterIndices().begin(),
43 fn->getPerFrameParameterIndices().end(),
44 std::back_inserter(result),
45 [offset = iSubFrame * (*fn->getParameterTransform()).numAllModelParameters()](
46 Eigen::Index kParam) -> Eigen::Index { return kParam + offset; });
47 }
48
49 return result;
50}
