62template <typename T>
63typename FloorErrorFunctionT<T>::MinHeightResult FloorErrorFunctionT<T>::calculateMinHeight(
64 const MeshStateT<T>& meshState) const {
65 MT_PROFILE_FUNCTION();
66
67 MT_CHECK_NOTNULL(meshState.posedMesh_);
68
69 const size_t effectiveK = std::min(k_, vertexIndices_.size());
70
71 // Max heap for k lowest projections (top = largest of the k lowest)
72 using ProjectionPair = std::pair<T, size_t>; // (projection, vertex_index)
73 std::priority_queue<ProjectionPair, std::vector<ProjectionPair>, std::less<>> minQueue;
74
75 for (const auto vertexIdx : vertexIndices_) {
76 MT_CHECK(
77 vertexIdx < meshState.posedMesh_->vertices.size(),
78 "Vertex index {} out of range (mesh has {} vertices)",
79 vertexIdx,
80 meshState.posedMesh_->vertices.size());
81
82 const T projection = upDirection_.dot(meshState.posedMesh_->vertices[vertexIdx]);
83
84 if (minQueue.size() < effectiveK) {
85 minQueue.push({projection, vertexIdx});
86 } else if (projection < minQueue.top().first) {
87 minQueue.pop();
88 minQueue.push({projection, vertexIdx});
89 }
90 }
91
92 MinHeightResult result;
93 result.vertexIndices.reserve(effectiveK);
94 result.vertexWeights.reserve(effectiveK);
95
96 T projectionSum = T(0);
97 while (!minQueue.empty()) {
98 const auto& pair = minQueue.top();
99 result.vertexIndices.push_back(pair.second);
100 projectionSum += pair.first;
101 minQueue.pop();
102 }
103
104 const T weight = T(1) / T(effectiveK);
105 result.vertexWeights.resize(effectiveK, weight);
106 result.avgProjection = projectionSum / T(effectiveK);
107
108 return result;
109}

111template <typename T>
112double FloorErrorFunctionT<T>::getError(
113 const ModelParametersT<T>& /* modelParameters */,
114 const SkeletonStateT<T>& /* state */,
115 const MeshStateT<T>& meshState) {
116 MT_PROFILE_FUNCTION();
117
118 const auto minResult = calculateMinHeight(meshState);
119 const T diff = minResult.avgProjection - targetHeight_;
120
121 return this->weight_ * diff * diff;
122}

124template <typename T>
125double FloorErrorFunctionT<T>::getGradient(
126 const ModelParametersT<T>& /* modelParameters */,
127 const SkeletonStateT<T>& state,
128 const MeshStateT<T>& meshState,
129 Eigen::Ref<Eigen::VectorX<T>> gradient) {
130 MT_PROFILE_FUNCTION();
131
132 const auto minResult = calculateMinHeight(meshState);
133 const T diff = minResult.avgProjection - targetHeight_;
134
135 const T wgt = T(2) * diff * this->weight_;
136
137 for (size_t i = 0; i < minResult.vertexIndices.size(); ++i) {
138 calculateVertexGradient(
139 minResult.vertexIndices[i],
140 state,
141 meshState,
142 wgt * minResult.vertexWeights[i] * upDirection_,
143 gradient);
144 }
145
146 return this->weight_ * diff * diff;
147}

149template <typename T>
150double FloorErrorFunctionT<T>::getJacobian(
151 const ModelParametersT<T>& /* modelParameters */,
152 const SkeletonStateT<T>& state,
153 const MeshStateT<T>& meshState,
154 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
155 Eigen::Ref<Eigen::VectorX<T>> residual,
156 int& usedRows) {
157 MT_PROFILE_FUNCTION();
158
159 MT_CHECK(
160 jacobian.cols() == static_cast<Eigen::Index>(this->parameterTransform_.transform.cols()));
161 MT_CHECK(jacobian.rows() >= 1);
162 MT_CHECK(residual.rows() >= 1);
163
164 const auto minResult = calculateMinHeight(meshState);
165 const T diff = minResult.avgProjection - targetHeight_;
166
167 const T wgt = std::sqrt(this->weight_);
168 residual(0) = wgt * diff;
169
170 jacobian.row(0).setZero();
171 auto jac_row = jacobian.row(0);
172
173 for (size_t i = 0; i < minResult.vertexIndices.size(); ++i) {
174 calculateVertexJacobian(
175 minResult.vertexIndices[i],
176 state,
177 meshState,
178 wgt * minResult.vertexWeights[i] * upDirection_,
179 jac_row);
180 }
181
182 usedRows = 1;
183 return wgt * wgt * diff * diff;
184}

191template <typename T>
192template <typename Derived>
193void FloorErrorFunctionT<T>::calculateVertexJacobian(
194 size_t vertexIndex,
195 const SkeletonStateT<T>& state,
196 const MeshStateT<T>& meshState,
197 const Eigen::Vector3<T>& jacobianDirection,
198 Eigen::MatrixBase<Derived>& jacobian) const {
199 MT_PROFILE_FUNCTION();
200
201 MT_CHECK_NOTNULL(meshState.restMesh_);
202
203 SkinningWeightIteratorT<T> skinningIter(
204 this->character_, *meshState.restMesh_, state, vertexIndex);
205
206 while (!skinningIter.finished()) {
207 size_t jointIndex = 0;
208 T boneWeight;
209 Eigen::Vector3<T> pos;
210 std::tie(jointIndex, boneWeight, pos) = skinningIter.next();
211
212 MT_CHECK(jointIndex < this->skeleton_.joints.size());
213
214 const auto& jointState = state.jointState[jointIndex];
215 const size_t paramIndex = jointIndex * kParametersPerJoint;
216 const Eigen::Vector3<T> posd = pos - jointState.translation();
217
218 for (size_t d = 0; d < 3; d++) {
219 if (this->activeJointParams_[paramIndex + d]) {
220 jacobian_jointParams_to_modelParams<T>(
221 boneWeight * jacobianDirection.dot(jointState.getTranslationDerivative(d)),
222 paramIndex + d,
223 this->parameterTransform_,
224 jacobian);
225 }
226 if (this->activeJointParams_[paramIndex + 3 + d]) {
227 jacobian_jointParams_to_modelParams<T>(
228 boneWeight * jacobianDirection.dot(jointState.getRotationDerivative(d, posd)),
229 paramIndex + 3 + d,
230 this->parameterTransform_,
231 jacobian);
232 }
233 }
234 if (this->activeJointParams_[paramIndex + 6]) {
235 jacobian_jointParams_to_modelParams<T>(
236 boneWeight * jacobianDirection.dot(jointState.getScaleDerivative(posd)),
237 paramIndex + 6,
238 this->parameterTransform_,
239 jacobian);
240 }
241 }
