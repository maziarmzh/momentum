72template <typename T>
73double JointToJointOrientationErrorFunctionT<T>::getError(
74 const ModelParametersT<T>& /* params */,
75 const SkeletonStateT<T>& state,
76 const MeshStateT<T>& /* meshState */) {
77 if (state.jointState.empty() || constraints_.empty()) {
78 return 0.0;
79 }
80
81 double error = 0.0;
82
83 for (const auto& constr : constraints_) {
84 const auto& srcJointState = state.jointState[constr.sourceJoint];
85 const auto& refJointState = state.jointState[constr.referenceJoint];
86
87 // Compute relative rotation: R_rel = R_ref^T * R_src
88 const Matrix3<T> srcRot = srcJointState.rotation().toRotationMatrix();
89 const Matrix3<T> refRot = refJointState.rotation().toRotationMatrix();
90 const Matrix3<T> relativeRot = refRot.transpose() * srcRot;
91
92 // Compute error: difference between relative rotation and target rotation
93 const Matrix3<T> targetRot = constr.target.toRotationMatrix();
94 const Matrix3<T> rotError = relativeRot - targetRot;
95
96 error += constr.weight * rotError.squaredNorm() * this->weight_;
97 }
98
99 return error;
100}

102template <typename T>
103double JointToJointOrientationErrorFunctionT<T>::getGradient(
104 const ModelParametersT<T>& /* params */,
105 const SkeletonStateT<T>& state,
106 const MeshStateT<T>& /* meshState */,
107 Ref<VectorX<T>> gradient) {
108 MT_PROFILE_FUNCTION();
109
110 if (state.jointState.empty() || constraints_.empty()) {
111 return 0.0;
112 }
113
114 double error = 0.0;
115
116 for (const auto& constr : constraints_) {
117 const auto& srcJointState = state.jointState[constr.sourceJoint];
118 const auto& refJointState = state.jointState[constr.referenceJoint];
119
120 // Compute relative rotation: R_rel = R_ref^T * R_src
121 const Matrix3<T> srcRot = srcJointState.rotation().toRotationMatrix();
122 const Matrix3<T> refRot = refJointState.rotation().toRotationMatrix();
123 const Matrix3<T> relativeRot = refRot.transpose() * srcRot;
124
125 // Compute error
126 const Matrix3<T> targetRot = constr.target.toRotationMatrix();
127 const Matrix3<T> rotError = relativeRot - targetRot;
128 error += constr.weight * rotError.squaredNorm() * this->weight_;
129
130 const T wgt = T(2) * constr.weight * this->weight_;
131
132 // The error is ||R_ref^T * R_src - R_target||_F^2
133 // d(error)/dq = 2 * sum_ij (R_rel - R_target)_ij * d(R_rel)_ij/dq
134 //
135 // R_rel = R_ref^T * R_src, so each column of R_rel is:
136 // R_rel[:,k] = R_ref^T * R_src[:,k]
137 //
138 // This has the same structure as the position case but with 3 columns
139 // of the rotation matrix instead of a single position vector.
140
141 // Process source joint chain: d(R_ref^T * R_src)/d(src params)
142 // = R_ref^T * d(R_src)/d(src params)
143 // For each column k of R_src, the derivative with respect to rotation parameters
144 // gives cross product with the column vector.
145 for (int col = 0; col < 3; col++) {
146 const Vector3<T> srcCol = srcRot.col(col);
147 const Vector3<T> errCol = rotError.col(col);
148 // worldGrad = R_ref * errCol transforms back to world frame
149 const Vector3<T> worldGrad = refRot * errCol;
150
151 // Process source joint chain (positive contribution)
152 size_t currentJoint = constr.sourceJoint;
153 while (currentJoint != kInvalidIndex) {
154 const auto& currentJointState = state.jointState[currentJoint];
155 const size_t paramIndex = currentJoint * kParametersPerJoint;
156
157 // Only rotation derivatives affect the rotation matrix columns
158 for (size_t d = 0; d < 3; d++) {
159 if (this->activeJointParams_[paramIndex + 3 + d]) {
160 const T val = worldGrad.dot(currentJointState.getRotationDerivative(d, srcCol)) * wgt;
161 for (auto index =
162 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
163 index <
164 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d + 1];
165 ++index) {
166 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
167 val * this->parameterTransform_.transform.valuePtr()[index];
168 }
169 }
170 }
171
172 currentJoint = this->skeleton_.joints[currentJoint].parent;
173 }
174
175 // Process reference joint chain: d(R_ref^T * R_src[:,col])/d(ref params)
176 // This has two parts:
177 // 1. d(R_ref^T)/d(ref params) * R_src[:,col] (rotation of reference frame)
178 currentJoint = constr.referenceJoint;
179 while (currentJoint != kInvalidIndex) {
180 const auto& currentJointState = state.jointState[currentJoint];
181 const size_t paramIndex = currentJoint * kParametersPerJoint;
182
183 for (size_t d = 0; d < 3; d++) {
184 if (this->activeJointParams_[paramIndex + 3 + d]) {
185 // d(R_ref^T)/d(rot_i) * srcCol = -R_ref^T * [axis_i]_x * srcCol
186 // The gradient contribution is: errCol^T * (-R_ref^T * [axis_i]_x * srcCol)
187 // = -worldGrad^T * getRotationDerivative(d, srcCol)
188 const T val = -worldGrad.dot(currentJointState.getRotationDerivative(d, srcCol)) * wgt;
189 for (auto index =
190 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
191 index <
192 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d + 1];
193 ++index) {
194 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
195 val * this->parameterTransform_.transform.valuePtr()[index];
196 }
197 }
198 }
199
200 currentJoint = this->skeleton_.joints[currentJoint].parent;
201 }
202 }
203 }
204
205 return error;
206}

208template <typename T>
209double JointToJointOrientationErrorFunctionT<T>::getJacobian(
210 const ModelParametersT<T>& /* params */,
211 const SkeletonStateT<T>& state,
212 const MeshStateT<T>& /* meshState */,
213 Ref<MatrixX<T>> jacobian,
214 Ref<VectorX<T>> residual,
215 int& usedRows) {
216 MT_PROFILE_FUNCTION();
217
218 MT_CHECK(
219 jacobian.rows() >= gsl::narrow<Eigen::Index>(9 * constraints_.size()),
220 "Jacobian rows mismatch: Actual {}, Expected at least {}",
221 jacobian.rows(),
222 9 * constraints_.size());
223
224 MT_CHECK(
225 residual.rows() >= gsl::narrow<Eigen::Index>(9 * constraints_.size()),
226 "Residual rows mismatch: Actual {}, Expected at least {}",
227 residual.rows(),
228 9 * constraints_.size());
229
230 if (state.jointState.empty() || constraints_.empty()) {
231 usedRows = 0;
232 return 0.0;
233 }
234
235 const T wgt = std::sqrt(this->weight_);
236 int pos = 0;
237 double error = 0.0;
238
239 for (const auto& constr : constraints_) {
240 const auto& srcJointState = state.jointState[constr.sourceJoint];
241 const auto& refJointState = state.jointState[constr.referenceJoint];
242
243 // Compute relative rotation
244 const Matrix3<T> srcRot = srcJointState.rotation().toRotationMatrix();
245 const Matrix3<T> refRot = refJointState.rotation().toRotationMatrix();
246 const Matrix3<T> refRotT = refRot.transpose();
247 const Matrix3<T> relativeRot = refRotT * srcRot;
248
249 // Compute error
250 const Matrix3<T> targetRot = constr.target.toRotationMatrix();
251 const Matrix3<T> rotError = relativeRot - targetRot;
252 error += constr.weight * rotError.squaredNorm() * this->weight_;
253
254 const T fac = std::sqrt(constr.weight) * wgt;
255 const int baseRow = pos;
256 pos += 9;
257
258 // Set residual: 9 elements from the rotation error matrix (column-major)
259 for (int col = 0; col < 3; col++) {
260 for (int row = 0; row < 3; row++) {
261 residual(baseRow + col * 3 + row) = rotError(row, col) * fac;
262 }
263 }
264
265 // Jacobian: for each column of the rotation matrix
266 for (int col = 0; col < 3; col++) {
267 const Vector3<T> srcCol = srcRot.col(col);
268
269 // Source joint chain: d(R_ref^T * R_src[:,col])/d(src params)
270 // = R_ref^T * d(R_src[:,col])/d(src params)
271 size_t currentJoint = constr.sourceJoint;
272 while (currentJoint != kInvalidIndex) {
273 const auto& currentJointState = state.jointState[currentJoint];
274 const size_t paramIndex = currentJoint * kParametersPerJoint;
275
276 for (size_t d = 0; d < 3; d++) {
277 if (this->activeJointParams_[paramIndex + 3 + d]) {
278 // d(R_src[:,col])/d(rot_i) = getRotationDerivative(d, srcCol)
279 const Vector3<T> dWorldCol = currentJointState.getRotationDerivative(d, srcCol);
280 const Vector3<T> dRelativeCol = refRotT * dWorldCol;
281 for (auto index =
282 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
283 index <
284 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d + 1];
285 ++index) {
286 const auto paramCol = this->parameterTransform_.transform.innerIndexPtr()[index];
287 const T coeff = this->parameterTransform_.transform.valuePtr()[index] * fac;
288 for (int row = 0; row < 3; row++) {
289 jacobian(baseRow + col * 3 + row, paramCol) += dRelativeCol(row) * coeff;
290 }
291 }
292 }
293 }
294
295 currentJoint = this->skeleton_.joints[currentJoint].parent;
296 }
297
298 // Reference joint chain: d(R_ref^T * srcCol)/d(ref params)
299 // = d(R_ref^T)/d(ref params) * srcCol
300 currentJoint = constr.referenceJoint;
301 while (currentJoint != kInvalidIndex) {
302 const auto& currentJointState = state.jointState[currentJoint];
303 const size_t paramIndex = currentJoint * kParametersPerJoint;
304
305 for (size_t d = 0; d < 3; d++) {
306 if (this->activeJointParams_[paramIndex + 3 + d]) {
307 // d(R_ref^T)/d(rot_i) * srcCol = -R_ref^T * [axis_i]_x * srcCol
308 const Vector3<T> dRotCol = currentJointState.getRotationDerivative(d, srcCol);
309 const Vector3<T> dRelativeCol = -refRotT * dRotCol;
310 for (auto index =
311 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
312 index <
313 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d + 1];
314 ++index) {
315 const auto paramCol = this->parameterTransform_.transform.innerIndexPtr()[index];
316 const T coeff = this->parameterTransform_.transform.valuePtr()[index] * fac;
317 for (int row = 0; row < 3; row++) {
318 jacobian(baseRow + col * 3 + row, paramCol) += dRelativeCol(row) * coeff;
319 }
320 }
321 }
322 }
323
324 currentJoint = this->skeleton_.joints[currentJoint].parent;
325 }
326 }
327 }
328
329 usedRows = pos;
330 return error;
331}
