108template <typename T>
109bool overlaps(
110 const Vector3<T>& originA,
111 const Vector3<T>& directionA,
112 const Vector2<T>& radiiA,
113 T deltaA,
114 const Vector3<T>& originB,
115 const Vector3<T>& directionB,
116 const Vector2<T>& radiiB,
117 T deltaB,
118 T& outDistance,
119 Vector2<T>& outClosestPoints,
120 T& outOverlap) {
121 // Use the sum of max radii as a proximity cutoff so closestPointsOnSegments can early-out
122 // when segments are guaranteed too far apart for the capsules to overlap.
123 const T maxRadiiSum = radiiA.maxCoeff() + radiiB.maxCoeff();
124
125 auto [success, closestDist, closestPoints] =
126 closestPointsOnSegments<T>(originA, directionA, originB, directionB, maxRadiiSum);
127
128 if (!success) {
129 return false;
130 }
131
132 outClosestPoints = closestPoints;
133
134 // Linearly interpolate each tapered radius at the closest-point parameter along its segment,
135 // then sum to get the combined radius the gap must clear for the capsules to be disjoint.
136 const T radiusAtClosestPoints =
137 radiiA[0] + closestPoints[0] * deltaA + radiiB[0] + closestPoints[1] * deltaB;
138
139 outOverlap = radiusAtClosestPoints - closestDist;
140 outDistance = closestDist;
141
142 // Reject coincident centerlines (closestDist below Eps) to avoid degenerate contact normals
143 // downstream when the two segments are effectively the same line.
144 return (outOverlap > T(0)) && (closestDist >= Eps<T>(1e-8, 1e-17));
145}

148template <typename T>
149[[nodiscard]] T closestPointOnSegmentParameter(
150 const Vector3<T>& origin,
151 const Vector3<T>& direction,
152 const Vector3<T>& point) {
153 const T directionNormSq = direction.squaredNorm();
154 if (directionNormSq <= Eps<T>(1e-8, 1e-17)) {
155 return T(0);
156 }
157
158 return std::clamp((point - origin).dot(direction) / directionNormSq, T(0), T(1));
159}

162template <typename T>
163[[nodiscard]] T ellipsoidRadiusAlongDirection(
164 const Quaternion<T>& orientation,
165 const Vector3<T>& radii,
166 const Vector3<T>& unitDirection) {
167 const Vector3<T> localDirection = orientation.inverse() * unitDirection;
168 return radii.cwiseAbs().cwiseProduct(localDirection).norm();
169}

190template <typename T>
191[[nodiscard]] bool overlapsCapsuleEllipsoid(
192 const Vector3<T>& capsuleOrigin,
193 const Vector3<T>& capsuleDirection,
194 const Vector2<T>& capsuleRadii,
195 T capsuleDelta,
196 const Vector3<T>& ellipsoidCenter,
197 const Quaternion<T>& ellipsoidOrientation,
198 const Vector3<T>& ellipsoidRadii,
199 T& outDistance,
200 T& outCapsuleParam,
201 T& outOverlap,
202 T& outCapsuleRadius,
203 T& outEllipsoidRadius) {
204 outCapsuleParam =
205 closestPointOnSegmentParameter(capsuleOrigin, capsuleDirection, ellipsoidCenter);
206 const Vector3<T> capsulePoint = capsuleOrigin + capsuleDirection * outCapsuleParam;
207 const Vector3<T> separation = capsulePoint - ellipsoidCenter;
208 outDistance = separation.norm();
209 if (outDistance < Eps<T>(1e-8, 1e-17)) {
210 return false;
211 }
212
213 const Vector3<T> normal = separation / outDistance;
214 outCapsuleRadius = capsuleRadii[0] + outCapsuleParam * capsuleDelta;
215 outEllipsoidRadius = ellipsoidRadiusAlongDirection(ellipsoidOrientation, ellipsoidRadii, normal);
216 outOverlap = outCapsuleRadius + outEllipsoidRadius - outDistance;
217 return outOverlap > T(0);
218}

232template <typename T>
233[[nodiscard]] bool overlaps(
234 const CollisionGeometryStateT<T>& collisionState,
235 const CollisionGeometry& collisionGeometry,
236 size_t indexA,
237 size_t indexB,
238 CollisionOverlapResultT<T>& result) {
239 const auto typeA = collisionGeometry[indexA].type;
240 const auto typeB = collisionGeometry[indexB].type;
241
242 if (typeA == CollisionPrimitiveType::TaperedCapsule &&
243 typeB == CollisionPrimitiveType::TaperedCapsule) {
244 if (!overlaps(
245 collisionState.origin[indexA],
246 collisionState.direction[indexA],
247 collisionState.radius[indexA],
248 collisionState.delta[indexA],
249 collisionState.origin[indexB],
250 collisionState.direction[indexB],
251 collisionState.radius[indexB],
252 collisionState.delta[indexB],
253 result.distance,
254 result.closestPoints,
255 result.overlap)) {
256 return false;
257 }
258
259 result.positionA =
260 collisionState.origin[indexA] + collisionState.direction[indexA] * result.closestPoints[0];
261 result.positionB =
262 collisionState.origin[indexB] + collisionState.direction[indexB] * result.closestPoints[1];
263 result.radiusA =
264 collisionState.radius[indexA][0] + result.closestPoints[0] * collisionState.delta[indexA];
265 result.radiusB =
266 collisionState.radius[indexB][0] + result.closestPoints[1] * collisionState.delta[indexB];
267 return true;
268 }
269
270 if (typeA == CollisionPrimitiveType::TaperedCapsule &&
271 typeB == CollisionPrimitiveType::Ellipsoid) {
272 if (!overlapsCapsuleEllipsoid(
273 collisionState.origin[indexA],
274 collisionState.direction[indexA],
275 collisionState.radius[indexA],
276 collisionState.delta[indexA],
277 collisionState.origin[indexB],
278 collisionState.orientation[indexB],
279 collisionState.ellipsoidRadii[indexB],
280 result.distance,
281 result.closestPoints[0],
282 result.overlap,
283 result.radiusA,
284 result.radiusB)) {
285 return false;
286 }
287
288 result.closestPoints[1] = T(0);
289 result.positionA =
290 collisionState.origin[indexA] + collisionState.direction[indexA] * result.closestPoints[0];
291 result.positionB = collisionState.origin[indexB];
292 return true;
293 }
294
295 if (typeA == CollisionPrimitiveType::Ellipsoid &&
296 typeB == CollisionPrimitiveType::TaperedCapsule) {
297 if (!overlapsCapsuleEllipsoid(
298 collisionState.origin[indexB],
299 collisionState.direction[indexB],
300 collisionState.radius[indexB],
301 collisionState.delta[indexB],
302 collisionState.origin[indexA],
303 collisionState.orientation[indexA],
304 collisionState.ellipsoidRadii[indexA],
305 result.distance,
306 result.closestPoints[1],
307 result.overlap,
308 result.radiusB,
309 result.radiusA)) {
310 return false;
311 }
312
313 result.closestPoints[0] = T(0);
314 result.positionA = collisionState.origin[indexA];
315 result.positionB =
316 collisionState.origin[indexB] + collisionState.direction[indexB] * result.closestPoints[1];
317 return true;
318 }
319
320 if (typeA == CollisionPrimitiveType::Ellipsoid && typeB == CollisionPrimitiveType::Ellipsoid) {
321 const Vector3<T> separation = collisionState.origin[indexA] - collisionState.origin[indexB];
322 result.distance = separation.norm();
323 if (result.distance < Eps<T>(1e-8, 1e-17)) {
324 return false;
325 }
326
327 const Vector3<T> normal = separation / result.distance;
328 result.radiusA = ellipsoidRadiusAlongDirection(
329 collisionState.orientation[indexA], collisionState.ellipsoidRadii[indexA], normal);
330 result.radiusB = ellipsoidRadiusAlongDirection(
331 collisionState.orientation[indexB], collisionState.ellipsoidRadii[indexB], normal);
332 result.overlap = result.radiusA + result.radiusB - result.distance;
333 result.positionA = collisionState.origin[indexA];
334 result.positionB = collisionState.origin[indexB];
335 result.closestPoints = Vector2<T>::Zero();
336 return result.overlap > T(0);
337 }
338
339 return false;
340}

348template <typename T>
349void updateAabb(
350 axel::BoundingBox<T>& aabb,
351 const Vector3<T>& originA,
352 const Vector3<T>& direction,
353 const Vector2<T>& radii) {
354 const Vector3<T> radius0 = Vector3<T>::Constant(radii[0]);
355 const Vector3<T> radius1 = Vector3<T>::Constant(radii[1]);
356 const Vector3<T> originB = originA + direction;
357
358 const Vector3<T> minA = originA - radius0;
359 const Vector3<T> maxA = originA + radius0;
360 const Vector3<T> minB = originB - radius1;
361 const Vector3<T> maxB = originB + radius1;
362
363 aabb.aabb.min() = minA.cwiseMin(minB);
364 aabb.aabb.max() = maxA.cwiseMax(maxB);
365}

378template <typename T>
379void updateEllipsoidAabb(
380 axel::BoundingBox<T>& aabb,
381 const Vector3<T>& center,
382 const Quaternion<T>& orientation,
383 const Vector3<T>& radii) {
384 const Eigen::Matrix<T, 3, 3> rotation = orientation.toRotationMatrix();
385 const Vector3<T> absRadii = radii.cwiseAbs();
386 Vector3<T> halfExtents;
387 for (int i = 0; i < 3; ++i) {
388 halfExtents[i] = rotation.row(i).cwiseProduct(absRadii.transpose()).norm();
389 }
390 aabb.aabb.min() = center - halfExtents;
391 aabb.aabb.max() = center + halfExtents;
392}

396void updateAabb(
397 axel::BoundingBox<T>& aabb,
398 const CollisionGeometryStateT<T>& collisionState,
399 size_t index) {
400 if (collisionState.type[index] == CollisionPrimitiveType::TaperedCapsule) {
401 updateAabb(

433template <typename T, typename S>
434[[nodiscard]] bool isValidCollisionPair(
435 const CollisionGeometryStateT<T>& collisionState,
436 const CollisionGeometry& collisionGeometry,
437 const SkeletonT<S>& skeleton,
438 size_t i,
439 size_t j,
440 bool filterRestPoseOverlaps = true) {
441 const size_t p0 = collisionGeometry[i].parent;
442 const size_t p1 = collisionGeometry[j].parent;
443
444 // World-fixed primitives (kInvalidIndex parent):
445 // - If both are world-fixed, they can't collide with each other in a meaningful way.
446 // - If exactly one is world-fixed, it's always a valid collision pair (no rest-pose
447 // overlap or adjacency checks apply since the world geometry is independent of
448 // the character's skeleton).
449 if (p0 == kInvalidIndex || p1 == kInvalidIndex) {
450 return p0 != p1;
451 }
452
453 // Check for same or adjacent joints
454 if (skeleton.isSameOrAdjacentJoints(p0, p1)) {
455 return false;
456 }
457
458 if (!filterRestPoseOverlaps) {
459 return true;
460 }
461
462 // Check for rest-pose overlap (permanently overlapping pairs like adjacent body parts)
463 CollisionOverlapResultT<T> overlap;
464 return !overlaps(collisionState, collisionGeometry, i, j, overlap);
465}
