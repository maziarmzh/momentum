72template <typename T>
73bool overlaps(
74 const Vector3<T>& originA,
75 const Vector3<T>& directionA,
76 const Vector2<T>& radiiA,
77 T deltaA,
78 const Vector3<T>& originB,
79 const Vector3<T>& directionB,
80 const Vector2<T>& radiiB,
81 T deltaB,
82 T& outDistance,
83 Vector2<T>& outClosestPoints,
84 T& outOverlap) {
85 // Use the sum of max radii as a proximity cutoff so closestPointsOnSegments can early-out
86 // when segments are guaranteed too far apart for the capsules to overlap.
87 const T maxRadiiSum = radiiA.maxCoeff() + radiiB.maxCoeff();
88
89 auto [success, closestDist, closestPoints] =
90 closestPointsOnSegments<T>(originA, directionA, originB, directionB, maxRadiiSum);
91
92 if (!success) {
93 return false;
94 }
95
96 outClosestPoints = closestPoints;
97
98 // Linearly interpolate each tapered radius at the closest-point parameter along its segment,
99 // then sum to get the combined radius the gap must clear for the capsules to be disjoint.
100 const T radiusAtClosestPoints =
101 radiiA[0] + closestPoints[0] * deltaA + radiiB[0] + closestPoints[1] * deltaB;
102
103 outOverlap = radiusAtClosestPoints - closestDist;
104 outDistance = closestDist;
105
106 // Reject coincident centerlines (closestDist below Eps) to avoid degenerate contact normals
107 // downstream when the two segments are effectively the same line.
108 return (outOverlap > T(0)) && (closestDist >= Eps<T>(1e-8, 1e-17));
109}

117template <typename T>
118void updateAabb(
119 axel::BoundingBox<T>& aabb,
120 const Vector3<T>& originA,
121 const Vector3<T>& direction,
122 const Vector2<T>& radii) {
123 const Vector3<T> radius0 = Vector3<T>::Constant(radii[0]);
124 const Vector3<T> radius1 = Vector3<T>::Constant(radii[1]);
125 const Vector3<T> originB = originA + direction;
126
127 const Vector3<T> minA = originA - radius0;
128 const Vector3<T> maxA = originA + radius0;
129 const Vector3<T> minB = originB - radius1;
130 const Vector3<T> maxB = originB + radius1;
131
132 aabb.aabb.min() = minA.cwiseMin(minB);
133 aabb.aabb.max() = maxA.cwiseMax(maxB);
134}

151template <typename T, typename S>
152[[nodiscard]] bool isValidCollisionPair(
153 const CollisionGeometryStateT<T>& collisionState,
154 const CollisionGeometry& collisionGeometry,
155 const SkeletonT<S>& skeleton,
156 size_t i,
157 size_t j,
158 bool filterRestPoseOverlaps = true) {
159 const size_t p0 = collisionGeometry[i].parent;
160 const size_t p1 = collisionGeometry[j].parent;
161
162 // World-fixed capsules (kInvalidIndex parent):
163 // - If both are world-fixed, they can't collide with each other in a meaningful way.
164 // - If exactly one is world-fixed, it's always a valid collision pair (no rest-pose
165 // overlap or adjacency checks apply since the world geometry is independent of
166 // the character's skeleton).
167 if (p0 == kInvalidIndex || p1 == kInvalidIndex) {
168 return p0 != p1;
169 }
170
171 // Check for same or adjacent joints
172 if (skeleton.isSameOrAdjacentJoints(p0, p1)) {
173 return false;
174 }
175
176 if (!filterRestPoseOverlaps) {
177 return true;
178 }
179
180 // Check for rest-pose overlap (permanently overlapping pairs like adjacent body parts)
181 T distance;
182 Vector2<T> cp;
183 T overlap;
184 return !overlaps(
185 collisionState.origin[i],
186 collisionState.direction[i],
187 collisionState.radius[i],
188 collisionState.delta[i],
189 collisionState.origin[j],
190 collisionState.direction[j],
191 collisionState.radius[j],
192 collisionState.delta[j],
193 distance,
194 cp,
195 overlap);
196}
