31nlohmann::json getMomentumExtension(const nlohmann::json& extensionsAndExtras) {
32 if (extensionsAndExtras.count("extensions") != 0 &&
33 extensionsAndExtras["extensions"].count("FB_momentum") != 0) {
34 return extensionsAndExtras["extensions"]["FB_momentum"];
35 } else {
36 return nlohmann::json::object();

40fx::gltf::Document loadModel(
41 const std::variant<filesystem::path, std::span<const std::byte>>& input) {
42 fx::gltf::Document model;
43 constexpr uint32_t kMax = std::numeric_limits<uint32_t>::max();
44 constexpr fx::gltf::ReadQuotas kQuotas = {8, kMax, kMax};
45
46 std::visit(
47 [&](auto&& arg) {
48 using T = std::decay_t<decltype(arg)>;
49 if constexpr (std::is_same_v<T, filesystem::path>) {
50 // Read first 4 chars for file sensing (https://docs.fileformat.com/3d/glb/)
51 // It is ASCII string "glTF", and can be used to identify data as Binary glTF
52 std::string magic;
53 std::ifstream infile(arg);
54 MT_THROW_IF(!infile.is_open(), "Unable to open: {}", arg.string());
55 std::copy_n(std::istreambuf_iterator<char>(infile.rdbuf()), 4, std::back_inserter(magic));
56 if (magic == "glTF") {
57 model = fx::gltf::LoadFromBinary(arg.string(), kQuotas);
58 } else {
59 model = fx::gltf::LoadFromText(arg.string(), kQuotas);
60 }
61 } else if constexpr (std::is_same_v<T, std::span<const std::byte>>) {
62 ispanstream inputStream(arg);
63 model = fx::gltf::LoadFromBinary(inputStream, "", kQuotas);
64 }
65 },
66 input);
67
68 return model;
69}

73std::tuple<MotionParameters, IdentityParameters> getMotionFromModel(fx::gltf::Document& model) {
74 const auto& def = getMomentumExtension(model.extensionsAndExtras);
75 const auto motion = def.value("motion", nlohmann::json::object());
76
77 const size_t nframes = motion.value("nframes", 0);
78 const int32_t poseBuffer = motion.value("poses", int32_t(-1));
79 const int32_t offsetBuffer = motion.value("offsets", int32_t(-1));
80
81 if (nframes == 0 || (poseBuffer < 0 && offsetBuffer < 0)) {
82 return {};
83 }
84
85 // get values from gltf binary buffers
86 MatrixXf storedMotion;
87 VectorXf storedIdentity;
88 std::vector<std::string> parameterNames;
89 std::vector<std::string> jointNames;
90 if (poseBuffer >= 0) {
91 const auto poseValues = copyAccessorBuffer<float>(model, poseBuffer);
92 parameterNames = motion.value("parameterNames", std::vector<std::string>());
93
94 if (poseValues.size() != parameterNames.size() * nframes) {
95 return {};
96 }
97
98 storedMotion = Map<const MatrixXf>(poseValues.data(), parameterNames.size(), nframes);
99 }
100 if (offsetBuffer >= 0) {
101 const auto offsetValues = copyAccessorBuffer<float>(model, offsetBuffer);
102 jointNames = motion.value("jointNames", std::vector<std::string>());
103
104 if (offsetValues.size() != jointNames.size() * kParametersPerJoint) {
105 return {};
106 }
107
108 storedIdentity =
109 Map<const VectorXf>(offsetValues.data(), jointNames.size() * kParametersPerJoint);
110 }
111
112 return {{parameterNames, storedMotion}, {jointNames, storedIdentity}};
113}

115float calculateFpsFromAnimationData(const fx::gltf::Document& model) {
116 if (model.animations.empty()) {
117 return 0.0f;
118 }
119
120 // Collect FPS values from each sampler
121 std::vector<float> fpsValues;
122 fpsValues.reserve(32);
123
124 // Iterate through all animations and their samplers
125 for (const auto& animation : model.animations) {
126 for (const auto& sampler : animation.samplers) {
127 const int32_t inputAccessorIndex = sampler.input;
128 if (inputAccessorIndex < 0 ||
129 inputAccessorIndex >= static_cast<int32_t>(model.accessors.size())) {
130 continue;
131 }
132
133 const auto& accessor = model.accessors[inputAccessorIndex];
134 if (accessor.count < 2) {
135 continue;
136 }
137
138 // Get time range from accessor min/max
139 if (accessor.min.empty() || accessor.max.empty()) {
140 continue;
141 }
142
143 const float minTime = accessor.min[0];
144 const float maxTime = accessor.max[0];
145 const float duration = maxTime - minTime;
146
147 if (duration <= 0.0f) {
148 continue;
149 }
150
151 // Calculate FPS for this sampler: (keyframes - 1) / duration
152 const float fps = static_cast<float>(accessor.count - 1) / duration;
153 fpsValues.push_back(fps);
154 }
155 }
156
157 if (fpsValues.empty()) {
158 return 0.0f;
159 }
160
161 // Use median FPS to be robust against outliers
162 std::sort(fpsValues.begin(), fpsValues.end());
163 const float medianFps = fpsValues[fpsValues.size() / 2];
164 return medianFps;
165}

167std::tuple<MotionParameters, IdentityParameters, float> loadMotionFromModel(
168 fx::gltf::Document& model) {
169 // parse motion data
170 try {
171 // load motion
172 const auto& def = getMomentumExtension(model.extensionsAndExtras);
173 float fps = def.value("fps", 0.0f);
174
175 // If FPS is not found in momentum metadata, try to calculate from animation data
176 if (fps == 0.0f) {
177 fps = calculateFpsFromAnimationData(model);
178 if (fps > 0.0f) {
179 MT_LOGD("FPS not found in momentum metadata, calculated from animation data: {}", fps);
180 } else {
181 // Fall back to default if calculation fails
182 fps = 120.0f;
183 MT_LOGW("FPS not found in momentum metadata or animation data, using default: {}", fps);
184 }
185 }
186
187 const auto& [motion, identity] = getMotionFromModel(model);
188 return {motion, identity, fps};
189 } catch (std::runtime_error& err) {
190 MT_THROW("Unable to parse motion data : {}", err.what());
191 }
192}

194std::tuple<Character, MatrixXf, JointParameters, float> loadCharacterWithMotionCommon(
195 const std::variant<filesystem::path, std::span<const std::byte>>& input,
196 bool mapParameters) {
197 // ---------------------------------------------
198 // load Skeleton and Mesh
199 // ---------------------------------------------
200 try {
201 // Set maximum filesize to 4 gigabyte (glb hard limit due to uint32).
202 // Need to fork gltf load to support dynamic loading at some point if files get too big
203 fx::gltf::Document model = loadModel(input);
204 Character character = loadGltfCharacter(model);
205
206 // load motion
207 auto [motion, identity, fps] = loadMotionFromModel(model);
208
209 // check that there is actual motion data within the glb / gltf
210 MT_LOGW_IF(std::get<1>(motion).cols() == 0, "No motion data found in gltf file");
211 return {
212 character,
213 mapParameters ? mapMotionToCharacter(motion, character) : std::get<1>(motion),
214 mapParameters ? mapIdentityToCharacter(identity, character) : std::get<1>(identity),
215 fps};
216 } catch (std::runtime_error& err) {
217 MT_THROW("Unable to load gltf : {}", err.what());
218 }
219}

233// Validates a channel's target node and resolves it to a bone index.
234// Returns kInvalidIndex if the channel should be skipped.
235// Throws on bone index out of range or name mismatch.
236size_t validateAndResolveChannel(
237 const fx::gltf::Animation::Channel& channel,
238 const fx::gltf::Document& model,
239 const std::vector<size_t>& nodeToObjectMap,
240 const Skeleton& skeleton) {
241 if (channel.target.node >= nodeToObjectMap.size() ||
242 nodeToObjectMap.at(channel.target.node) == kInvalidIndex) {
243 return kInvalidIndex;
244 }
245
246 // Bounds already validated above
247 const auto boneIdx = nodeToObjectMap.at(channel.target.node);
248 MT_THROW_IF(
249 boneIdx >= skeleton.joints.size(),
250 "Bone index {} is larger than the number of joints in the skeleton ({})",
251 boneIdx,
252 skeleton.joints.size());
253
254 const auto& node = model.nodes.at(channel.target.node);
255 MT_THROW_IF(
256 node.name != skeleton.joints.at(boneIdx).name,
257 "Bone mismatch between gltf and character; GLTF: {}, Character: {}",
258 node.name,
259 skeleton.joints[boneIdx].name);
260
261 return boneIdx;
262}

264// Reads translation/rotation/scale data from a channel sampler and applies it to skeleton states.
265// Returns false if the channel's target path is unrecognized and should be skipped.
266bool applyChannelDataToStates(
267 const fx::gltf::Document& model,
268 const fx::gltf::Animation::Sampler& sampler,
269 const std::string& targetPath,
270 size_t boneIdx,
271 const Character& character,
272 std::vector<SkeletonState>& states,
273 std::vector<float>& timestamps) {
274 // Read the appropriate transform component from the sampler output
275 std::vector<Vector3f> translations_m;
276 std::vector<Quaternionf> rotations;
277 std::vector<Vector3f> scales;
278 if (targetPath == "translation") {
279 translations_m = copyAccessorBuffer<Vector3f>(model, sampler.output);
280 } else if (targetPath == "rotation") {
281 rotations = copyAccessorBuffer<Quaternionf>(model, sampler.output);
282 } else if (targetPath == "scale") {
283 scales = copyAccessorBuffer<Vector3f>(model, sampler.output);
284 } else {
285 return false;
286 }
287
288 // Validate and store timestamps
289 auto timestampsCur = copyAccessorBuffer<float>(model, sampler.input);
290 if (timestamps.empty()) {
291 timestamps = timestampsCur;
292 } else {
293 MT_THROW_IF(
294 timestamps.size() != timestampsCur.size(),
295 "Timestamps do not match between channels, got {} and {}",
296 timestamps.size(),
297 timestampsCur.size());
298 }
299
300 // Lazily initialize states on first channel
301 if (states.empty()) {
302 states.resize(
303 timestamps.size(),
304 SkeletonState(
305 JointParameters::Zero(character.parameterTransform.numJointParameters()),
306 character.skeleton));
307 }
308
309 MT_THROW_IF(states.size() != timestamps.size(), "States and timestamps do not match in size");
310
311 // Apply the transform component to each frame
312 if (translations_m.size() == states.size()) {
313 for (size_t i = 0; i < states.size(); ++i) {
314 MT_THROW_IF(
315 boneIdx >= states[i].jointState.size(),
316 "Bone index {} exceeds jointState size {}",
317 boneIdx,
318 states[i].jointState.size());
319 states[i].jointState[boneIdx].localTransform.translation = toCm() * translations_m[i];
320 }
321 }
322 if (rotations.size() == states.size()) {
323 for (size_t i = 0; i < states.size(); ++i) {
324 MT_THROW_IF(
325 boneIdx >= states[i].jointState.size(),
326 "Bone index {} exceeds jointState size {}",
327 boneIdx,
328 states[i].jointState.size());
329 states[i].jointState[boneIdx].localTransform.rotation = rotations[i];
330 }
331 }
332 if (scales.size() == states.size()) {
333 for (size_t i = 0; i < states.size(); ++i) {
334 MT_THROW_IF(
335 boneIdx >= states[i].jointState.size(),
336 "Bone index {} exceeds jointState size {}",
337 boneIdx,
338 states[i].jointState.size());
339 states[i].jointState[boneIdx].localTransform.scale = scales[i].x();
340 }
341 }
342
343 return true;
344}

346// Computes global transforms from local transforms by walking the joint hierarchy.
347void computeGlobalTransforms(const Skeleton& skeleton, std::vector<SkeletonState>& states) {
348 for (auto& s : states) {
349 for (size_t iJoint = 0; iJoint < skeleton.joints.size(); ++iJoint) {
350 Transform parentTransform;
351 const auto parentJoint = skeleton.joints[iJoint].parent;
352 if (parentJoint != kInvalidIndex) {
353 parentTransform = s.jointState[parentJoint].transform;
354 }
355 s.jointState[iJoint].transform = parentTransform * s.jointState[iJoint].localTransform;
356 }
357 }
358}

362std::tuple<Character, std::vector<SkeletonState>, std::vector<float>>
363loadCharacterWithSkeletonStatesCommon(
364 const std::variant<filesystem::path, std::span<const std::byte>>& input) {
365 const fx::gltf::Document model = loadModel(input);
366 std::vector<size_t> nodeToObjectMap;
367 Character character = loadGltfCharacterInternal(model, nodeToObjectMap);
368
369 if (model.animations.empty()) {
370 MT_LOGW("No animation found in gltf file");
371 return {character, {}, {}};
372 }
373 if (model.animations.size() > 1) {
374 MT_LOGW("Multiple animations found in gltf file. Only the first one will be used.");
375 }
376
377 const auto& animation = model.animations.front();
378 std::vector<float> timestamps;
379 std::vector<SkeletonState> states;
380
381 for (const auto& channel : animation.channels) {
382 const auto boneIdx =
383 validateAndResolveChannel(channel, model, nodeToObjectMap, character.skeleton);
384 if (boneIdx == kInvalidIndex) {
385 continue;
386 }
387
388 const auto& sampler = animation.samplers[channel.sampler];
389 if (!applyChannelDataToStates(
390 model, sampler, channel.target.path, boneIdx, character, states, timestamps)) {
391 continue;
392 }
393 }
394
395 computeGlobalTransforms(character.skeleton, states);
396
397 return {character, states, timestamps};
398}
