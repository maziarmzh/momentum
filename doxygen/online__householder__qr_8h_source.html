37template <typename MatrixType>
38class ColumnIndexedMatrix {
39 public:
40 using VectorType = Eigen::Matrix<typename MatrixType::Scalar, Eigen::Dynamic, 1>;
41
42 explicit ColumnIndexedMatrix(Eigen::Ref<MatrixType> mat) : mat_(mat), useColumnIndices_(false) {}
43
44 explicit ColumnIndexedMatrix(
45 Eigen::Ref<MatrixType> mat,
46 momentum::span<const Eigen::Index> colIndices)
47 : mat_(mat), columnIndices_(colIndices), useColumnIndices_(true) {
48 validateColumnIndices(colIndices, mat.cols());
49 }
50
51 [[nodiscard]] auto col(Eigen::Index colIdx) {
52 if (useColumnIndices_) {
53 return mat_.col(columnIndices_[colIdx]);
54 } else {
55 return mat_.col(colIdx);
56 }
57 }
58
59 [[nodiscard]] auto rows() const {
60 return mat_.rows();
61 }
62
63 [[nodiscard]] auto cols() const {
64 if (useColumnIndices_) {
65 return (Eigen::Index)columnIndices_.size();
66 } else {
67 return mat_.cols();
68 }
69 }
70
71 [[nodiscard]] momentum::span<const Eigen::Index> columnIndices() const {
72 return columnIndices_;
73 }
74
78 // TODO: When `useColumnIndices_` is true, the loop reads `mat_.col(iCol)` instead of
79 // `mat_.col(columnIndices_[iCol])`. This silently ignores the remap and returns the
80 // wrong values whenever `colIndices` is a non-identity permutation/subset.
81 [[nodiscard]] VectorType transpose_times(Eigen::Ref<VectorType> b) const {
82 if (useColumnIndices_) {
83 VectorType result(columnIndices_.size());
84 for (Eigen::Index iCol = 0; iCol < (Eigen::Index)columnIndices_.size(); ++iCol) {
85 result[iCol] = mat_.col(iCol).dot(b);
86 }
87 return result;
88 } else {
89 return mat_.transpose() * b;
90 }
91 }
92
93 private:
94 Eigen::Ref<MatrixType> mat_;
95 momentum::span<const Eigen::Index> columnIndices_;
96 bool useColumnIndices_;
97};
