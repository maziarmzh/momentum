136template <size_t Alignment>
137[[nodiscard]] size_t computeOffset(const Eigen::Ref<Eigen::MatrixXf>& mat) {
138 constexpr size_t sizeOfScalar = sizeof(typename Eigen::MatrixXf::Scalar);
139 const size_t addressOffset =
140 Alignment / sizeOfScalar - (((size_t)mat.data() % Alignment) / sizeOfScalar);
141
142 // If the current alignment already meets the requirement, no offset is needed.
143 if (addressOffset == Alignment / sizeOfScalar) {
144 return 0;
145 }
146
147 return addressOffset;
148}

155template <size_t Alignment>
156void checkAlignment(const Eigen::Ref<Eigen::MatrixXf>& mat, size_t offset = 0) {
157 MT_THROW_IF(
158 (uintptr_t(mat.data() + offset)) % Alignment != 0,
159 "Matrix ({}x{}, ptr: {}) is not aligned ({}) correctly.",
160 mat.rows(),
161 mat.cols(),
162 uintptr_t(mat.data()),
163 Alignment);
164}
