34void saveFbxCommon(
35 const filesystem::path& filename,
36 const Character& character,
37 const MatrixXf& jointValues,
38 const double framerate,
39 const bool saveMesh,
40 const bool skipActiveJointParamCheck,
41 const FbxCoordSystemInfo& coordSystemInfo,
42 Permissive permissive,
43 std::span<const std::vector<Marker>> markerSequence,
44 std::string_view fbxNamespace,
45 const MatrixXf& poses) {
46 // ---------------------------------------------
47 // initialize FBX SDK and prepare for export
48 // ---------------------------------------------
49 auto* manager = ::fbxsdk::FbxManager::Create();
50 auto* ios = ::fbxsdk::FbxIOSettings::Create(manager, IOSROOT);
51 manager->SetIOSettings(ios);
52
53 // Create an exporter.
54 ::fbxsdk::FbxExporter* lExporter = ::fbxsdk::FbxExporter::Create(manager, "");
55
56 // Declare the path and filename of the file containing the scene.
57 // In this case, we are assuming the file is in the same directory as the executable.
58 // Going through string() because on windows, wchar_t (native filesystem path) are different from
59 // char https://en.cppreference.com/w/cpp/language/types This avoids a build error on windows
60 // only.
61 std::string sFilename = filename.string();
62 const char* lFilename = sFilename.c_str();
63
64 // Initialize the exporter.
65 bool lExportStatus = lExporter->Initialize(lFilename, -1, manager->GetIOSettings());
66
67 MT_THROW_IF(
68 !lExportStatus,
69 "Unable to initialize fbx exporter {}",
70 lExporter->GetStatus().GetErrorString());
71
72 // Normalize namespace: ensure it ends with ':' if not empty
73 std::string namespacePrefix(fbxNamespace);
74 if (!namespacePrefix.empty() && namespacePrefix.back() != ':') {
75 namespacePrefix += ":";
76 }
77
78 // ---------------------------------------------
79 // create the scene
80 // ---------------------------------------------
81 ::fbxsdk::FbxScene* scene = ::fbxsdk::FbxScene::Create(manager, "momentum_scene");
82 ::fbxsdk::FbxNode* root = scene->GetRootNode();
83 MT_THROW_IF(root == nullptr, "Unable to get root node from FBX scene");
84
85 // set the coordinate system
86 ::fbxsdk::FbxAxisSystem axis = toFbx(coordSystemInfo);
87 axis.ConvertScene(scene);
88
89 // Create skeleton hierarchy
90 auto skeletonResult = createSkeletonNodes(character, scene);
91 addMetaData(skeletonResult.rootNode, character);
92 createLocatorNodes(character, scene, skeletonResult.nodes);
93 createCollisionGeometryNodes(character, scene, skeletonResult.nodes);
94
95 // Create mesh with blend shapes
96 MeshBlendShapeResult meshResult;
97 if (saveMesh && character.mesh != nullptr) {
98 meshResult = createMeshNode(character, scene, root, skeletonResult.jointToNodeMap, permissive);
99 }
100
101 // Add skeleton to scene root
102 if (!skeletonResult.nodes.empty()) {
103 root->AddChild(skeletonResult.rootNode);
104 }
105
106 // ---------------------------------------------
107 // create animation curves if we have motion
108 // ---------------------------------------------
109 if (jointValues.cols() != 0) {
110 if (jointValues.rows() == character.parameterTransform.numJointParameters()) {
111 createAnimationCurves(
112 character,
113 scene,
114 skeletonResult.nodes,
115 jointValues,
116 framerate,
117 skipActiveJointParamCheck);
118 } else {
119 MT_LOGE(
120 "Rows of joint values {} do not match joint parameter dimension {} so not saving any motion.",
121 jointValues.rows(),
122 character.parameterTransform.numJointParameters());
123 }
124 }
125
126 // ---------------------------------------------
127 // create marker nodes and animation
128 // ---------------------------------------------
129 if (!markerSequence.empty()) {
130 createMarkerNodes(scene, markerSequence, framerate);
131 }
132
133 // ---------------------------------------------
134 // create blend shape animation curves
135 // Auto-detect blend shapes from model parameters
136 // ---------------------------------------------
137 if (saveMesh) {
138 const MatrixXf blendShapeWeights = extractBlendShapeWeights(character, poses);
139 if (blendShapeWeights.cols() > 0 && !meshResult.blendShapeChannels.empty()) {
140 createBlendShapeAnimationCurves(
141 scene, meshResult.blendShapeChannels, blendShapeWeights, framerate);
142 }
143 const MatrixXf faceExpressionWeights = extractFaceExpressionWeights(character, poses);
144 if (faceExpressionWeights.cols() > 0 && !meshResult.faceExprChannels.empty()) {
145 createBlendShapeAnimationCurves(
146 scene, meshResult.faceExprChannels, faceExpressionWeights, framerate);
147 }
148 }
149
150 // ---------------------------------------------
151 // apply namespace prefix to all nodes
152 // ---------------------------------------------
153 if (!namespacePrefix.empty()) {
154 prependNamespaceToAllNodes(scene->GetRootNode(), namespacePrefix);
155 }
156
157 // ---------------------------------------------
158 // close the fbx exporter
159 // ---------------------------------------------
160
161 // finally export the scene
162 lExporter->Export(scene);
163 lExporter->Destroy();
164
165 // destroy the scene and the manager
166 if (scene != nullptr) {
167 scene->Destroy();
168 }
169 manager->Destroy();
170}

226void saveFbx(
227 const filesystem::path& filename,
228 const Character& character,
229 const MatrixXf& poses,
230 const VectorXf& identity,
231 const double framerate,
232 std::span<const std::vector<Marker>> markerSequence,
233 const FileSaveOptions& options) {
234 CharacterParameters params;
235 if (identity.size() == character.parameterTransform.numJointParameters()) {
236 params.offsets = identity;
237 } else {
238 params.offsets = character.parameterTransform.bindPose();
239 }
240
241 CharacterState state;
242 MatrixXf jointValues;
243 if (poses.cols() > 0) {
244 params.pose = poses.col(0);
245 state.set(params, character, false, false, false);
246
247 jointValues.resize(state.skeletonState.jointParameters.v.size(), poses.cols());
248
249 jointValues.col(0) = state.skeletonState.jointParameters.v;
250
251 for (Eigen::Index f = 1; f < poses.cols(); f++) {
252 params.pose = poses.col(f);
253 state.set(params, character, false, false, false);
254 jointValues.col(f) = state.skeletonState.jointParameters.v;
255 }
256 }

328MatrixXf loadFbxBlendShapeWeights(const filesystem::path& filename) {
329 // Initialize FBX SDK
330 auto* manager = ::fbxsdk::FbxManager::Create();
331 auto* ios = ::fbxsdk::FbxIOSettings::Create(manager, IOSROOT);
332 manager->SetIOSettings(ios);
333
334 // RAII-like cleanup for FBX SDK objects
335 ::fbxsdk::FbxScene* scene = nullptr;
336 const auto cleanup = [&]() {
337 if (scene != nullptr) {
338 scene->Destroy();
339 }
340 manager->Destroy();
341 };
342
343 auto* importer = ::fbxsdk::FbxImporter::Create(manager, "");
344 const std::string sFilename = filename.string();
345 if (!importer->Initialize(sFilename.c_str(), -1, manager->GetIOSettings())) {
346 cleanup();
347 return {};
348 }
349
350 scene = ::fbxsdk::FbxScene::Create(manager, "scene");
351 importer->Import(scene);
352 importer->Destroy();
353
354 // Find the animation stack
355 if (scene->GetSrcObjectCount<::fbxsdk::FbxAnimStack>() == 0) {
356 cleanup();
357 return {};
358 }
359 auto* animStack = scene->GetSrcObject<::fbxsdk::FbxAnimStack>(0);
360 if (animStack->GetMemberCount<::fbxsdk::FbxAnimLayer>() == 0) {
361 cleanup();
362 return {};
363 }
364 auto* animLayer = animStack->GetMember<::fbxsdk::FbxAnimLayer>(0);
365
366 // Collect all blend shape channels from all meshes
367 std::vector<::fbxsdk::FbxBlendShapeChannel*> allChannels;
368 for (int iMesh = 0; iMesh < scene->GetSrcObjectCount<::fbxsdk::FbxMesh>(); ++iMesh) {
369 auto* mesh = scene->GetSrcObject<::fbxsdk::FbxMesh>(iMesh);
370 for (int iDeformer = 0; iDeformer < mesh->GetDeformerCount(::fbxsdk::FbxDeformer::eBlendShape);
371 ++iDeformer) {
372 auto* blendShape = static_cast<::fbxsdk::FbxBlendShape*>(
373 mesh->GetDeformer(iDeformer, ::fbxsdk::FbxDeformer::eBlendShape));
374 for (int iChannel = 0; iChannel < blendShape->GetBlendShapeChannelCount(); ++iChannel) {
375 allChannels.push_back(blendShape->GetBlendShapeChannel(iChannel));
376 }
377 }
378 }
379
380 if (allChannels.empty()) {
381 cleanup();
382 return {};
383 }
384
385 // Determine the number of frames from the first animated channel
386 Eigen::Index numFrames = 0;
387 for (auto* channel : allChannels) {
388 auto* curve = channel->DeformPercent.GetCurve(animLayer);
389 if (curve != nullptr) {
390 numFrames = std::max(numFrames, static_cast<Eigen::Index>(curve->KeyGetCount()));
391 }
392 }
393
394 if (numFrames == 0) {
395 cleanup();
396 return {};
397 }
398
399 // Read weights from animation curves
400 const auto numChannels = static_cast<Eigen::Index>(allChannels.size());
401 MatrixXf weights = MatrixXf::Zero(numChannels, numFrames);
402
403 for (Eigen::Index i = 0; i < numChannels; i++) {
404 auto* curve = allChannels[i]->DeformPercent.GetCurve(animLayer);
405 if (curve == nullptr) {
406 continue;
407 }
408
409 const int keyCount = curve->KeyGetCount();
410 for (int k = 0; k < keyCount && k < numFrames; k++) {
411 // FBX DeformPercent is in [0, 100], convert to [0, 1]
412 weights(i, k) = curve->KeyGetValue(k) / 100.0f;
413 }
414 }
415
416 cleanup();
417 return weights;
418}
