49 if (extensionsAndExtras.count("extensions") != 0 &&
50 extensionsAndExtras["extensions"].count("FB_momentum") != 0) {

53 return nlohmann::json::object();

57void loadGlobalExtensions(const fx::gltf::Document& model, Character& character) {
58 try {
59 // load additional momentum data if present
60 const auto& def = getMomentumExtension(model.extensionsAndExtras);
61
62 if (def.count("transform") > 0) {
63 character.parameterTransform = parameterTransformFromJson(character, def["transform"]);
64 } else {
65 character.parameterTransform =
66 ParameterTransform::empty(character.skeleton.joints.size() * kParametersPerJoint);
67 }
68 if (def.count("parameterSet") > 0) {
69 character.parameterTransform.parameterSets =
70 parameterSetsFromJson(character, def["parameterSet"]);
71 }
72 if (def.count("poseConstraints") > 0) {
73 character.parameterTransform.poseConstraints =
74 poseConstraintsFromJson(character, def["poseConstraints"]);
75 }
76 if (def.count("parameterLimits") > 0) {
77 character.parameterLimits = parameterLimitsFromJson(character, def["parameterLimits"]);
78 }
79 if (def.count("metadata") > 0) {
80 character.metadata = def.value<std::string>("metadata", {});
81 // ensure metadata is valid JSON
82 try {
83 auto json = nlohmann::json::parse(character.metadata);
84 } catch (const nlohmann::json::parse_error& e) {
85 MT_LOGW("Failed to parse metadata: {}", e.what());
86 }
87 }
88 } catch (std::runtime_error& err) {
89 MT_THROW("Unable to load gltf : {}", err.what());
90 }
91}

93Character populateCharacterFromModel(
94 const fx::gltf::Document& model,
95 std::vector<size_t>& nodeToObjectMap) {
96 // #TODO: set character name
97 Character result;
98
99 // ---------------------------------------------
100 // load the joints, collision geometry and locators
101 // ---------------------------------------------
102 std::tie(
103 result.name, result.skeleton.joints, result.collision, result.locators, nodeToObjectMap) =
104 loadHierarchy(model);
105 MT_CHECK(
106 nodeToObjectMap.size() == model.nodes.size(),
107 "nodeMap: {}, nodes: {}",
108 nodeToObjectMap.size(),
109 model.nodes.size());
110
111 // ---------------------------------------------
112 // load the mesh -- load from the nodes not in the hierarchy, and combine all meshes into one
113 // ---------------------------------------------
114 std::vector<size_t> remainingNodes;
115 remainingNodes.reserve(model.nodes.size());
116 for (auto nodeId = 0; nodeId < nodeToObjectMap.size(); nodeId++) {
117 if (nodeToObjectMap[nodeId] == kInvalidIndex) {
118 remainingNodes.push_back(nodeId);
119 }
120 }
121
122 BlendShape_u loadedBlendShape;
123 std::tie(result.mesh, result.skinWeights, loadedBlendShape) =
124 loadSkinnedMesh(model, remainingNodes, nodeToObjectMap);
125
126 // Set the blendShape if we loaded any
127 if (loadedBlendShape != nullptr) {
128 result.blendShape = std::move(loadedBlendShape);
129 }
130
131 // Check if there's no skeleton and no skinning nodes found
132 // but there are meshes in the scene. We will create a joint to parent
133 // them.
134 if (result.skeleton.joints.empty() && result.mesh != nullptr && result.skinWeights == nullptr) {
135 result.skeleton.joints = createSkeletonWithOnlyRoot();
136 for (auto& locator : result.locators) {
137 locator.parent = 0;
138 }
139 result.skinWeights = bindMeshToJoint(result.mesh, 0);
140 }
141
142 loadGlobalExtensions(model, result);
143
144 // Finalize jointMap and inverseBind pose
145 result.resetJointMap();
146 result.initInverseBindPose();
147
148 return result;
149}

151fx::gltf::Document loadModel(
152 const std::variant<filesystem::path, std::span<const std::byte>>& input) {
153 fx::gltf::Document model;
154 constexpr uint32_t kMax = std::numeric_limits<uint32_t>::max();
155 constexpr fx::gltf::ReadQuotas kQuotas = {8, kMax, kMax};
156
157 std::visit(
158 [&](auto&& arg) {
159 using T = std::decay_t<decltype(arg)>;
160 if constexpr (std::is_same_v<T, filesystem::path>) {
161 // Read first 4 chars for file sensing (https://docs.fileformat.com/3d/glb/)
162 // It is ASCII string "glTF", and can be used to identify data as Binary glTF
163 std::string magic;
164 std::ifstream infile(arg);
165 MT_THROW_IF(!infile.is_open(), "Unable to open: {}", arg.string());
166 std::copy_n(std::istreambuf_iterator<char>(infile.rdbuf()), 4, std::back_inserter(magic));
167 if (magic == "glTF") {
168 model = fx::gltf::LoadFromBinary(arg.string(), kQuotas);
169 } else {
170 model = fx::gltf::LoadFromText(arg.string(), kQuotas);
171 }
172 } else if constexpr (std::is_same_v<T, std::span<const std::byte>>) {
173 ispanstream inputStream(arg);
174 model = fx::gltf::LoadFromBinary(inputStream, "", kQuotas);
175 }
176 },
177 input);
178
179 return model;
180}

197fx::gltf::Document makeCharacterDocument(
198 const Character& character,
199 const float fps,
200 std::span<const SkeletonState> skeletonStates,
201 std::span<const std::vector<Marker>> markerSequence,
202 bool embedResource,
203 const FileSaveOptions& options) {
204 GltfBuilder fileBuilder;
205
206 const auto kCharacterIsEmpty = character.skeleton.joints.empty() && character.mesh == nullptr;
207 if (!kCharacterIsEmpty) {
208 fileBuilder.addCharacter(character, Vector3f::Zero(), Quaternionf::Identity(), options);
209 }
210 // Add potential motion or offsets, even if the character is empty
211 // (it could be a motion database for example)
212 if (!skeletonStates.empty()) {
213 fileBuilder.addSkeletonStates(character, fps, skeletonStates);
214 }
215 if (!markerSequence.empty()) {
216 MT_THROW_IF(
217 !skeletonStates.empty() && (skeletonStates.size() != markerSequence.size()),
218 "Size of skeleton states vector {} does not correspond to size of marker sequence vector {}",
219 skeletonStates.size(),
220 markerSequence.size());
221
222 fileBuilder.addMarkerSequence(fps, markerSequence);
223 }
224 if (embedResource) {
225 fileBuilder.forceEmbedResources();
226 }
227
228 return fileBuilder.getDocument();
229}

277std::vector<int64_t> loadMotionTimestamps(const filesystem::path& gltfFilename) {
278 try {
279 fx::gltf::Document model = loadModel(gltfFilename);
280 const auto& def = getMomentumExtension(model.extensionsAndExtras);
281
282 // Check if motion section exists and has timestamps
283 if (def.count("motion") == 0) {
284 MT_LOGW("No motion data found in gltf file when loading timestamps");
285 return {};
286 }
287
288 const auto& motion = def["motion"];
289 if (!motion.contains("timestamps")) {
290 MT_LOGW("No timestamps found in gltf file");
291 return {};
292 }
293
294 // Extract timestamps from the JSON array
295 return motion["timestamps"].get<std::vector<int64_t>>();
296 } catch (const std::exception& err) {
297 MT_THROW(
298 "Unable to load timestamps from gltf file '{}'. Error: {}",
299 gltfFilename.string(),
300 err.what());
301 }

314std::tuple<Character, MatrixXf, ModelParameters, float> loadCharacterWithMotionModelParameterScales(
315 const filesystem::path& gltfFilename) {
316 // Read the file into a buffer
317 std::ifstream file(gltfFilename, std::ios::binary | std::ios::ate);
318 MT_THROW_IF(!file.is_open(), "Unable to open file: {}", gltfFilename.string());
319
320 const std::streamsize size = file.tellg();
321 file.seekg(0, std::ios::beg);
322
323 std::vector<std::byte> buffer(size);
324 if (!file.read(reinterpret_cast<char*>(buffer.data()), size)) {
325 MT_THROW("Unable to read file: {}", gltfFilename.string());
326 }
327
328 // Delegate to the byteSpan overload to avoid code duplication
329 return loadCharacterWithMotionModelParameterScales(std::span<const std::byte>(buffer));
330}

400MarkerSequence loadMarkerSequenceFromModel(fx::gltf::Document model) {
401 MarkerSequence result;
402
403 if (model.animations.empty()) {
404 return result;
405 }
406
407 const auto& def = getMomentumExtension(model.extensionsAndExtras);
408 const float fps = def.value("fps", 120.0f);
409 result.fps = fps;
410 // TODO: result.name = model.name;
411
412 const auto& animation = model.animations.front();
413
414 for (const auto& channel : animation.channels) {
415 // make sure we're on a translation channel
416 if (channel.target.path != "translation") {
417 continue;
418 }
419
420 // get the node for the channel
421 if (channel.target.node < 0 || static_cast<size_t>(channel.target.node) >= model.nodes.size()) {
422 continue;
423 }
424 const auto& node = model.nodes.at(channel.target.node);
425
426 // ignore skins and cameras; don't ignore meshes because a marker node may have a mesh for viz
427 if (node.skin >= 0 || node.camera >= 0) {
428 continue;
429 }
430
431 const auto& extension = getMomentumExtension(node.extensionsAndExtras);
432 const std::string type = extension.value("type", "");
433 if (type == "marker") {
434 // found a marker, get the sampler
435 const auto& sampler = animation.samplers[channel.sampler];
436
437 // get the data from the sampler
438 const auto timestamps = copyAccessorBuffer<float>(model, sampler.input);
439 auto positions = copyAccessorBuffer<Vector3f>(model, sampler.output);
440 toMomentumVec3f(positions);
441 if (positions.empty()) {
442 continue;
443 }
444 MT_CHECK(
445 timestamps.size() == positions.size(),
446 "timestamps: {}, positions: {}",
447 timestamps.size(),
448 positions.size());
449
450 // resize the output array if necessary
451 const size_t length = static_cast<size_t>(std::lround(timestamps.back() * fps)) + 1;
452 if (length > result.frames.size()) {
453 result.frames.resize(length);
454 }
455
456 // go over all data and enter into the output array
457 const auto& confArray =
458 extension.count("confidence") != 0 ? extension["confidence"] : nlohmann::json();
459 for (size_t i = 0; i < positions.size(); i++) {
460 const auto index = static_cast<size_t>(std::lround(timestamps[i] * fps));
461 result.frames[index].emplace_back();
462 auto& marker = result.frames[index].back();
463
464 marker.name = node.name;
465 marker.occluded = false;
466 marker.pos = positions[i].cast<double>();
467 if (confArray.is_array() && i < confArray.size()) {
468 marker.confidence = confArray[i].get<float>();
469 }
470 }
471 }
472 }
473
474 return result;
475}
