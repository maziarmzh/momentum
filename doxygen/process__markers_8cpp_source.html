27std::span<const GloveFrameData> validateAndSliceGloveData(
28 std::span<const GloveFrameData> gloveData,
29 size_t markerFrames,
30 size_t firstFrame,
31 size_t lastFrame,
32 const char* label) {
33 if (gloveData.empty()) {
34 return {};
35 }
36 MT_THROW_IF(
37 gloveData.size() != markerFrames,
38 "{} glove data has {} frames but marker data has {} frames",
39 label,
40 gloveData.size(),
41 markerFrames);
42 return gloveData.subspan(firstFrame, lastFrame - firstFrame);
43}

48std::vector<CameraKeypointData> sliceCameraKeypointData(
49 std::span<const CameraKeypointData> data,
50 size_t firstFrame,
51 size_t lastFrame) {
52 std::vector<CameraKeypointData> result;
53 if (data.empty()) {
54 return result;
55 }
56 result.reserve(data.size());
57 for (const auto& camData : data) {
58 CameraKeypointData sliced;
59 sliced.camera = camData.camera;
60 if (camData.frameData.size() >= lastFrame) {
61 sliced.frameData.assign(
62 camData.frameData.begin() + firstFrame, camData.frameData.begin() + lastFrame);
63 } else if (camData.frameData.size() > firstFrame) {
64 sliced.frameData.assign(camData.frameData.begin() + firstFrame, camData.frameData.end());
65 }
66 result.push_back(std::move(sliced));
67 }
68 return result;
69}

73void calibrateMarkers(
74 momentum::Character& character,
75 momentum::ModelParameters& identity,
76 std::span<const std::vector<momentum::Marker>> markerData,
77 const CalibrationConfig& calibrationConfig,
78 size_t firstFrame,
79 size_t maxFrames,
80 std::span<const GloveFrameData> leftGloveData,
81 std::span<const GloveFrameData> rightGloveData,
82 const std::optional<GloveConfig>& gloveConfig,
83 std::span<const CameraKeypointData> cameraKeypointData,
84 std::vector<size_t>* selectedFrameIndices,
85 Eigen::MatrixXf* selectedFrameMotion) {
86 MT_THROW_IF(
87 firstFrame > markerData.size(),
88 "First frame {} can't exceed total frames {}",
89 firstFrame,
90 markerData.size());
91 const size_t lastFrame =
92 maxFrames > 0 ? std::min(firstFrame + maxFrames, markerData.size()) : markerData.size();
93 const std::span<const std::vector<momentum::Marker>> inputData(
94 markerData.data() + firstFrame, lastFrame - firstFrame);
95 MT_THROW_IF(
96 inputData.size() < 2,
97 "Calibration requires at least 2 frames, got {} (from {} total with firstFrame={}, maxFrames={}).",
98 inputData.size(),
99 markerData.size(),
100 firstFrame,
101 maxFrames);
102
103 const auto leftGloveSlice =
104 validateAndSliceGloveData(leftGloveData, markerData.size(), firstFrame, lastFrame, "Left");
105 const auto rightGloveSlice =
106 validateAndSliceGloveData(rightGloveData, markerData.size(), firstFrame, lastFrame, "Right");
107
108 MT_CHECK(
109 !(calibrationConfig.globalScaleOnly & calibrationConfig.locatorsOnly),
110 "globalScaleOnly and locatorsOnly are exclusive; they cannot both be true.");
111
112 const auto slicedKeypointData =
113 sliceCameraKeypointData(cameraKeypointData, firstFrame, lastFrame);
114
115 if (calibrationConfig.locatorsOnly) {
116 // The output locators will be written to character.
117 calibrateLocators(inputData, calibrationConfig, identity, character);
118 } else {
119 // The output locators will be written to character. The output identity will be saved in the
120 // identity variable.
121 calibrateModel(
122 inputData,
123 calibrationConfig,
124 character,
125 identity,
126 {0.0f, 0.0f, 0.0f},
127 leftGloveSlice,
128 rightGloveSlice,
129 gloveConfig,
130 slicedKeypointData,
131 selectedFrameIndices,
132 selectedFrameMotion);
133 }
134}

136Eigen::MatrixXf processMarkers(
137 momentum::Character& character,
138 momentum::ModelParameters& identity,
139 const std::vector<std::vector<momentum::Marker>>& markerData,
140 const TrackingConfig& trackingConfig,
141 const CalibrationConfig& calibrationConfig,
142 bool calibrate,
143 size_t firstFrame,
144 size_t maxFrames,
145 std::span<const GloveFrameData> leftGloveData,
146 std::span<const GloveFrameData> rightGloveData,
147 const std::optional<GloveConfig>& gloveConfig,
148 std::span<const CameraKeypointData> cameraKeypointData) {
149 MT_THROW_IF(
150 firstFrame > markerData.size(),
151 "First frame {} can't exceed total frames {}",
152 firstFrame,
153 markerData.size());
154 const size_t lastFrame =
155 maxFrames > 0 ? std::min(firstFrame + maxFrames, markerData.size()) : markerData.size();
156 const std::span<const std::vector<momentum::Marker>> inputData(
157 markerData.data() + firstFrame, lastFrame - firstFrame);
158 MT_THROW_IF(
159 inputData.empty(),
160 "Input marker data is empty (firstFrame={}, maxFrames={}).",
161 firstFrame,
162 maxFrames);
163
164 const auto leftGloveSlice =
165 validateAndSliceGloveData(leftGloveData, markerData.size(), firstFrame, lastFrame, "Left");
166 const auto rightGloveSlice =
167 validateAndSliceGloveData(rightGloveData, markerData.size(), firstFrame, lastFrame, "Right");
168
169 const auto slicedKeypointData =
170 sliceCameraKeypointData(cameraKeypointData, firstFrame, lastFrame);
171
172 // calibrate model and locators
173 if (calibrate) {
174 MT_CHECK(
175 !(calibrationConfig.globalScaleOnly & calibrationConfig.locatorsOnly),
176 "globalScaleOnly and locatorsOnly are exclusive; they cannot both be true.");
177
178 if (calibrationConfig.locatorsOnly) {
179 // The output locators will be written to character.
180 calibrateLocators(inputData, calibrationConfig, identity, character);
181 } else {
182 // The output locators will be written to character. The output identity will be saved in the
183 // identity variable.
184 calibrateModel(
185 inputData,
186 calibrationConfig,
187 character,
188 identity,
189 {0.0f, 0.0f, 0.0f},
190 leftGloveSlice,
191 rightGloveSlice,
192 gloveConfig,
193 slicedKeypointData);
194 }
195 }
196
197 // track motion; identity parameters will be repeated for every frame in finalMotion.
198 // If glove data is provided and the character has glove bones (from calibration),
199 // the solver will attach constraints to them. If no glove bones exist yet,
200 // constraints fall back to the wrist joints directly.
201 Eigen::MatrixXf finalMotion = trackPosesPerframe(
202 inputData,
203 character,
204 identity,
205 trackingConfig,
206 1,
207 leftGloveSlice,
208 rightGloveSlice,
209 gloveConfig);
210
211 if (trackingConfig.debug) {
212 auto errors = getLocatorError(inputData, finalMotion, character);
213 MT_LOGI("Average marker error: {}", errors.first);
214 MT_LOGI("Max marker error: {}", errors.second);
215 }
216 return finalMotion;
217}

219void processMarkerFile(
220 const std::string& inputMarkerFile,
221 const std::string& outputFile,
222 const TrackingConfig& trackingConfig,
223 const CalibrationConfig& calibrationConfig,
224 const ModelOptions& modelOptions,
225 bool calibrate,
226 size_t firstFrame,
227 size_t maxFrames) {
228 // validate output file extension and quit early
229 const auto outExtension = filesystem::path(outputFile).extension();
230 if (outExtension != ".fbx" && outExtension != ".gltf" && outExtension != ".glb") {
231 MT_LOGE(
232 "Invalid output file type {}; supported types are glb, gltf, and fbx",
233 outExtension.string());
234 return;
235 }
236
237 // read input character and optionally its identity parameters
238 auto [character, identity] = loadCharacterWithIdentity(modelOptions);
239
240 // read marker data
241 std::vector<std::string> validMarkerNames;
242 for (const auto& locator : character.locators) {
243 validMarkerNames.push_back(locator.name);
244 }
245 const auto actors = momentum::loadMarkers(inputMarkerFile, UpVector::Y, validMarkerNames);
246
247 // first try to see if we have an actor with the same name as the character
248 std::optional<MarkerSequence> actor;
249 for (const auto& a : actors) {
250 if (a.name == character.name) {
251 actor = a;
252 break;
253 }
254 }
255
256 // if not, get the main actor
257 if (!actor) {
258 const int index = findMainSubjectIndex(actors);
259 if (index >= 0) {
260 actor = actors[index];
261 }
262 }
263
264 // did we find an actor?
265 if (!actor) {
266 MT_LOGE("No marker sequence found in the marker file: {}", inputMarkerFile);
267 return;
268 }
269 try {
270 Eigen::MatrixXf finalMotion = processMarkers(
271 character,
272 identity,
273 actor->frames,
274 trackingConfig,
275 calibrationConfig,
276 calibrate,
277 firstFrame,
278 maxFrames);
279
280 // name the character after the actor name
281 character.name = actor->name;
282
283 // save results
284 const size_t lastFrame = maxFrames > 0 ? std::min(firstFrame + maxFrames, actor->frames.size())
285 : actor->frames.size();
286 saveMotion(
287 outputFile,
288 character,
289 identity,
290 finalMotion,
291 std::span<const std::vector<momentum::Marker>>(
292 actor->frames.data() + firstFrame, actor->frames.data() + lastFrame),
293 actor->fps);
294 MT_LOGI("{} saved", outputFile);
295 } catch (std::exception& e) {
296 MT_LOGE("Failed to track the markers from file {}, exception: {}", inputMarkerFile, e.what());
297 }
298}
