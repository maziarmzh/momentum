27std::span<const GloveFrameData> validateAndSliceGloveData(
28 std::span<const GloveFrameData> gloveData,
29 size_t markerFrames,
30 size_t firstFrame,
31 size_t lastFrame,
32 const char* label) {
33 if (gloveData.empty()) {
34 return {};
35 }
36 MT_THROW_IF(
37 gloveData.size() != markerFrames,
38 "{} glove data has {} frames but marker data has {} frames",
39 label,
40 gloveData.size(),
41 markerFrames);
42 return gloveData.subspan(firstFrame, lastFrame - firstFrame);
43}

47void calibrateMarkers(
48 momentum::Character& character,
49 momentum::ModelParameters& identity,
50 std::span<const std::vector<momentum::Marker>> markerData,
51 const CalibrationConfig& calibrationConfig,
52 size_t firstFrame,
53 size_t maxFrames,
54 std::span<const GloveFrameData> leftGloveData,
55 std::span<const GloveFrameData> rightGloveData,
56 const std::optional<GloveConfig>& gloveConfig) {
57 MT_THROW_IF(
58 firstFrame > markerData.size(),
59 "First frame {} can't exceed total frames {}",
60 firstFrame,
61 markerData.size());
62 const size_t lastFrame =
63 maxFrames > 0 ? std::min(firstFrame + maxFrames, markerData.size()) : markerData.size();
64 const std::span<const std::vector<momentum::Marker>> inputData(
65 markerData.data() + firstFrame, lastFrame - firstFrame);
66 MT_THROW_IF(
67 inputData.size() < 2,
68 "Calibration requires at least 2 frames, got {} (from {} total with firstFrame={}, maxFrames={}).",
69 inputData.size(),
70 markerData.size(),
71 firstFrame,
72 maxFrames);
73
74 const auto leftGloveSlice =
75 validateAndSliceGloveData(leftGloveData, markerData.size(), firstFrame, lastFrame, "Left");
76 const auto rightGloveSlice =
77 validateAndSliceGloveData(rightGloveData, markerData.size(), firstFrame, lastFrame, "Right");
78
79 MT_CHECK(
80 !(calibrationConfig.globalScaleOnly & calibrationConfig.locatorsOnly),
81 "globalScaleOnly and locatorsOnly are exclusive; they cannot both be true.");
82
83 if (calibrationConfig.locatorsOnly) {
84 // The output locators will be written to character.
85 calibrateLocators(inputData, calibrationConfig, identity, character);
86 } else {
87 // The output locators will be written to character. The output identity will be saved in the
88 // identity variable.
89 calibrateModel(
90 inputData,
91 calibrationConfig,
92 character,
93 identity,
94 {0.0f, 0.0f, 0.0f},
95 leftGloveSlice,
96 rightGloveSlice,
97 gloveConfig);
98 }

101Eigen::MatrixXf processMarkers(
102 momentum::Character& character,
103 momentum::ModelParameters& identity,
104 const std::vector<std::vector<momentum::Marker>>& markerData,
105 const TrackingConfig& trackingConfig,
106 const CalibrationConfig& calibrationConfig,
107 bool calibrate,
108 size_t firstFrame,
109 size_t maxFrames,
110 std::span<const GloveFrameData> leftGloveData,
111 std::span<const GloveFrameData> rightGloveData,
112 const std::optional<GloveConfig>& gloveConfig) {
113 MT_THROW_IF(
114 firstFrame > markerData.size(),
115 "First frame {} can't exceed total frames {}",
116 firstFrame,
117 markerData.size());
118 const size_t lastFrame =
119 maxFrames > 0 ? std::min(firstFrame + maxFrames, markerData.size()) : markerData.size();
120 const std::span<const std::vector<momentum::Marker>> inputData(
121 markerData.data() + firstFrame, lastFrame - firstFrame);
122 MT_THROW_IF(
123 inputData.empty(),
124 "Input marker data is empty (firstFrame={}, maxFrames={}).",
125 firstFrame,
126 maxFrames);
127
128 const auto leftGloveSlice =
129 validateAndSliceGloveData(leftGloveData, markerData.size(), firstFrame, lastFrame, "Left");
130 const auto rightGloveSlice =
131 validateAndSliceGloveData(rightGloveData, markerData.size(), firstFrame, lastFrame, "Right");
132
133 // calibrate model and locators
134 if (calibrate) {
135 MT_CHECK(
136 !(calibrationConfig.globalScaleOnly & calibrationConfig.locatorsOnly),
137 "globalScaleOnly and locatorsOnly are exclusive; they cannot both be true.");
138
139 if (calibrationConfig.locatorsOnly) {
140 // The output locators will be written to character.
141 calibrateLocators(inputData, calibrationConfig, identity, character);
142 } else {
143 // The output locators will be written to character. The output identity will be saved in the
144 // identity variable.
145 calibrateModel(
146 inputData,
147 calibrationConfig,
148 character,
149 identity,
150 {0.0f, 0.0f, 0.0f},
151 leftGloveSlice,
152 rightGloveSlice,
153 gloveConfig);
154 }
155 }
156
157 // track motion; identity parameters will be repeated for every frame in finalMotion.
158 // If glove data is provided and the character has glove bones (from calibration),
159 // the solver will attach constraints to them. If no glove bones exist yet,
160 // constraints fall back to the wrist joints directly.
161 Eigen::MatrixXf finalMotion = trackPosesPerframe(
162 inputData,
163 character,
164 identity,
165 trackingConfig,
166 1,
167 leftGloveSlice,
168 rightGloveSlice,
169 gloveConfig);
170
171 if (trackingConfig.debug) {
172 auto errors = getLocatorError(inputData, finalMotion, character);
173 MT_LOGI("Average marker error: {}", errors.first);
174 MT_LOGI("Max marker error: {}", errors.second);
175 }
176 return finalMotion;
177}

179void processMarkerFile(
180 const std::string& inputMarkerFile,
181 const std::string& outputFile,
182 const TrackingConfig& trackingConfig,
183 const CalibrationConfig& calibrationConfig,
184 const ModelOptions& modelOptions,
185 bool calibrate,
186 size_t firstFrame,
187 size_t maxFrames) {
188 // validate output file extension and quit early
189 const auto outExtension = filesystem::path(outputFile).extension();
190 if (outExtension != ".fbx" && outExtension != ".gltf" && outExtension != ".glb") {
191 MT_LOGE(
192 "Invalid output file type {}; supported types are glb, gltf, and fbx",
193 outExtension.string());
194 return;
195 }
196
197 // read input character and optionally its identity parameters
198 auto [character, identity] = loadCharacterWithIdentity(modelOptions);
199
200 // read marker data
201 std::vector<std::string> validMarkerNames;
202 for (const auto& locator : character.locators) {
203 validMarkerNames.push_back(locator.name);
204 }
205 const auto actors = momentum::loadMarkers(inputMarkerFile, UpVector::Y, validMarkerNames);
206
207 // first try to see if we have an actor with the same name as the character
208 std::optional<MarkerSequence> actor;
209 for (const auto& a : actors) {
210 if (a.name == character.name) {
211 actor = a;
212 break;
213 }
214 }
215
216 // if not, get the main actor
217 if (!actor) {
218 const int index = findMainSubjectIndex(actors);
219 if (index >= 0) {
220 actor = actors[index];
221 }
222 }
223
224 // did we find an actor?
225 if (!actor) {
226 MT_LOGE("No marker sequence found in the marker file: {}", inputMarkerFile);
227 return;
228 }
229 try {
230 Eigen::MatrixXf finalMotion = processMarkers(
231 character,
232 identity,
233 actor->frames,
234 trackingConfig,
235 calibrationConfig,
236 calibrate,
237 firstFrame,
238 maxFrames);
239
240 // name the character after the actor name
241 character.name = actor->name;
242
243 // save results
244 const size_t lastFrame = maxFrames > 0 ? std::min(firstFrame + maxFrames, actor->frames.size())
245 : actor->frames.size();
246 saveMotion(
247 outputFile,
248 character,
249 identity,
250 finalMotion,
251 std::span<const std::vector<momentum::Marker>>(
252 actor->frames.data() + firstFrame, actor->frames.data() + lastFrame),
253 actor->fps);
254 MT_LOGI("{} saved", outputFile);
255 } catch (std::exception& e) {
256 MT_LOGE("Failed to track the markers from file {}, exception: {}", inputMarkerFile, e.what());
257 }
258}
