14// Returns the scale factor in meters for a given LengthUnit.
15template <typename T>
16constexpr T unitInMeters(LengthUnit u) {
17 switch (u) {
18 case LengthUnit::Meter:
19 return T(1);
20 case LengthUnit::Decimeter:
21 return T(0.1);
22 case LengthUnit::Centimeter:
23 return T(0.01);
24 case LengthUnit::Millimeter:
25 return T(0.001);
26 }
27 return T(1); // unreachable
28}

30// Computes the signed permutation matrix P that converts coordinates from
31// system 'from' to system 'to': v_to = P * v_from.
32//
33// Each coordinate system defines a canonical assignment of semantic axes
34// (right, forward, up) to world axes (x, y, z):
35// Y-up right-hand: right=+X, forward=-Z, up=+Y
36// Z-up right-hand: right=+X, forward=+Y, up=+Z
37// X-up right-hand: right=+Y, forward=+Z, up=+X
38// For left-handed systems, the forward axis is negated.
39//
40// The change-of-basis matrix is: P = toAxes * fromAxes^T
41// where each column of 'axes' maps a semantic axis to its world direction.
42template <typename T>
43Matrix3<T> permutationMatrix(const CoordinateSystem& from, const CoordinateSystem& to) {
44 auto getAxes = [](UpAxis up, Handedness hand) -> Matrix3<T> {
45 Matrix3<T> m;
46 switch (up) {
47 case UpAxis::Y:
48 m.col(0) = Vector3<T>(1, 0, 0);
49 m.col(1) = (hand == Handedness::Right) ? Vector3<T>(0, 0, -1) : Vector3<T>(0, 0, 1);
50 m.col(2) = Vector3<T>(0, 1, 0);
51 break;
52 case UpAxis::Z:
53 m.col(0) = Vector3<T>(1, 0, 0);
54 m.col(1) = (hand == Handedness::Right) ? Vector3<T>(0, 1, 0) : Vector3<T>(0, -1, 0);
55 m.col(2) = Vector3<T>(0, 0, 1);
56 break;
57 case UpAxis::X:
58 m.col(0) = Vector3<T>(0, 1, 0);
59 m.col(1) = (hand == Handedness::Right) ? Vector3<T>(0, 0, 1) : Vector3<T>(0, 0, -1);
60 m.col(2) = Vector3<T>(1, 0, 0);
61 break;
62 }
63 return m;
64 };
65
66 const Matrix3<T> fromAxes = getAxes(from.up, from.hand);
67 const Matrix3<T> toAxes = getAxes(to.up, to.hand);
68 return toAxes * fromAxes.transpose();
69}

73template <typename T>
74T scaleFactor(const CoordinateSystem& from, const CoordinateSystem& to) {
75 return unitInMeters<T>(from.unit) / unitInMeters<T>(to.unit);
76}

78template <typename T>
79Vector3<T>
80changeVector(const Vector3<T>& v, const CoordinateSystem& from, const CoordinateSystem& to) {
81 const T s = scaleFactor<T>(from, to);
82 const Matrix3<T> P = permutationMatrix<T>(from, to);
83 return s * (P * v);
84}

86template <typename T>
87Quaternion<T>
88changeQuaternion(const Quaternion<T>& q, const CoordinateSystem& from, const CoordinateSystem& to) {
89 const Matrix3<T> R = q.toRotationMatrix();
90 const Matrix3<T> newR = changeRotationMatrix(R, from, to);
91 return Quaternion<T>(newR);
92}

94template <typename T>
95Matrix3<T> changeRotationMatrix(
96 const Matrix3<T>& m,
97 const CoordinateSystem& from,
98 const CoordinateSystem& to) {
99 const Matrix3<T> P = permutationMatrix<T>(from, to);
100 return P * m * P.transpose();
101}
