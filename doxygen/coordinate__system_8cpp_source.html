14// Returns the scale factor in meters for a given LengthUnit.
15template <typename T>
16constexpr T unitInMeters(LengthUnit u) {
17 switch (u) {
18 case LengthUnit::Meter:
19 return T(1);
20 case LengthUnit::Decimeter:
21 return T(0.1);
22 case LengthUnit::Centimeter:
23 return T(0.01);
24 case LengthUnit::Millimeter:
25 return T(0.001);
26 }
27 // Defensive fallback for ill-formed LengthUnit values (e.g. cast from int).
28 // All declared enumerators are handled above, so this path is unreachable for valid inputs.
29 return T(1);
30}

32// Computes the signed permutation matrix P that converts coordinates from
33// system 'from' to system 'to': v_to = P * v_from.
34//
35// Convention: each coordinate system maps the semantic axes
36// (right, forward, up) to world axes (x, y, z) as follows:
37// Y-up right-hand: right=+X, forward=-Z, up=+Y (OpenGL-style)
38// Z-up right-hand: right=+X, forward=+Y, up=+Z (Blender / robotics-style)
39// X-up right-hand: right=+Y, forward=+Z, up=+X
40// For left-handed systems, only the forward axis is negated; right and up
41// keep the same world directions. This is the minimal flip that preserves
42// up alignment while reversing chirality.
43//
44// Note: only handedness changes the sign of det(axes); axis permutation
45// alone is always a proper rotation. The forward-only flip means
46// det(axes) = -1 iff Handedness::Left, so P keeps right vectors right
47// and up vectors up — only forward-pointing components flip when the
48// handedness differs between 'from' and 'to'.
49//
50// The change-of-basis matrix is: P = toAxes * fromAxes^T
51// where each column of 'axes' is the world-space direction of a semantic
52// axis: col(0)=right, col(1)=forward, col(2)=up. So 'axes' maps a vector
53// expressed in (right, forward, up) coordinates into world (x, y, z),
54// and 'axes^T' is its inverse (the matrix is orthogonal by construction).
55template <typename T>
56Matrix3<T> permutationMatrix(const CoordinateSystem& from, const CoordinateSystem& to) {
57 // Returns the 3x3 matrix whose columns are the world-axis directions of
58 // the (right, forward, up) semantic axes for the given (up, handedness).
59 auto getAxes = [](UpAxis up, Handedness hand) -> Matrix3<T> {
60 Matrix3<T> m;
61 switch (up) {
62 case UpAxis::Y:
63 m.col(0) = Vector3<T>(1, 0, 0); // right = +X
64 // forward = -Z (right-handed) or +Z (left-handed)
65 m.col(1) = (hand == Handedness::Right) ? Vector3<T>(0, 0, -1) : Vector3<T>(0, 0, 1);
66 m.col(2) = Vector3<T>(0, 1, 0); // up = +Y
67 break;
68 case UpAxis::Z:
69 m.col(0) = Vector3<T>(1, 0, 0); // right = +X
70 // forward = +Y (right-handed) or -Y (left-handed)
71 m.col(1) = (hand == Handedness::Right) ? Vector3<T>(0, 1, 0) : Vector3<T>(0, -1, 0);
72 m.col(2) = Vector3<T>(0, 0, 1); // up = +Z
73 break;
74 case UpAxis::X:
75 m.col(0) = Vector3<T>(0, 1, 0); // right = +Y
76 // forward = +Z (right-handed) or -Z (left-handed)
77 m.col(1) = (hand == Handedness::Right) ? Vector3<T>(0, 0, 1) : Vector3<T>(0, 0, -1);
78 m.col(2) = Vector3<T>(1, 0, 0); // up = +X
79 break;
80 }
81 return m;
82 };
83
84 const Matrix3<T> fromAxes = getAxes(from.up, from.hand);
85 const Matrix3<T> toAxes = getAxes(to.up, to.hand);
86 // 'axes' is orthogonal (columns are unit basis vectors with at most a sign
87 // flip), so transpose() == inverse() — no need for a numeric inverse.
88 return toAxes * fromAxes.transpose();
89}

93template <typename T>
94T scaleFactor(const CoordinateSystem& from, const CoordinateSystem& to) {
95 return unitInMeters<T>(from.unit) / unitInMeters<T>(to.unit);
96}

98template <typename T>
99Vector3<T>
100changeVector(const Vector3<T>& v, const CoordinateSystem& from, const CoordinateSystem& to) {
101 const T s = scaleFactor<T>(from, to);
102 const Matrix3<T> P = permutationMatrix<T>(from, to);
103 return s * (P * v);
104}

106template <typename T>
107Quaternion<T>
108changeQuaternion(const Quaternion<T>& q, const CoordinateSystem& from, const CoordinateSystem& to) {
109 // Round-trip via rotation matrix: this transparently handles handedness
110 // flips. P*R*P^T has det = det(P)^2 * det(R) = +1 regardless of whether
111 // 'from' and 'to' share handedness, so the result is always a proper
112 // rotation that Eigen's Quaternion(Matrix3) constructor can normalize.
113 const Matrix3<T> R = q.toRotationMatrix();
114 const Matrix3<T> newR = changeRotationMatrix(R, from, to);
115 return Quaternion<T>(newR);
116}

118template <typename T>
119Matrix3<T> changeRotationMatrix(
120 const Matrix3<T>& m,
121 const CoordinateSystem& from,
122 const CoordinateSystem& to) {
123 const Matrix3<T> P = permutationMatrix<T>(from, to);
124 return P * m * P.transpose();
125}
