48size_t computeSampleStride(size_t numFrames, size_t targetFrames, size_t greedyMax = 0) {
49 if (targetFrames == 0 || numFrames == 0) {
50 return 1;
51 }
52 size_t stride = (numFrames - 1) / targetFrames;
53 if (greedyMax > 0) {
54 stride = std::min(stride, greedyMax);
55 }
56 return std::max(size_t(1), stride);
57}

75std::vector<size_t> sampleFrames(
76 momentum::Character& character,
77 const MatrixXf& initialMotion,
78 std::span<const std::vector<momentum::Marker>> markerData,
79 const ParameterSet& parameters,
80 const size_t frameStride,
81 const size_t numSamples) {
82 // sample frames so that we get the most variance from the initial input
83 const auto numFrames = static_cast<size_t>(initialMotion.cols());
84 if (numFrames == 0) {
85 return {};
86 }
87 MT_CHECK(frameStride > 0, "frameStride must be > 0.");
88 size_t solvedFrames = (numFrames - 1) / frameStride + 1;
89 std::vector<size_t> frameIndices;
90 const size_t numActualSamples = std::min(numSamples, solvedFrames);
91
92 // get the indices of the parameters to be used
93 std::vector<size_t> usedParameters;
94 for (size_t i = 0; i < static_cast<size_t>(initialMotion.rows()); ++i) {
95 if (parameters.test(i)) {
96 usedParameters.push_back(i);
97 }
98 }
99
100 // calculate per frame error
101 const ParameterTransform& pt = character.parameterTransform;
102
103 // map locator name to its index
104 std::map<std::string, size_t> locatorLookup;
105 for (size_t i = 0; i < character.locators.size(); i++) {
106 locatorLookup[character.locators[i].name] = i;
107 }
108
109 std::vector<double> frameErrors(solvedFrames, 0.0f);
110 SkeletonState state;
111 for (size_t iFrame = 0, fi = 0; iFrame < numFrames; iFrame += frameStride, fi++) {
112 const auto jointParams = pt.apply(initialMotion.col(iFrame));
113 state.set(jointParams, character.skeleton, false);
114
115 const auto& markerList = markerData[iFrame];
116 for (const auto& jMarker : markerList) {
117 if (jMarker.occluded) {
118 continue;
119 }
120 auto query = locatorLookup.find(jMarker.name);
121 if (query == locatorLookup.end()) {
122 continue;
123 }
124 size_t locatorIdx = query->second;
125 if (locatorIdx >= character.locators.size()) {
126 continue;
127 }
128 const auto& locator = character.locators[locatorIdx];
129 if (locator.parent >= state.jointState.size()) {
130 continue;
131 }
132 const Vector3f locatorPos = state.jointState[locator.parent].transform * locator.offset;
133 const Vector3f diff = locatorPos - jMarker.pos.cast<float>();
134 const float markerError = diff.norm();
135 frameErrors[fi] += markerError;
136 }
137 }
138 std::vector<double> sortedErrors = frameErrors;
139 std::sort(sortedErrors.begin(), sortedErrors.end());
140
141 // do not use the worst 1/4 of the fitted errors, there's likely to be some outliers and
142 // errors due to the initialization
143 const double threshold = sortedErrors[(sortedErrors.size() * 3) / 4];
144
145 // get the motion of only the used parameters and then normalize
146 // by the mean and variance
147 MatrixXf normalized;
148 {
149 const MatrixXf subMotion =
150 initialMotion(usedParameters, Eigen::seq(0, numFrames - 1, frameStride));
151 const VectorXf mean = subMotion.rowwise().mean();
152 const MatrixXf centered = subMotion.colwise() - mean;
153 const VectorXf std = centered.array().square().rowwise().sum() / (numFrames - 1);
154
155 // normalize the motion by the sqrt of the variance, i.e. leave a little bit of scale in
156 normalized = centered.array().colwise() / (std.array().sqrt().sqrt().cwiseMax(1e-5f));
157 }
158
159 for (size_t i = 0; i < frameErrors.size(); ++i) {
160 if (frameErrors[i] > threshold) {
161 normalized.col(i) = VectorXf::Ones(normalized.rows()) * 1000.0f;
162 }
163 }
164
165 // calculate the distances from each frame to the already selected indices
166 frameIndices.push_back(0);
167 VectorXf distances = VectorXf::Zero(solvedFrames);
168 for (size_t fi = 0; fi < solvedFrames; fi++) {
169 distances[fi] = (normalized.col(frameIndices[0]) - normalized.col(fi)).norm();
170 }
171
172 // finally add additional samples with the largest distance
173 for (size_t i = frameIndices.size(); i < numActualSamples; ++i) {
174 size_t maxFrame = 0;
175 const float maxval = distances.maxCoeff(&maxFrame);
176 if (maxval < 1e-5f) {
177 break;
178 }
179 frameIndices.push_back(maxFrame);
180
181 for (size_t fi = 0; fi < solvedFrames; fi++) {
182 const float dist = (normalized.col(maxFrame) - normalized.col(fi)).cwiseAbs().maxCoeff();
183 distances[fi] = std::min(distances[fi], dist);
184 }
185 }
186
187 for (auto&& fi : frameIndices) {
188 fi *= frameStride;
189 }
190
191 return frameIndices;
192}

252void addBlendShapeRegularizer(
253 SequenceSolverFunctionT<float>& solverFunc,
254 size_t solverFrame,
255 size_t solvedFrames,
256 const Character& character) {
257 const ParameterTransform& pt = character.parameterTransform;
258 if (!pt.getBlendShapeParameters().any() || solverFrame != 0) {
259 return;
260 }
261 auto blendShapeConstrFunc = std::make_shared<ModelParametersErrorFunction>(character);
262 blendShapeConstrFunc->setWeight(solvedFrames * 0.1f);
263 Eigen::VectorXf weights = Eigen::VectorXf::Zero(pt.numAllModelParameters());
264 for (Eigen::Index i = 0; i < pt.blendShapeParameters.size(); ++i) {
265 if (pt.blendShapeParameters[i] >= 0) {
266 weights[pt.blendShapeParameters[i]] = 1.0f;
267 }
268 }
269 blendShapeConstrFunc->setTargetParameters(
270 ModelParameters::Zero(pt.numAllModelParameters()), weights);
271 solverFunc.addErrorFunction(solverFrame, blendShapeConstrFunc);
272}

276void addSequenceFrameConstraints(
277 SequenceSolverFunctionT<float>& solverFunc,
278 size_t solverFrame,
279 size_t iFrame,
280 size_t solvedFrames,
281 const Character& character,
282 const TrackingConfig& config,
283 const std::vector<std::vector<PositionData>>& constrData,
284 const std::vector<std::vector<SkinnedLocatorConstraint>>& skinnedConstData,
285 const std::vector<SkinnedLocatorTriangleConstraintT<float>>& skinnedLocatorMeshContraints,
286 const std::vector<PlaneDataT<float>>& floorConstraints,
287 bool enforceFloorInFirstFrame,
288 const std::string& firstFramePoseConstraintSet,
289 float targetHeightCm,
290 const std::optional<GloveConfig>& gloveConfig,
291 const std::vector<std::vector<JointToJointPositionDataT<float>>>& leftGlovePosData,
292 const std::vector<std::vector<JointToJointOrientationDataT<float>>>& leftGloveOriData,
293 const std::vector<std::vector<JointToJointPositionDataT<float>>>& rightGlovePosData,
294 const std::vector<std::vector<JointToJointOrientationDataT<float>>>& rightGloveOriData) {
295 auto posConstrWeight = PositionErrorFunction::kLegacyWeight * config.markerWeight;
296 if (solverFrame == 0 && (enforceFloorInFirstFrame || !firstFramePoseConstraintSet.empty())) {
297 posConstrWeight *= solvedFrames;
298 }
299
300 // prepare positional constraints
301 if (!constrData.at(iFrame).empty()) {
302 auto posConstrFunc = std::make_shared<PositionErrorFunction>(character, config.lossAlpha);
303 posConstrFunc->setConstraints(constrData.at(iFrame));
304 posConstrFunc->setWeight(posConstrWeight);
305 solverFunc.addErrorFunction(solverFrame, posConstrFunc);
306 }
307
308 if (!skinnedConstData.at(iFrame).empty()) {
309 auto skinnedConstrFunc = std::make_shared<SkinnedLocatorErrorFunction>(character);
310 skinnedConstrFunc->setConstraints(skinnedConstData.at(iFrame));
311 skinnedConstrFunc->setWeight(posConstrWeight);
312 solverFunc.addErrorFunction(solverFrame, skinnedConstrFunc);
313 }
314
315 if (!skinnedLocatorMeshContraints.empty() && solverFrame == 0) {
316 auto skinnedTriangleConstrFunc =
317 std::make_shared<SkinnedLocatorTriangleErrorFunctionT<float>>(character);
318 skinnedTriangleConstrFunc->setConstraints(skinnedLocatorMeshContraints);
319 skinnedTriangleConstrFunc->setWeight(
320 solvedFrames * posConstrWeight * config.meshConstraintWeight);
321 solverFunc.addErrorFunction(solverFrame, skinnedTriangleConstrFunc);
322 }
323
324 addBlendShapeRegularizer(solverFunc, solverFrame, solvedFrames, character);
325
326 if (targetHeightCm > 0.0f && solverFrame == 0) {
327 auto heightConstrFunc = std::make_shared<HeightErrorFunctionT<float>>(
328 character, targetHeightCm, Eigen::Vector3f::UnitY(), 10);
329 heightConstrFunc->setWeight(solvedFrames * 1.0f);
330 solverFunc.addErrorFunction(solverFrame, heightConstrFunc);
331 }
332
333 // prepare floor constraints
334 if (!floorConstraints.empty()) {
335 bool halfPlane = true;
336 float weightMultiplier = 1.0f;
337 if (enforceFloorInFirstFrame && solverFrame == 0) {
338 halfPlane = false;
339 weightMultiplier = solvedFrames;
340 }
341 auto halfPlaneConstrFunc =
342 std::make_shared<PlaneErrorFunction>(character, /*half plane*/ halfPlane);
343 halfPlaneConstrFunc->setConstraints(floorConstraints);
344 halfPlaneConstrFunc->setWeight(PlaneErrorFunction::kLegacyWeight * weightMultiplier);
345 solverFunc.addErrorFunction(solverFrame, halfPlaneConstrFunc);
346 }
347
348 if (!firstFramePoseConstraintSet.empty() && solverFrame == 0) {
349 if (character.parameterTransform.poseConstraints.count(firstFramePoseConstraintSet) > 0) {
350 const auto poseLimits = getPoseConstraintParameterLimits(
351 firstFramePoseConstraintSet, character.parameterTransform);
352 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character, poseLimits);
353 limitConstrFunc->setWeight(solvedFrames);
354 solverFunc.addErrorFunction(solverFrame, limitConstrFunc);
355 }
356 }
357
358 // add glove constraints for this frame
359 if (gloveConfig) {
360 addGloveConstraintsToSequenceSolver(
361 solverFunc,
362 solverFrame,
363 character,
364 leftGlovePosData,
365 leftGloveOriData,
366 iFrame,
367 gloveConfig->positionWeight,
368 gloveConfig->orientationWeight);
369 addGloveConstraintsToSequenceSolver(
370 solverFunc,
371 solverFrame,
372 character,
373 rightGlovePosData,
374 rightGloveOriData,
375 iFrame,
376 gloveConfig->positionWeight,
377 gloveConfig->orientationWeight);
378 }
379}

398Eigen::MatrixXf trackSequence(
399 const std::span<const std::vector<Marker>> markerData,
400 const Character& character,
401 const ParameterSet& globalParams,
402 const MatrixXf& initialMotion,
403 const TrackingConfig& config,
404 const std::vector<size_t>& frames,
405 float regularizer,
406 bool enforceFloorInFirstFrame,
407 const std::string& firstFramePoseConstraintSet,
408 float targetHeightCm,
409 std::span<const GloveFrameData> leftGloveData,
410 std::span<const GloveFrameData> rightGloveData,
411 const std::optional<GloveConfig>& gloveConfig) {
412 // sanity checks
413 const size_t numFrames = markerData.size();
414 MT_CHECK(numFrames > 0, "Input marker data is empty.");
415 MT_CHECK(
416 initialMotion.cols() >= numFrames,
417 "Number of frames in data {} exceeds input motion columns {}",
418 numFrames,
419 initialMotion.cols());
420 MT_CHECK(
421 initialMotion.rows() == character.parameterTransform.numAllModelParameters(),
422 "Input motion parameters {} do not match character model parameters {}",
423 initialMotion.rows(),
424 character.parameterTransform.numAllModelParameters());
425 MT_CHECK(
426 leftGloveData.empty() || leftGloveData.size() == numFrames,
427 "Left glove data has {} frames but marker data has {} frames",
428 leftGloveData.size(),
429 numFrames);
430 MT_CHECK(
431 rightGloveData.empty() || rightGloveData.size() == numFrames,
432 "Right glove data has {} frames but marker data has {} frames",
433 rightGloveData.size(),
434 numFrames);
435
436 const ParameterTransform& pt = character.parameterTransform;
437
438 // universal parameters include "scaling" and "locators" (if exists); pose parameters need to
439 // exclude "locators". universalParams is to indicate to the solver which parameters are "global"
440 // (ie. not time varying). The input globalParams indicate which parameters within universalParams
441 // we want to solve for. globalParams is either a subset or all of universalParams.
442 ParameterSet poseParams = pt.getPoseParameters();
443 ParameterSet universalParams = pt.getScalingParameters() | pt.getBlendShapeParameters();
444 const auto locatorSet =
445 pt.getParameterSet("locators", true) | pt.getParameterSet("skinnedLocators", true);
446 poseParams &= ~locatorSet;
447 universalParams |= locatorSet;
448
449 // Include glove DOFs as universal (not time-varying) parameters
450 if (gloveConfig) {
451 const auto gloveSet = pt.getParameterSet("gloves", true);
452 poseParams &= ~gloveSet;
453 universalParams |= gloveSet;
454 }
455
456 // Apply caller-specified parameter mask to pose parameters
457 if (config.activeParams) {
458 poseParams &= *config.activeParams;
459 }
460
461 std::vector<momentum::SkinnedLocatorTriangleConstraintT<float>> skinnedLocatorMeshContraints;
462 if (!character.skinnedLocators.empty() && character.mesh) {
463 skinnedLocatorMeshContraints = createSkinnedLocatorMeshConstraints(character);
464 MT_LOGI(
465 "Created {} mesh constraints for {} skinned locators",
466 skinnedLocatorMeshContraints.size(),
467 character.skinnedLocators.size());
468 }
469
470 // set up the solver function
471 std::vector<size_t> sortedFrames = frames;
472 std::sort(sortedFrames.begin(), sortedFrames.end());
473 size_t solvedFrames = sortedFrames.size();
474 auto solverFunc = SequenceSolverFunction(
475 character, character.parameterTransform, universalParams, solvedFrames);
476
477 // floor penetration constraints; we assume the world is y-up and floor is y=0 for mocap data.
478 const auto& floorConstraints = createFloorConstraints<float>(
479 "Floor_",
480 character.locators,
481 Vector3f::UnitY(),
482 /* y offset */ 0.0f,
483 /* weight */ 5.0f);
484
485 // marker constraints
486 const auto constrData = createConstraintData(markerData, character.locators);
487 const auto skinnedConstData = createSkinnedConstraintData(markerData, character.skinnedLocators);
488
489 // glove constraint data (one per hand × position/orientation)
490 std::vector<std::vector<JointToJointPositionDataT<float>>> leftGlovePosData, rightGlovePosData;
491 std::vector<std::vector<JointToJointOrientationDataT<float>>> leftGloveOriData, rightGloveOriData;
492 if (gloveConfig) {
493 leftGlovePosData = createGlovePositionConstraintData(leftGloveData, character, *gloveConfig, 0);
494 leftGloveOriData =
495 createGloveOrientationConstraintData(leftGloveData, character, *gloveConfig, 0);
496 rightGlovePosData =
497 createGlovePositionConstraintData(rightGloveData, character, *gloveConfig, 1);
498 rightGloveOriData =
499 createGloveOrientationConstraintData(rightGloveData, character, *gloveConfig, 1);
500 }
501
502 // add per-frame constraint data to the solver
503 for (size_t solverFrame = 0; solverFrame < solvedFrames; ++solverFrame) {
504 const size_t& iFrame = sortedFrames[solverFrame];
505 if ((constrData.at(iFrame).size() + skinnedConstData.at(iFrame).size()) >
506 markerData[iFrame].size() * config.minVisPercent) {
507 addSequenceFrameConstraints(
508 solverFunc,
509 solverFrame,
510 iFrame,
511 solvedFrames,
512 character,
513 config,
514 constrData,
515 skinnedConstData,
516 skinnedLocatorMeshContraints,
517 floorConstraints,
518 enforceFloorInFirstFrame,
519 firstFramePoseConstraintSet,
520 targetHeightCm,
521 gloveConfig,
522 leftGlovePosData,
523 leftGloveOriData,
524 rightGlovePosData,
525 rightGloveOriData);
526 }
527 // Set per-frame initial value
528 solverFunc.setFrameParameters(solverFrame, initialMotion.col(iFrame));
529 }
530
531 // add parameter limits
532 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character);
533 limitConstrFunc->setWeight(0.1);
534 solverFunc.addErrorFunction(kAllFrames, limitConstrFunc);
535
536 // add collision error
537 if (config.collisionErrorWeight != 0 && character.collision != nullptr) {
538 auto collisionConstrFunc = std::make_shared<CollisionErrorFunctionStateless>(character);
539 collisionConstrFunc->setWeight(config.collisionErrorWeight);
540 solverFunc.addErrorFunction(kAllFrames, collisionConstrFunc);
541 }
542
543 // add a smoothness constraint in parameter space
544 if (config.smoothing != 0) {
545 auto smoothConstrFunc = std::make_shared<ModelParametersSequenceErrorFunction>(character);
546 smoothConstrFunc->setWeight(config.smoothing);
547 solverFunc.addSequenceErrorFunction(kAllFrames, smoothConstrFunc);
548 }
549
550 // minimize the change to global params
551 if (globalParams.count() > 0 && regularizer != 0) {
552 auto regularizerFunc = std::make_shared<ModelParametersErrorFunction>(character);
553 Eigen::VectorXf universalMask(pt.numAllModelParameters());
554 for (size_t i = 0; i < universalMask.size(); ++i) {
555 if (globalParams.test(i)) {
556 universalMask[i] = regularizer;
557 } else {
558 universalMask[i] = 0.0;
559 }
560 }
561 regularizerFunc->setTargetParameters(initialMotion.col(0), universalMask);
562 // Sufficient to add to the first frame since it won't change.
563 solverFunc.addErrorFunction(0, regularizerFunc);
564 }
565
566 // solver configration
567 SequenceSolverOptions solverOptions;
568 solverOptions.maxIterations = config.maxIter;
569 solverOptions.minIterations = 2;
570 solverOptions.progressBar = config.debug;
571 solverOptions.doLineSearch = false;
572 solverOptions.multithreaded = true;
573 solverOptions.verbose = config.debug;
574 solverOptions.threshold = 1.f;
575 solverOptions.regularization = config.regularization;
576
577 // solve the problem
578 SequenceSolver solver = SequenceSolver(solverOptions, &solverFunc);
579 solver.setEnabledParameters(poseParams | globalParams);
580 // returns all the dofs with initial values nicely packed into a vector
581 VectorXf dofs = solverFunc.getJoinedParameterVector();
582 solver.solve(dofs);
583 double error = solverFunc.getError(dofs);
584 MT_LOGI_IF(config.debug, "Solver residual: {}", error);
585
586 // set results to output
587 size_t sortedIndex = 0;
588 MatrixXf outMotion(pt.numAllModelParameters(), numFrames);
589 for (size_t fi = 0; fi < numFrames; fi++) {
590 if (sortedIndex < sortedFrames.size() - 1 && fi == sortedFrames[sortedIndex + 1]) {
591 sortedIndex++;
592 }
593 outMotion.col(fi) = solverFunc.getFrameParameters(sortedIndex).v;
594 }
595 return outMotion;
596}

642 const std::span<const std::vector<Marker>> markerData,

670 markerData,

672 initialMotion,
673 config,
674 frameIndices,
675 isContinuous,
676 leftGloveData,
677 rightGloveData,

688bool solveRigidInitialization(
689 GaussNewtonSolverQR& solver,
690 GaussNewtonSolverQROptions& solverOptions,
691 Eigen::VectorXf& dof,
692 const ParameterTransform& pt,
693 const ParameterSet& poseParams,
694 const std::shared_ptr<ModelParametersErrorFunction>& smoothConstrFunc,
695 const TrackingConfig& config,
696 bool isContinuous,
697 size_t iFrame) {
698 MT_LOGI_IF(config.debug && isContinuous, "Solving for an initial rigid pose at frame {}", iFrame);
699
700 // Set up different config for initialization
701 solverOptions.maxIterations = 50; // make sure it converges
702 solver.setOptions(solverOptions);
703 solver.setEnabledParameters(pt.getRigidParameters());
704 if (smoothConstrFunc) {
705 smoothConstrFunc->setWeight(0.0); // turn off smoothing - it doesn't affect rigid dofs
706 }
707
708 solver.solve(dof);
709
710 // Recover solver config
711 solverOptions.maxIterations = config.maxIter;
712 solver.setOptions(solverOptions);
713 solver.setEnabledParameters(poseParams);
714 if (smoothConstrFunc) {
715 smoothConstrFunc->setWeight(config.smoothing);
716 }
717
718 return false;
719}

735Eigen::MatrixXf trackPosesForFrames(
736 const std::span<const std::vector<Marker>> markerData,
737 const Character& character,
738 const MatrixXf& initialMotion,
739 const TrackingConfig& config,
740 const std::vector<size_t>& frameIndices,
741 bool isContinuous,
742 std::span<const GloveFrameData> leftGloveData,
743 std::span<const GloveFrameData> rightGloveData,
744 const std::optional<GloveConfig>& gloveConfig,
745 const std::string& progressLabel) {
746 const size_t numFrames = markerData.size();
747 MT_CHECK(numFrames > 0, "Input marker data is empty.");
748 MT_CHECK(
749 initialMotion.cols() >= numFrames,
750 "Number of frames in data {} exceeds input motion columns {}",
751 numFrames,
752 initialMotion.cols());
753 MT_CHECK(
754 initialMotion.rows() == character.parameterTransform.numAllModelParameters(),
755 "Input motion parameters {} do not match character model parameters {}",
756 initialMotion.rows(),
757 character.parameterTransform.numAllModelParameters());
758 MT_CHECK(
759 leftGloveData.empty() || leftGloveData.size() == numFrames,
760 "Left glove data has {} frames but marker data has {} frames",
761 leftGloveData.size(),
762 numFrames);
763 MT_CHECK(
764 rightGloveData.empty() || rightGloveData.size() == numFrames,
765 "Right glove data has {} frames but marker data has {} frames",
766 rightGloveData.size(),
767 numFrames);
768
769 const ParameterTransform& pt = character.parameterTransform;
770
771 std::vector<size_t> sortedFrames = frameIndices;
772 std::sort(sortedFrames.begin(), sortedFrames.end());
773
774 // pose parameters need to exclude "locators"
775 ParameterSet poseParams = pt.getPoseParameters();
776 const auto& locatorSet = pt.parameterSets.find("locators");
777 if (locatorSet != pt.parameterSets.end()) {
778 poseParams &= ~locatorSet->second;
779 }
780
781 // Exclude glove DOFs from per-frame pose solving
782 if (gloveConfig) {
783 poseParams &= ~pt.getParameterSet("gloves", true);
784 }
785
786 // Apply caller-specified parameter mask
787 if (config.activeParams) {
788 poseParams &= *config.activeParams;
789 }
790
791 // set up the solver
792 auto solverFunc = SkeletonSolverFunction(character, pt);
793 GaussNewtonSolverQROptions solverOptions;
794 solverOptions.maxIterations = config.maxIter;
795 solverOptions.minIterations = 2;
796 solverOptions.doLineSearch = false;
797 solverOptions.verbose = config.debug;
798 solverOptions.threshold = 1.f;
799 solverOptions.regularization = config.regularization;
800 auto solver = GaussNewtonSolverQR(solverOptions, &solverFunc);
801 solver.setEnabledParameters(poseParams);
802
803 // parameter limits constraint
804 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character);
805 limitConstrFunc->setWeight(0.1);
806 solverFunc.addErrorFunction(limitConstrFunc);
807
808 // positional constraint function for markers
809 auto posConstrFunc = std::make_shared<PositionErrorFunction>(character, config.lossAlpha);
810 posConstrFunc->setWeight(PositionErrorFunction::kLegacyWeight * config.markerWeight);
811 solverFunc.addErrorFunction(posConstrFunc);
812
813 auto skinnedLocatorPosConstrFunc = std::make_shared<SkinnedLocatorErrorFunction>(character);
814 skinnedLocatorPosConstrFunc->setWeight(
815 PositionErrorFunction::kLegacyWeight * config.markerWeight);
816 solverFunc.addErrorFunction(skinnedLocatorPosConstrFunc);
817
818 // floor penetration constraint data; we assume the world is y-up and floor is y=0 for mocap data.
819 const auto& floorConstraints = createFloorConstraints<float>(
820 "Floor_",
821 character.locators,
822 Vector3f::UnitY(),
823 /* y offset */ 0.0f,
824 /* weight */ 5.0f);
825 auto halfPlaneConstrFunc = std::make_shared<PlaneErrorFunction>(character, /*half plane*/ true);
826 halfPlaneConstrFunc->setConstraints(floorConstraints);
827 halfPlaneConstrFunc->setWeight(PlaneErrorFunction::kLegacyWeight);
828 solverFunc.addErrorFunction(halfPlaneConstrFunc);
829
830 // Pre-process marker gaps (interpolate temporary gaps, blend off permanent dropouts).
831 // Creates a mutable copy since the input span is const.
832 auto processedMarkerData = std::vector<std::vector<Marker>>(markerData.begin(), markerData.end());
833 preprocessMarkerGaps(processedMarkerData, config.gapFillConfig);
834
835 // marker constraint data (uses gap-filled markers)
836 auto constrData = createConstraintData(processedMarkerData, character.locators);
837 auto skinnedConstrData =
838 createSkinnedConstraintData(processedMarkerData, character.skinnedLocators);
839
840 // smoothness constraint only for the joints and exclude global dofs because the global transform
841 // needs to be accurate (may not matter in practice?)
842 // Only use temporal smoothness if isContinuous is true
843 std::shared_ptr<ModelParametersErrorFunction> smoothConstrFunc;
844 if (isContinuous) {
845 smoothConstrFunc = std::make_shared<ModelParametersErrorFunction>(
846 character, poseParams & ~pt.getRigidParameters());
847 smoothConstrFunc->setWeight(config.smoothing);
848 solverFunc.addErrorFunction(smoothConstrFunc);
849 }
850
851 // add collision error
852 std::shared_ptr<CollisionErrorFunction> collisionErrorFunction;
853 if (config.collisionErrorWeight != 0 && character.collision != nullptr) {
854 collisionErrorFunction = std::make_shared<CollisionErrorFunction>(character);
855 collisionErrorFunction->setWeight(config.collisionErrorWeight);
856 solverFunc.addErrorFunction(collisionErrorFunction);
857 }
858
859 // set up glove error functions (registered once, constraints swapped per frame)
860 std::optional<GloveErrorFunctions> leftGloveFuncs, rightGloveFuncs;
861 if (gloveConfig) {
862 leftGloveFuncs =
863 setupGloveErrorFunctions(solverFunc, character, leftGloveData, *gloveConfig, 0);
864 rightGloveFuncs =
865 setupGloveErrorFunctions(solverFunc, character, rightGloveData, *gloveConfig, 1);
866 }
867
868 // initialize parameters to contain identity information
869 // the identity fields will be used but untouched during optimization
870 // globalParams could also be repurposed to pass in initial pose value
871 Eigen::VectorXf dof = initialMotion.col(sortedFrames.empty() ? 0 : sortedFrames[0]);
872 size_t solverFrame = 0;
873 double priorError = 0.0;
874 double error = 0.0;
875
876 // Use the initial global transform if it's not zero
877 bool needsInit = isGlobalTransformZero(dof, pt.getRigidParameters());
878
879 MatrixXf outMotion = initialMotion;
880 Eigen::Index outputIndex = 0;
881 { // scope the ProgressBar so it returns
882 ProgressBar progress(progressLabel, sortedFrames.size());
883 for (const auto iFrame : sortedFrames) {
884 // For continuous tracking, keep the solved dof from previous frame (temporal coherence)
885 // For non-continuous tracking, always start from initial motion (independent solving)
886 if (!isContinuous) {
887 dof = initialMotion.col(iFrame);
888 needsInit = true;
889 }
890 // For continuous tracking, dof is preserved from previous iteration (or initial value)
891
892 if ((constrData.at(iFrame).size() + skinnedConstrData.at(iFrame).size()) >
893 processedMarkerData[iFrame].size() * config.minVisPercent) {
894 // add positional constraints
895 posConstrFunc->clearConstraints(); // clear constraint data from the previous frame
896 posConstrFunc->setConstraints(constrData.at(iFrame));
897
898 skinnedLocatorPosConstrFunc->clearConstraints();
899 skinnedLocatorPosConstrFunc->setConstraints(skinnedConstrData.at(iFrame));
900
901 // update glove constraints for this frame
902 updateGloveConstraintsForBothHands(leftGloveFuncs, rightGloveFuncs, iFrame);
903
904 // initialization: solve only for the rigid parameters as preprocessing
905 if (needsInit) {
906 needsInit = solveRigidInitialization(
907 solver,
908 solverOptions,
909 dof,
910 pt,
911 poseParams,
912 smoothConstrFunc,
913 config,
914 isContinuous,
915 iFrame);
916 }
917
918 // set smoothness target as the last pose for continuous tracking
919 if (smoothConstrFunc) {
920 smoothConstrFunc->setTargetParameters(dof, smoothConstrFunc->getTargetWeights());
921 }
922
923 priorError += solverFunc.getError(dof);
924 error += solver.solve(dof);
925 ++solverFrame;
926 }
927
928 // store result
929 while (outputIndex <= iFrame) {
930 outMotion.col(outputIndex++) = dof;
931 }
932 progress.increment();
933 }
934
935 while (outputIndex < numFrames) {
936 outMotion.col(outputIndex++) = dof;
937 }
938 }
939 if (config.debug) {
940 if (solverFrame > 0) {
941 MT_LOGI("Pre optimization residual: {}", priorError / solverFrame);
942 MT_LOGI("Average per-frame residual: {}", error / solverFrame);
943 } else {
944 MT_LOGW("no valid frames to solve");
945 }
946 }
947 return outMotion;
948}

963float computeCharacterHeight(const Character& character, const ModelParameters& identity) {
964 if (!character.mesh) {
965 MT_LOGW("Character has no mesh, cannot compute height");
966 return 0.0f;
967 }
968
969 const ParameterTransform& pt = character.parameterTransform;
970
971 // Create a model parameters vector with scale and shape, but zero pose
972 ModelParameters neutralParams = identity;
973
974 // Zero out all pose parameters
975 const ParameterSet poseParams = pt.getPoseParameters();
976 for (size_t i = 0; i < neutralParams.v.size(); ++i) {
977 if (poseParams.test(i)) {
978 neutralParams.v[i] = 0.0f;
979 }
980 }
981
982 // Create skeleton state with the neutral (zero pose) parameters
983 const auto jointParams = pt.apply(neutralParams.v);
984 SkeletonState state;
985 state.set(jointParams, character.skeleton, false);
986
987 // Apply blend shape skinning to get the deformed mesh
988 Mesh skinnedMesh;
989 skinWithBlendShapes(character, state, neutralParams, skinnedMesh);
990
991 // Find min and max Y values across all vertices
992 if (skinnedMesh.vertices.empty()) {
993 MT_LOGW("Skinned mesh has no vertices, cannot compute height");
994 return 0.0f;
995 }
996
997 float minY = std::numeric_limits<float>::max();
998 float maxY = std::numeric_limits<float>::lowest();
999
1000 for (const auto& vertex : skinnedMesh.vertices) {
1001 minY = std::min(minY, vertex.y());
1002 maxY = std::max(maxY, vertex.y());
1003 }
1004
1005 const float height = maxY - minY;
1006 return height;
1007}

1040void calibrateModel(
1041 const std::span<const std::vector<Marker>> markerData,
1042 const CalibrationConfig& config,
1043 Character& character,
1044 ModelParameters& identity,
1045 const std::array<float, 3>& regularizerWeights,
1046 std::span<const GloveFrameData> leftGloveData,
1047 std::span<const GloveFrameData> rightGloveData,
1048 const std::optional<GloveConfig>& gloveConfig) {
1049 const size_t numFrames = markerData.size();
1050 MT_THROW_IF(numFrames < 2, "Calibration requires at least 2 frames, got {}.", numFrames);
1051 MT_THROW_IF(
1052 identity.v.size() != character.parameterTransform.numAllModelParameters(),
1053 "Input identity parameters {} do not match character parameters {}",
1054 identity.v.size(),
1055 character.parameterTransform.numAllModelParameters());
1056 MT_THROW_IF(
1057 !leftGloveData.empty() && leftGloveData.size() != numFrames,
1058 "Left glove data has {} frames but marker data has {} frames",
1059 leftGloveData.size(),
1060 numFrames);
1061 MT_THROW_IF(
1062 !rightGloveData.empty() && rightGloveData.size() != numFrames,
1063 "Right glove data has {} frames but marker data has {} frames",
1064 rightGloveData.size(),
1065 numFrames);
1066
1067 MT_THROW_IF(config.calibFrames == 0, "calibFrames must be > 0.");
1068 if (numFrames < config.calibFrames) {
1069 MT_LOGW(
1070 "Only {} frames available for calibration (requested {}); using all frames.",
1071 numFrames,
1072 config.calibFrames);
1073 }
1074
1075 // uniformly sample frames for calibration
1076 const size_t frameStride = computeSampleStride(numFrames, config.calibFrames);
1077
1078 // create a solving character with markers as bones
1079 Character solvingCharacter = character;
1080 solvingCharacter = createLocatorCharacter(solvingCharacter, "locator_");
1081 if (!solvingCharacter.skinnedLocators.empty()) {
1082 solvingCharacter = addSkinnedLocatorParametersToTransform(solvingCharacter);
1083 }
1084
1085 // Add glove bones to the solving character if glove data is provided
1086 if (gloveConfig && (!leftGloveData.empty() || !rightGloveData.empty())) {
1087 solvingCharacter = createGloveCharacter(solvingCharacter, *gloveConfig);
1088 }
1089
1090 // Extended quantities are for the solvingCharacter, which includes locators as bones
1091 // w/o Extended quantities are for character with fixed locators
1092 const ParameterTransform& transformExtended = solvingCharacter.parameterTransform;
1093 const ParameterTransform& transform = character.parameterTransform;
1094
1095 ParameterSet locatorSet = transformExtended.getParameterSet("locators", true) |
1096 transformExtended.getParameterSet("skinnedLocators", true);
1097 ParameterSet gloveSet = transformExtended.getParameterSet("gloves", true);
1098 ParameterSet calibBodySetExtended;
1099 ParameterSet calibBodySet;
1100 if (config.globalScaleOnly) {
1101 const size_t paramIndex = transform.getParameterIdByName("scale_global");
1102 const size_t paramIndexExt = transformExtended.getParameterIdByName("scale_global");
1103 MT_THROW_IF(
1104 paramIndex == kInvalidIndex || paramIndexExt == kInvalidIndex,
1105 "Can't calibrate global scale since it is not defined in the parameter list!");
1106
1107 calibBodySetExtended.set(paramIndexExt);
1108 calibBodySet.set(paramIndex);
1109 } else {
1110 calibBodySetExtended = transformExtended.getScalingParameters();
1111 calibBodySet = transform.getScalingParameters();
1112
1113 if (config.calibShape) {
1114 calibBodySetExtended |= transformExtended.getBlendShapeParameters();
1115 calibBodySet |= transform.getBlendShapeParameters();
1116 }
1117 }
1118
1119 // special trackingConfig for initialization: zero out smoothness and collision
1120 TrackingConfig trackingConfig{
1121 {config.minVisPercent, config.lossAlpha, config.maxIter, config.regularization, config.debug},
1122 0.0,
1123 0.0,
1124 1.0f,
1125 {},
1126 std::nullopt,
1127 config.meshConstraintWeight,
1128 {}};
1129
1130 // only keep one motion; no need to duplicate.
1131 // identity information will be initialized and updated in the motion matrix throughout all the
1132 // solves.
1133 MatrixXf motion = MatrixXf::Zero(transformExtended.numAllModelParameters(), numFrames);
1134 std::vector<size_t> frameIndices;
1135
1136 { // Initialization
1137 MT_LOGI_IF(config.debug, "Solving for an initial pose and skeleton");
1138
1139 // first solve for initial tracking poses with fixed identity and locators to default
1140 // Because we are solving for poses only, use character to save compute.
1141 if (config.greedySampling > 0) {
1142 // first only track the first frame
1143 MT_LOGI_IF(config.debug, "Pre-solving for the first frame");
1144 std::vector<size_t> firstFrame;
1145 firstFrame.push_back(0);
1146 motion.topRows(transform.numAllModelParameters()) = trackPosesForFrames(
1147 markerData,
1148 character,
1149 motion.topRows(transform.numAllModelParameters()),
1150 trackingConfig,
1151 firstFrame,
1152 false, // Not continuous for calibration keyframes
1153 {},
1154 {},
1155 std::nullopt,
1156 "Calibrating first frame");
1157 motion.topRows(transform.numAllModelParameters()) = trackSequence(
1158 markerData,
1159 character,
1160 calibBodySet,
1161 motion.topRows(transform.numAllModelParameters()),
1162 trackingConfig,
1163 firstFrame,
1164 regularizerWeights.at(0),
1165 config.enforceFloorInFirstFrame,
1166 config.firstFramePoseConstraintSet,
1167 config.targetHeightCm); // still solving a subset
1168 std::tie(identity.v, character.locators, character.skinnedLocators) =
1169 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
1170
1171 // track sequence with selected stride. we need to sample at least config.calibFrames frames
1172 // so make sure we have a stride that allows that
1173 const size_t sampleStride =
1174 computeSampleStride(numFrames, config.calibFrames, config.greedySampling);
1175
1176 motion.topRows(transform.numAllModelParameters()) =
1177 trackPosesPerframe(markerData, character, identity, trackingConfig, sampleStride);
1178
1179 const ParameterSet ps = transformExtended.getPoseParameters() &
1180 ~transformExtended.getRigidParameters() & ~locatorSet;
1181 frameIndices = sampleFrames(
1182 character,
1183 motion.topRows(transform.numAllModelParameters()),
1184 markerData,
1185 ps,
1186 sampleStride,
1187 config.calibFrames);
1188
1189 // then solve for identity and poses with fixed locators, initialized with solved poses
1190 // this works using "character" because additional parameters for the locators are appended at
1191 // the end, so the indices work out using topRows() without special treatment.
1192 motion.topRows(transform.numAllModelParameters()) = trackSequence(
1193 markerData,
1194 character,
1195 calibBodySet, // only solve for identity and not markers
1196 motion.topRows(transform.numAllModelParameters()),
1197 trackingConfig,
1198 frameIndices,
1199 regularizerWeights.at(0), // note: ideally allow large change at initialization with 0
1200 // or low regularization weight
1201 config.enforceFloorInFirstFrame,
1202 config.firstFramePoseConstraintSet,
1203 config.targetHeightCm);
1204 } else {
1205 motion.topRows(transform.numAllModelParameters()) =
1206 trackPosesPerframe(markerData, character, identity, trackingConfig, frameStride);
1207
1208 // then solve for identity and poses with fixed locators, initialized with solved poses
1209 // this works using "character" because additional parameters for the locators are appended at
1210 // the end, so the indices work out using topRows() without special treatment.
1211 motion.topRows(transform.numAllModelParameters()) = trackSequence(
1212 markerData,
1213 character,
1214 calibBodySet, // only solve for identity and not markers
1215 motion.topRows(transform.numAllModelParameters()),
1216 trackingConfig,
1217 regularizerWeights.at(0), // note: ideally allow large change at initialization with 0
1218 // or low regularization weight
1219 frameStride,
1220 config.enforceFloorInFirstFrame,
1221 config.firstFramePoseConstraintSet,
1222 config.targetHeightCm);
1223 }
1224 }
1225
1226 // Solve everything together for a few iterations
1227 for (size_t iIter = 0; iIter < config.majorIter; ++iIter) {
1228 MT_LOGI_IF(config.debug, "Iteration {} of calibration", iIter);
1229
1230 if (config.greedySampling > 0) {
1231 motion = trackSequence(
1232 markerData,
1233 solvingCharacter,
1234 locatorSet | calibBodySetExtended | gloveSet,
1235 motion,
1236 trackingConfig,
1237 frameIndices,
1238 regularizerWeights.at(
1239 1), // note: ideally use a small regularization to prevent too large a change
1240 config.enforceFloorInFirstFrame,
1241 config.firstFramePoseConstraintSet,
1242 config.targetHeightCm,
1243 leftGloveData,
1244 rightGloveData,
1245 gloveConfig); // still solving a subset
1246 } else {
1247 motion = trackSequence(
1248 markerData,
1249 solvingCharacter,
1250 locatorSet | calibBodySetExtended | gloveSet,
1251 motion,
1252 trackingConfig,
1253 regularizerWeights.at(
1254 1), // note: ideally use a small regularization to prevent too large a change
1255 frameStride,
1256 config.enforceFloorInFirstFrame,
1257 config.firstFramePoseConstraintSet,
1258 config.targetHeightCm,
1259 leftGloveData,
1260 rightGloveData,
1261 gloveConfig); // still solving a subset
1262 }
1263 // extract solving results to identity and character so we can pass them to trackPosesPerframe
1264 // below.
1265 std::tie(identity.v, character.locators, character.skinnedLocators) =
1266 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
1267 if (config.calibShape && solvingCharacter.blendShape) {
1268 auto blendShapeParams =
1269 extractBlendWeights(solvingCharacter.parameterTransform, ModelParameters(motion.col(0)));
1270 MT_LOGI("Solved for blend shape coeffs: {}", blendShapeParams.v.transpose());
1271 // NOTE: We intentionally do NOT bake the mesh here. The caller
1272 // (Python) is responsible for calling bake_blend_shape() which
1273 // both updates the mesh and strips the blend shape parameters
1274 // from the parameter transform.
1275 }
1276
1277 // The sequence solve above could get stuck with euler singularity but per-frame solve could get
1278 // it out. Pass in the first frame from previous solve as a better initial guess than the zero
1279 // pose.
1280 if (config.greedySampling > 0) {
1281 motion.topRows(transform.numAllModelParameters()) = trackPosesForFrames(
1282 markerData,
1283 character,
1284 motion.topRows(transform.numAllModelParameters()),
1285 trackingConfig,
1286 frameIndices,
1287 false, // Not continuous for calibration keyframes
1288 {},
1289 {},
1290 std::nullopt,
1291 "Calibrating per-frame");
1292 } else {
1293 const VectorXf initPose = motion.col(0).head(transform.numAllModelParameters());
1294 motion.topRows(transform.numAllModelParameters()) =
1295 trackPosesPerframe(markerData, character, initPose, trackingConfig, frameStride);
1296 }
1297 }
1298
1299 // Finally, fine tune marker offsets with fix identity.
1300 MT_LOGI_IF(config.debug, "Fine-tune marker offsets");
1301
1302 if (config.greedySampling > 0) {
1303 motion = trackSequence(
1304 markerData,
1305 solvingCharacter,
1306 locatorSet | gloveSet,
1307 motion,
1308 trackingConfig,
1309 frameIndices,
1310 regularizerWeights.at(
1311 2), // note: ideally use a higher regularizer weight to prevent too large a change
1312 config.enforceFloorInFirstFrame,
1313 config.firstFramePoseConstraintSet,
1314 config.targetHeightCm,
1315 leftGloveData,
1316 rightGloveData,
1317 gloveConfig);
1318 } else {
1319 motion = trackSequence(
1320 markerData,
1321 solvingCharacter,
1322 locatorSet | gloveSet,
1323 motion,
1324 trackingConfig,
1325 regularizerWeights.at(
1326 2), // note: ideally use a higher regularizer weight to prevent too large a change
1327 frameStride,
1328 config.enforceFloorInFirstFrame,
1329 config.firstFramePoseConstraintSet,
1330 config.targetHeightCm,
1331 leftGloveData,
1332 rightGloveData,
1333 gloveConfig);
1334 }
1335 std::tie(identity.v, character.locators, character.skinnedLocators) =
1336 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
1337
1338 // Bake solved glove offsets into the character skeleton, following the same
1339 // pattern as locator extraction above. No-op if gloveConfig is nullopt.
1340 bakeGloveOffsetsFromParams(
1341 character, ModelParameters(motion.col(0)), solvingCharacter, gloveConfig);
1342
1343 // TODO: A hack to return the solved first frame as initialization for tracking later.
1344 identity.v = motion.col(0).head(transform.numAllModelParameters());
1345
1346 // Log final per-locator mesh distances
1347 logSkinnedLocatorMeshDistances(character, "Final");
1348
1349 // Log the calibrated character height
1350 const float height = computeCharacterHeight(character, identity);
1351 MT_LOGI("Calibrated character height: {:.4f} cm", height);
1352}

1399 0.0,
1400 0.0,

1407 // only keep one motion for both character and solvingCharacter; no need to duplicate.

1448 // solutions.

1450 markerData,
1451 character,
1452 motion.topRows(transform.numAllModelParameters()),
1453 trackingConfig,
1454 frameIndices,
1455 false, // Not continuous for calibration keyframes
1456 {},
1457 {},
1458 std::nullopt,
1459 "Calibrating locators per-frame");

1464 markerData,
1465 solvingCharacter,
1466 locatorSet,
1467 motion,
1468 trackingConfig,
1469 frameIndices,
1470 0.0,
1471 config.enforceFloorInFirstFrame,
1472 config.firstFramePoseConstraintSet);
1473 // Extract solved locators
1474 fullParams.pose = motion.col(0);
1475 character.locators = extractLocatorsFromCharacter(solvingCharacter, fullParams);
1476 }

1553std::pair<float, float> getLocatorError(
1554 std::span<const std::vector<momentum::Marker>> markerData,
1555 const MatrixXf& motion,
1556 momentum::Character& character) {
1557 const size_t numFrames = markerData.size();
1558 MT_CHECK(numFrames > 0, "Input marker data is empty.");
1559 MT_CHECK(
1560 motion.cols() >= numFrames,
1561 "Motion has {} columns but marker data has {} frames",
1562 motion.cols(),
1563 numFrames);
1564 MT_CHECK(
1565 motion.rows() == character.parameterTransform.numAllModelParameters(),
1566 "Input motion parameters {} do not match character model parameters {}",
1567 motion.rows(),
1568 character.parameterTransform.numAllModelParameters());
1569
1570 const ParameterTransform& pt = character.parameterTransform;
1571
1572 // map locator name to its index
1573 std::map<std::string, size_t> locatorLookup;
1574 for (size_t i = 0; i < character.locators.size(); i++) {
1575 locatorLookup[character.locators[i].name] = i;
1576 }
1577
1578 // map skinned locator name to its index
1579 std::unordered_map<std::string, size_t> skinnedLocatorLookup;
1580 for (size_t i = 0; i < character.skinnedLocators.size(); i++) {
1581 skinnedLocatorLookup[character.skinnedLocators[i].name] = i;
1582 }
1583
1584 SkeletonState state;
1585
1586 // go over all frames and pose the locators and compute the error
1587 double error = 0.0;
1588 double maxError = 0.0;
1589 size_t frameNum = 0.0;
1590 std::string markerName;
1591 for (size_t iFrame = 0; iFrame < numFrames; ++iFrame) {
1592 const auto jointParams = pt.apply(motion.col(iFrame));
1593 state.set(jointParams, character.skeleton, false);
1594
1595 double frameError = 0.0;
1596
1597 const auto& markerList = markerData[iFrame];
1598 size_t validMarkers = 0;
1599 for (const auto& jMarker : markerList) {
1600 if (jMarker.occluded) {
1601 continue;
1602 }
1603
1604 std::optional<Vector3f> locatorPos;
1605
1606 // First try regular locators
1607 auto query = locatorLookup.find(jMarker.name);
1608 if (query != locatorLookup.end()) {
1609 size_t locatorIdx = query->second;
1610 if (locatorIdx < character.locators.size()) {
1611 const auto& locator = character.locators[locatorIdx];
1612 if (locator.parent < state.jointState.size()) {
1613 locatorPos = state.jointState[locator.parent].transform * locator.offset;
1614 }
1615 }
1616 }
1617
1618 // If not found, try skinned locators
1619 if (!locatorPos.has_value()) {
1620 auto skinnedQuery = skinnedLocatorLookup.find(jMarker.name);
1621 if (skinnedQuery != skinnedLocatorLookup.end()) {
1622 size_t locatorIdx = skinnedQuery->second;
1623 if (locatorIdx < character.skinnedLocators.size()) {
1624 const auto& skinnedLocator = character.skinnedLocators[locatorIdx];
1625 locatorPos = getSkinnedLocatorPosition(
1626 skinnedLocator, skinnedLocator.position, character.inverseBindPose, state);
1627 }
1628 }
1629 }
1630
1631 if (!locatorPos.has_value()) {
1632 continue;
1633 }
1634
1635 const Vector3f diff = *locatorPos - jMarker.pos.cast<float>();
1636 const float markerError = diff.norm();
1637 frameError += markerError;
1638 if (markerError > maxError) {
1639 maxError = markerError;
1640 frameNum = iFrame;
1641 markerName = jMarker.name;
1642 }
1643 validMarkers++;
1644 }
1645
1646 if (validMarkers > 0) {
1647 error += frameError / validMarkers;
1648 }
1649 }
1650 MT_LOGI("Max marker error: {} at frame {} for marker {}", maxError, frameNum, markerName);
1651 return {static_cast<float>(error / numFrames), static_cast<float>(maxError)};
1652}
