15template <typename T>
16void MppcaT<T>::set(
17 const VectorX<T>& inPi,
18 const MatrixX<T>& mmu,
19 momentum::span<const MatrixX<T>> W,
20 const VectorX<T>& sigma2) {
21 using CovarianceMatrix = LowRankCovarianceMatrixT<T>;
22
23 d = mmu.cols();
24 p = sigma2.rows();
25
26 // TODO: Validate input dimensions: inPi.rows() == p, mmu.rows() == p, W.size() == p,
27 // and each W[c] has d columns. Currently mismatches will trigger out-of-bounds
28 // access or silent corruption rather than a clear error.
29 Rpre.setZero(p);
30 Cinv.resize(p);
31 L.resize(p);
32
33 CovarianceMatrix covariance;
34
35 // For each mixture component c, PPCA models the data covariance as
36 // C_c = sigma_c^2 * I + W_c * W_c^T,
37 // i.e. low-rank signal (W_c) plus isotropic noise (sigma_c^2). The log of the
38 // Gaussian normalizer used in the per-component log-density is
39 // -0.5 * d * log(2*pi) (the half_d_log_2pi term below).
40 const T half_d_log_2pi = T(0.5) * static_cast<T>(d) * std::log(twopi<T>());
41
42 for (size_t c = 0; c < p; c++) {
43 // LowRankCovarianceMatrix takes A such that C = sigma^2*I + A^T*A; here A = W_c^T,
44 // so A^T*A = W_c * W_c^T as required by the PPCA model. It produces the QR
45 // factorization R with R^T * R = sigma^2*I + W_c * W_c^T = C_c.
46 covariance.reset(std::sqrt(sigma2(c)), W[c].transpose());
47
48 // R is upper-triangular, so back-substituting the identity yields R^{-1}.
49 // L_tmp = R^{-T} (the lower-triangular factor of C^{-1} = R^{-1} * R^{-T}).
50 const Eigen::MatrixX<T> L_tmp = covariance.R()
51 .template triangularView<Eigen::Upper>()
52 .solve(Eigen::MatrixX<T>::Identity(d, d))
53 .transpose();
54 // Cinv = R^{-1} * R^{-T} = (R^T * R)^{-1} = C^{-1}.
55 Cinv[c] = covariance.R().template triangularView<Eigen::Upper>().solve(L_tmp);
56 L[c] = L_tmp;
57
58 const T C_logDeterminant = covariance.logDeterminant();
59
60 // Per-component log responsibility for input x is
61 // log r_c(x) = log pi_c - 0.5*log|C_c| - 0.5*d*log(2*pi) - 0.5*(x-mu_c)^T C_c^{-1}
62 // (x-mu_c).
63 // Rpre stores the data-independent prefactor (everything but the quadratic term),
64 // so callers can add the cached -0.5 * Mahalanobis distance at evaluation time
65 // and apply log-sum-exp across components for a numerically stable mixture log-likelihood.
66 Rpre(c) = std::log(inPi(c)) - T(0.5) * C_logDeterminant - half_d_log_2pi;
67 }
68 mu = mmu;
69
70 // Reset names to the data-space dimension; caller is expected to populate them
71 // afterwards if needed. Note: this discards any names previously set on the model.
72 names = std::vector<std::string>(d);
73}

75template <typename T, typename T2>
76std::vector<Eigen::MatrixX<T2>> castMatrices(momentum::span<const Eigen::MatrixX<T>> m_in) {
77 std::vector<Eigen::MatrixX<T2>> result;
78 result.reserve(m_in.size());
79 for (const auto& m : m_in) {
80 result.emplace_back(m.template cast<T2>());
81 }
82 return result;
83}

86template <typename T2>
87MppcaT<T2> MppcaT<T>::cast() const {
88 MppcaT<T2> result;
89 result.d = this->d;
90 result.p = this->p;
91 result.names = this->names;
92
93 result.mu = this->mu.template cast<T2>();
94 result.Cinv = castMatrices<T, T2>(this->Cinv);
95 result.L = castMatrices<T, T2>(this->L);
96 result.Rpre = this->Rpre.template cast<T2>();
97
98 return result;
99}

101template <typename T>
102bool MppcaT<T>::isApprox(const MppcaT<T>& mppcaT) const {
103 auto vecIsApprox = [](auto& vL, auto& vR) {
104 if (vL.size() != vR.size()) {
105 return false;
106 }
107 for (size_t i = 0; i < vL.size(); i++) {
108 if (!vL[i].isApprox(vR[i])) {
109 return false;
110 }
111 }
112 return true;
113 };
114 return (
115 (d == mppcaT.d) && (p == mppcaT.p) && (names == mppcaT.names) && mu.isApprox(mppcaT.mu) &&
116 vecIsApprox(Cinv, mppcaT.Cinv) && vecIsApprox(L, mppcaT.L) && Rpre.isApprox(mppcaT.Rpre));
117}
