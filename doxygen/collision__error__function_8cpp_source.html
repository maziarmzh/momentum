45template <typename T>
46void CollisionErrorFunctionT<T>::updateCollisionPairs(bool filterRestPoseOverlaps) {
47 validPairs_.clear();
48
49 const SkeletonStateT<T> state(
50 this->parameterTransform_.zero().template cast<T>(), this->skeleton_);
51 collisionState_.update(state, collisionGeometry_);
52
53 const auto n = collisionGeometry_.size();
54
55 aabbs_.resize(n);
56 for (size_t i = 0; i < n; ++i) {
57 aabbs_[i].id = gsl::narrow_cast<axel::Index>(i);
58 updateAabb(
59 aabbs_[i],
60 collisionState_.origin[i],
61 collisionState_.direction[i],
62 collisionState_.radius[i]);
63 }
64
65 if (n == 0) {
66 return;
67 }
68
69 T distance;
70 Vector2<T> cp;
71 T overlap;
72 for (size_t i = 0; i < n; ++i) {
73 for (size_t j = i + 1; j < n; ++j) {
74 // Skip pairs that already overlap in rest pose (unless disabled).
75 if (filterRestPoseOverlaps &&
76 overlaps(
77 collisionState_.origin[i],
78 collisionState_.direction[i],
79 collisionState_.radius[i],
80 collisionState_.delta[i],
81 collisionState_.origin[j],
82 collisionState_.direction[j],
83 collisionState_.radius[j],
84 collisionState_.delta[j],
85 distance,
86 cp,
87 overlap)) {
88 continue;
89 }
90
91 // Skip adjacent joints (common ancestor distance <= 1).
92 size_t p0 = collisionGeometry_[i].parent;
93 size_t p1 = collisionGeometry_[j].parent;
94 size_t count = 0;
95 while (p0 != p1) {
96 if (p0 > p1) {
97 p0 = this->skeleton_.joints[p0].parent;
98 } else {
99 p1 = this->skeleton_.joints[p1].parent;
100 }
101 count++;
102 }
103
104 if (count > 1) {
105 validPairs_.push_back({i, j, p0});
106 }
107 }
108 }
109}

127template <typename T>
128bool CollisionErrorFunctionT<T>::checkCollision(
129 const CollisionGeometryStateT<T>& colState,
130 size_t indexA,
131 size_t indexB,
132 T& distance,
133 Vector2<T>& cp,
134 T& overlap) const {
135 if (!aabbs_[indexA].intersects(aabbs_[indexB])) {
136 return false;
137 }
138 return overlaps(
139 colState.origin[indexA],
140 colState.direction[indexA],
141 colState.radius[indexA],
142 colState.delta[indexA],
143 colState.origin[indexB],
144 colState.direction[indexB],
145 colState.radius[indexB],
146 colState.delta[indexB],
147 distance,
148 cp,
149 overlap);
150}

152template <typename T>
153void CollisionErrorFunctionT<T>::accumulateGradientAlongChain(
154 const SkeletonStateT<T>& state,
155 const Vector3<T>& position,
156 const Vector3<T>& direction,
157 T weight,
158 size_t startJoint,
159 size_t stopJoint,
160 Ref<VectorX<T>> gradient) const {
161 size_t jointIndex = startJoint;
162 while (jointIndex != kInvalidIndex && jointIndex != stopJoint) {
163 const auto& jointState = state.jointState[jointIndex];
164 const size_t paramIndex = jointIndex * kParametersPerJoint;
165 const Vector3<T> posd = position - jointState.translation();
166
167 for (size_t d = 0; d < 3; d++) {
168 if (this->activeJointParams_[paramIndex + d]) {
169 const T val = direction.dot(jointState.getTranslationDerivative(d)) * weight;
170 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
171 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
172 ++index) {
173 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
174 val * this->parameterTransform_.transform.valuePtr()[index];
175 }
176 }
177 if (this->activeJointParams_[paramIndex + 3 + d]) {
178 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * weight;
179 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
180 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
181 ++index) {
182 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
183 val * this->parameterTransform_.transform.valuePtr()[index];
184 }
185 }
186 }
187 if (this->activeJointParams_[paramIndex + 6]) {
188 const T val = direction.dot(jointState.getScaleDerivative(posd)) * weight;
189 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
190 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
191 ++index) {
192 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
193 val * this->parameterTransform_.transform.valuePtr()[index];
194 }
195 }
196
197 jointIndex = this->skeleton_.joints[jointIndex].parent;
198 }

251template <typename T>
252std::vector<Vector2i> CollisionErrorFunctionT<T>::getCollisionPairs() const {
253 std::vector<Vector2i> collidingPairs;
254 for (const auto& pair : validPairs_) {
255 T distance;
256 Vector2<T> cp;
257 T overlap;
258 if (checkCollision(collisionState_, pair.indexA, pair.indexB, distance, cp, overlap)) {
259 collidingPairs.emplace_back(static_cast<int>(pair.indexA), static_cast<int>(pair.indexB));
260 }
261 }
262 return collidingPairs;
263}

265template <typename T>
266double CollisionErrorFunctionT<T>::getError(
267 const ModelParametersT<T>& /* params */,
268 const SkeletonStateT<T>& state,
269 const MeshStateT<T>& /* meshState */) {
270 if (state.jointState.empty()) {
271 return 0.0;
272 }
273
274 MT_CHECK(
275 state.jointParameters.size() ==
276 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
277
278 computeBroadPhase(state);
279
280 double error = 0;
281 size_t collidingCount = 0;
282 for (const auto& pair : validPairs_) {
283 T distance;
284 Vector2<T> cp;
285 T overlap;
286 if (!checkCollision(collisionState_, pair.indexA, pair.indexB, distance, cp, overlap)) {
287 continue;
288 }
289 collidingCount++;
290 error += sqr(overlap) * kCollisionWeight * this->weight_;
291 }
292
293 MT_LOGD(
294 "CollisionError: {} colliding / {} valid pairs, error={:.8f}, weight={:.4f}",
295 collidingCount,
296 validPairs_.size(),
297 error,
298 this->weight_);
299
300 return error;
301}

303template <typename T>
304double CollisionErrorFunctionT<T>::getGradient(
305 const ModelParametersT<T>& /* params */,
306 const SkeletonStateT<T>& state,
307 const MeshStateT<T>& /* meshState */,
308 Ref<VectorX<T>> gradient) {
309 if (state.jointState.empty()) {
310 return 0.0;
311 }
312
313 MT_CHECK(
314 state.jointParameters.size() ==
315 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
316
317 computeBroadPhase(state);
318
319 double error = 0;
320 for (const auto& pair : validPairs_) {
321 T distance;
322 Vector2<T> cp;
323 T overlap;
324 if (!checkCollision(collisionState_, pair.indexA, pair.indexB, distance, cp, overlap)) {
325 continue;
326 }
327
328 error += sqr(overlap) * kCollisionWeight * this->weight_;
329
330 const Vector3<T> position_i =
331 collisionState_.origin[pair.indexA] + collisionState_.direction[pair.indexA] * cp[0];
332 const Vector3<T> position_j =
333 collisionState_.origin[pair.indexB] + collisionState_.direction[pair.indexB] * cp[1];
334 const Vector3<T> direction = position_i - position_j;
335 const T wgt = -T(2) * kCollisionWeight * this->weight_ * (overlap / distance);
336
337 accumulateGradientAlongChain(
338 state,
339 position_i,
340 direction,
341 wgt,
342 collisionGeometry_[pair.indexA].parent,
343 pair.commonAncestor,
344 gradient);
345 accumulateGradientAlongChain(
346 state,
347 position_j,
348 direction,
349 -wgt,
350 collisionGeometry_[pair.indexB].parent,
351 pair.commonAncestor,
352 gradient);
353 }
354
355 return error;
356}

358template <typename T>
359double CollisionErrorFunctionT<T>::getJacobian(
360 const ModelParametersT<T>& /* params */,
361 const SkeletonStateT<T>& state,
362 const MeshStateT<T>& /* meshState */,
363 Ref<MatrixX<T>> jacobian,
364 Ref<VectorX<T>> residual,
365 int& usedRows) {
366 MT_PROFILE_FUNCTION();
367 MT_CHECK(
368 jacobian.rows() >= gsl::narrow<Eigen::Index>(getJacobianSize()),
369 "Jacobian rows mismatch: Actual {}, Expected {}",
370 jacobian.rows(),
371 gsl::narrow<Eigen::Index>(getJacobianSize()));
372 MT_CHECK(
373 residual.rows() >= gsl::narrow<Eigen::Index>(getJacobianSize()),
374 "Residual rows mismatch: Actual {}, Expected {}",
375 residual.rows(),
376 gsl::narrow<Eigen::Index>(getJacobianSize()));
377
378 if (state.jointState.empty()) {
379 return 0.0;
380 }
381
382 MT_CHECK(
383 state.jointParameters.size() ==
384 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
385
386 computeBroadPhase(state);
387
388 const T wgt = std::sqrt(kCollisionWeight * this->weight_);
389 int pos = 0;
390 double error = 0;
391
392 for (const auto& pair : validPairs_) {
393 T distance;
394 Vector2<T> cp;
395 T overlap;
396 if (!checkCollision(collisionState_, pair.indexA, pair.indexB, distance, cp, overlap)) {
397 continue;
398 }
399
400 error += sqr(overlap) * kCollisionWeight * this->weight_;
401
402 const Vector3<T> position_i =
403 collisionState_.origin[pair.indexA] + collisionState_.direction[pair.indexA] * cp[0];
404 const Vector3<T> position_j =
405 collisionState_.origin[pair.indexB] + collisionState_.direction[pair.indexB] * cp[1];
406 const Vector3<T> direction = position_i - position_j;
407 const T fac = T(2) / distance * wgt;
408 const int row = pos++;
409
410 accumulateJacobianAlongChain(
411 state,
412 position_i,
413 direction,
414 -fac,
415 collisionGeometry_[pair.indexA].parent,
416 pair.commonAncestor,
417 jacobian,
418 row);
419 accumulateJacobianAlongChain(
420 state,
421 position_j,
422 direction,
423 fac,
424 collisionGeometry_[pair.indexB].parent,
425 pair.commonAncestor,
426 jacobian,
427 row);
428
429 residual(row) = overlap * wgt;
430 }
431
432 usedRows = pos;
433 return error;
434}
