47template <typename T>
48void CollisionErrorFunctionT<T>::updateCollisionPairs(bool filterRestPoseOverlaps) {
49 validPairs_.clear();
50
51 const SkeletonStateT<T> state(
52 this->parameterTransform_.zero().template cast<T>(), this->skeleton_);
53 collisionState_.update(state, collisionGeometry_);
54
55 const auto n = collisionGeometry_.size();
56
57 aabbs_.resize(n);
58 for (size_t i = 0; i < n; ++i) {
59 aabbs_[i].id = gsl::narrow_cast<axel::Index>(i);
60 updateAabb(
61 aabbs_[i],
62 collisionState_.origin[i],
63 collisionState_.direction[i],
64 collisionState_.radius[i]);
65 }
66
67 if (n == 0) {
68 return;
69 }
70
71 for (size_t i = 0; i < n; ++i) {
72 for (size_t j = i + 1; j < n; ++j) {
73 if (isValidCollisionPair(
74 collisionState_, collisionGeometry_, this->skeleton_, i, j, filterRestPoseOverlaps)) {
75 const size_t lca = this->skeleton_.commonAncestor(
76 collisionGeometry_[i].parent, collisionGeometry_[j].parent);
77 validPairs_.push_back({i, j, lca});
78 }
79 }
80 }
81}

99template <typename T>
100bool CollisionErrorFunctionT<T>::checkCollision(
101 const CollisionGeometryStateT<T>& colState,
102 size_t indexA,
103 size_t indexB,
104 T& distance,
105 Vector2<T>& cp,
106 T& overlap) const {
107 if (!aabbs_[indexA].intersects(aabbs_[indexB])) {
108 return false;
109 }
110 return overlaps(
111 colState.origin[indexA],
112 colState.direction[indexA],
113 colState.radius[indexA],
114 colState.delta[indexA],
115 colState.origin[indexB],
116 colState.direction[indexB],
117 colState.radius[indexB],
118 colState.delta[indexB],
119 distance,
120 cp,
121 overlap);
122}

124template <typename T>
125void CollisionErrorFunctionT<T>::accumulateGradientAlongChain(
126 const SkeletonStateT<T>& state,
127 const Vector3<T>& position,
128 const Vector3<T>& direction,
129 T weight,
130 size_t startJoint,
131 size_t stopJoint,
132 Ref<VectorX<T>> gradient,
133 T scaleCorrection) const {
134 size_t jointIndex = startJoint;
135 while (jointIndex != kInvalidIndex && jointIndex != stopJoint) {
136 const auto& jointState = state.jointState[jointIndex];
137 const size_t paramIndex = jointIndex * kParametersPerJoint;
138 const Vector3<T> posd = position - jointState.translation();
139
140 for (size_t d = 0; d < 3; d++) {
141 if (this->activeJointParams_[paramIndex + d]) {
142 const T val = direction.dot(jointState.getTranslationDerivative(d)) * weight;
143 gradient_jointParams_to_modelParams(
144 val, paramIndex + d, this->parameterTransform_, gradient);
145 }
146 if (this->activeJointParams_[paramIndex + 3 + d]) {
147 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * weight;
148 gradient_jointParams_to_modelParams(
149 val, paramIndex + 3 + d, this->parameterTransform_, gradient);
150 }
151 }
152 if (this->activeJointParams_[paramIndex + 6]) {
153 const T val = (direction.dot(jointState.getScaleDerivative(posd)) + scaleCorrection) * weight;
154 gradient_jointParams_to_modelParams(val, paramIndex + 6, this->parameterTransform_, gradient);
155 }
156
157 jointIndex = this->skeleton_.joints[jointIndex].parent;
158 }
159}

200template <typename T>
201std::vector<Vector2i> CollisionErrorFunctionT<T>::getCollisionPairs() const {
202 std::vector<Vector2i> collidingPairs;
203 for (const auto& pair : validPairs_) {
204 T distance;
205 Vector2<T> cp;
206 T overlap;
207 if (checkCollision(collisionState_, pair.indexA, pair.indexB, distance, cp, overlap)) {
208 collidingPairs.emplace_back(static_cast<int>(pair.indexA), static_cast<int>(pair.indexB));
209 }
210 }
211 return collidingPairs;
212}

214template <typename T>
215double CollisionErrorFunctionT<T>::getError(
216 const ModelParametersT<T>& /* params */,
217 const SkeletonStateT<T>& state,
218 const MeshStateT<T>& /* meshState */) {
219 if (state.jointState.empty()) {
220 return 0.0;
221 }
222
223 MT_CHECK(
224 state.jointParameters.size() ==
225 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
226
227 computeBroadPhase(state);
228
229 double error = 0;
230 size_t collidingCount = 0;
231 for (const auto& pair : validPairs_) {
232 T distance;
233 Vector2<T> cp;
234 T overlap;
235 if (!checkCollision(collisionState_, pair.indexA, pair.indexB, distance, cp, overlap)) {
236 continue;
237 }
238 collidingCount++;
239 error += sqr(overlap) * kCollisionWeight * this->weight_;
240 }
241
242 MT_LOGD(
243 "CollisionError: {} colliding / {} valid pairs, error={:.8f}, weight={:.4f}",
244 collidingCount,
245 validPairs_.size(),
246 error,
247 this->weight_);
248
249 return error;
250}

252template <typename T>
253double CollisionErrorFunctionT<T>::getGradient(
254 const ModelParametersT<T>& /* params */,
255 const SkeletonStateT<T>& state,
256 const MeshStateT<T>& /* meshState */,
257 Ref<VectorX<T>> gradient) {
258 if (state.jointState.empty()) {
259 return 0.0;
260 }
261
262 MT_CHECK(
263 state.jointParameters.size() ==
264 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
265
266 computeBroadPhase(state);
267
268 double error = 0;
269 for (const auto& pair : validPairs_) {
270 T distance;
271 Vector2<T> cp;
272 T overlap;
273 if (!checkCollision(collisionState_, pair.indexA, pair.indexB, distance, cp, overlap)) {
274 continue;
275 }
276
277 error += sqr(overlap) * kCollisionWeight * this->weight_;
278
279 const Vector3<T> position_i =
280 collisionState_.origin[pair.indexA] + collisionState_.direction[pair.indexA] * cp[0];
281 const Vector3<T> position_j =
282 collisionState_.origin[pair.indexB] + collisionState_.direction[pair.indexB] * cp[1];
283 const Vector3<T> direction = position_i - position_j;
284 const T wgt = -T(2) * kCollisionWeight * this->weight_ * (overlap / distance);
285
286 // Effective radii at the closest points (for scale derivative correction)
287 const T radiusA_at_cp =
288 collisionState_.radius[pair.indexA][0] + cp[0] * collisionState_.delta[pair.indexA];
289 const T radiusB_at_cp =
290 collisionState_.radius[pair.indexB][0] + cp[1] * collisionState_.delta[pair.indexB];
291
292 accumulateGradientAlongChain(
293 state,
294 position_i,
295 direction,
296 wgt,
297 collisionGeometry_[pair.indexA].parent,
298 pair.commonAncestor,
299 gradient,
300 -distance * radiusA_at_cp * ln2<T>());
301 accumulateGradientAlongChain(
302 state,
303 position_j,
304 direction,
305 -wgt,
306 collisionGeometry_[pair.indexB].parent,
307 pair.commonAncestor,
308 gradient,
309 distance * radiusB_at_cp * ln2<T>());
310
311 // Scale derivatives above the common ancestor: translation and rotation derivatives cancel
312 // above the LCA (both capsule endpoints move identically), but scale derivatives do not
313 // because the two endpoints are at different distances from the scaled joint, and scaling
314 // also changes capsule radii.
315 {
316 const T netScaleVal =
317 (direction.squaredNorm() - distance * (radiusA_at_cp + radiusB_at_cp)) * ln2<T>() * wgt;
318 size_t ancestorIndex = pair.commonAncestor;
319 while (ancestorIndex != kInvalidIndex) {
320 const size_t paramIndex = ancestorIndex * kParametersPerJoint;
321 if (this->activeJointParams_[paramIndex + 6]) {
322 gradient_jointParams_to_modelParams(
323 netScaleVal, paramIndex + 6, this->parameterTransform_, gradient);
324 }
325 ancestorIndex = this->skeleton_.joints[ancestorIndex].parent;
326 }
327 }
328 }
329
330 return error;
331}

333template <typename T>
334double CollisionErrorFunctionT<T>::getJacobian(
335 const ModelParametersT<T>& /* params */,
336 const SkeletonStateT<T>& state,
337 const MeshStateT<T>& /* meshState */,
338 Ref<MatrixX<T>> jacobian,
339 Ref<VectorX<T>> residual,
340 int& usedRows) {
341 MT_PROFILE_FUNCTION();
342 MT_CHECK(
343 jacobian.rows() >= gsl::narrow<Eigen::Index>(getJacobianSize()),
344 "Jacobian rows mismatch: Actual {}, Expected {}",
345 jacobian.rows(),
346 gsl::narrow<Eigen::Index>(getJacobianSize()));
347 MT_CHECK(
348 residual.rows() >= gsl::narrow<Eigen::Index>(getJacobianSize()),
349 "Residual rows mismatch: Actual {}, Expected {}",
350 residual.rows(),
351 gsl::narrow<Eigen::Index>(getJacobianSize()));
352
353 if (state.jointState.empty()) {
354 return 0.0;
355 }
356
357 MT_CHECK(
358 state.jointParameters.size() ==
359 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
360
361 computeBroadPhase(state);
362
363 const T wgt = std::sqrt(kCollisionWeight * this->weight_);
364 int pos = 0;
365 double error = 0;
366
367 for (const auto& pair : validPairs_) {
368 T distance;
369 Vector2<T> cp;
370 T overlap;
371 if (!checkCollision(collisionState_, pair.indexA, pair.indexB, distance, cp, overlap)) {
372 continue;
373 }
374
375 error += sqr(overlap) * kCollisionWeight * this->weight_;
376
377 const Vector3<T> position_i =
378 collisionState_.origin[pair.indexA] + collisionState_.direction[pair.indexA] * cp[0];
379 const Vector3<T> position_j =
380 collisionState_.origin[pair.indexB] + collisionState_.direction[pair.indexB] * cp[1];
381 const Vector3<T> direction = position_i - position_j;
382 const T fac = wgt / distance;
383 const int row = pos++;
384
385 // Effective radii at the closest points (for scale derivative correction)
386 const T radiusA_at_cp =
387 collisionState_.radius[pair.indexA][0] + cp[0] * collisionState_.delta[pair.indexA];
388 const T radiusB_at_cp =
389 collisionState_.radius[pair.indexB][0] + cp[1] * collisionState_.delta[pair.indexB];
390
391 accumulateJacobianAlongChain(
392 state,
393 position_i,
394 direction,
395 -fac,
396 collisionGeometry_[pair.indexA].parent,
397 pair.commonAncestor,
398 jacobian,
399 row,
400 -distance * radiusA_at_cp * ln2<T>());
401 accumulateJacobianAlongChain(
402 state,
403 position_j,
404 direction,
405 fac,
406 collisionGeometry_[pair.indexB].parent,
407 pair.commonAncestor,
408 jacobian,
409 row,
410 distance * radiusB_at_cp * ln2<T>());
411
412 // Scale derivatives above the common ancestor
413 {
414 const T netScaleVal = -fac * ln2<T>() * direction.squaredNorm() +
415 wgt * (radiusA_at_cp + radiusB_at_cp) * ln2<T>();
416 size_t ancestorIndex = pair.commonAncestor;
417 while (ancestorIndex != kInvalidIndex) {
418 const size_t paramIndex = ancestorIndex * kParametersPerJoint;
419 if (this->activeJointParams_[paramIndex + 6]) {
420 jacobian_jointParams_to_modelParams(
421 netScaleVal, paramIndex + 6, row, this->parameterTransform_, jacobian);
422 }
423 ancestorIndex = this->skeleton_.joints[ancestorIndex].parent;
424 }
425 }
426
427 residual(row) = overlap * wgt;
428 }
429
430 usedRows = pos;
431 return error;
432}
