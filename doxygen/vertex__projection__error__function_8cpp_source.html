27template <typename T>
28void VertexProjectionErrorFunctionT<T>::evalFunction(
29 size_t constrIndex,
30 const SkeletonStateT<T>& /*state*/,
31 const MeshStateT<T>& /*meshState*/,
32 std::span<const Eigen::Vector3<T>> worldVecs,
33 FuncType& f,
34 std::span<DfdvType> dfdv) const {
35 MT_CHECK(constrIndex < this->constraints_.size(), "Constraint index out of range");
36 MT_CHECK(worldVecs.size() == 1, "Expected exactly one world vector for vertex projection");
37
38 const auto& constraint = this->constraints_[constrIndex];
39 const Eigen::Vector3<T>& v = worldVecs[0];
40
41 // Project 3D point to 2D using the projection matrix
42 // p_hom = P * [v; 1]
43 Eigen::Vector4<T> vHom;
44 vHom << v, T(1);
45 const Eigen::Vector3<T> pHom = constraint.projectionMatrix * vHom;
46
47 // Near-clip: skip vertices behind the camera (matching legacy _nearClip = 1.0)
48 if (pHom(2) < T(1)) {
49 f.setZero();
50 return;
51 }
52
53 // Perspective divide
54 const T invZ = T(1) / pHom(2);
55 const Eigen::Vector2<T> projected(pHom(0) * invZ, pHom(1) * invZ);
56
57 // Residual: f = projected - target
58 f = projected - constraint.target;
59
60 // Derivative: df/dv = d(p/w)/dv where p = Proj * v_hom
61 if (!dfdv.empty()) {
62 MT_CHECK(dfdv.size() == 1, "Expected exactly one dfdv for vertex projection");
63
64 // df/dv = [d(x/z)/dv; d(y/z)/dv]
65 // Using quotient rule: d(a/b)/dx = (da/dx * b - a * db/dx) / b^2
66 const T invZ2 = invZ * invZ;
67
68 // P[:3, :3] is the 3x3 submatrix of projection matrix
69 const Eigen::Matrix<T, 3, 3> P33 = constraint.projectionMatrix.template leftCols<3>();
70
71 // df/dv row 0: d(x/z)/dv = (P[0,:3] * z - x * P[2,:3]) / z^2
72 // df/dv row 1: d(y/z)/dv = (P[1,:3] * z - y * P[2,:3]) / z^2
73 dfdv[0].row(0) = (P33.row(0) * pHom(2) - pHom(0) * P33.row(2)) * invZ2;
74 dfdv[0].row(1) = (P33.row(1) * pHom(2) - pHom(1) * P33.row(2)) * invZ2;
75 }
76}
