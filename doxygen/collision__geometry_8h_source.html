120template <typename S>
121struct CollisionPrimitiveT {
122 using Scalar = S;
123
125 CollisionPrimitiveType type;
126
128 TransformT<S> transformation;
129
131 Vector2<S> radius;
132
134 Vector3<S> ellipsoidRadii;
135
137 size_t parent;
138
140 S length;
141
142 CollisionPrimitiveT()
143 : type(CollisionPrimitiveType::TaperedCapsule),
144 transformation(TransformT<S>()),
145 radius(Vector2<S>::Zero()),
146 ellipsoidRadii(Vector3<S>::Zero()),
147 parent(kInvalidIndex),
148 length(S(0)) {
149 // Empty
150 }
151
153 /* implicit */ CollisionPrimitiveT(const TaperedCapsuleT<S>& capsule)
154 : type(CollisionPrimitiveType::TaperedCapsule),
155 transformation(capsule.transformation),
156 radius(capsule.radius),
157 ellipsoidRadii(Vector3<S>::Zero()),
158 parent(capsule.parent),
159 length(capsule.length) {}
160
162 /* implicit */ CollisionPrimitiveT(const CollisionEllipsoidT<S>& ellipsoid)
163 : type(CollisionPrimitiveType::Ellipsoid),
164 transformation(ellipsoid.transformation),
165 radius(Vector2<S>::Zero()),
166 ellipsoidRadii(ellipsoid.radii),
167 parent(ellipsoid.parent),
168 length(S(0)) {}
169
171 CollisionPrimitiveT& operator=(const TaperedCapsuleT<S>& capsule) {
172 type = CollisionPrimitiveType::TaperedCapsule;
173 transformation = capsule.transformation;
174 radius = capsule.radius;
175 ellipsoidRadii.setZero();
176 parent = capsule.parent;
177 length = capsule.length;
178 return *this;
179 }
180
182 CollisionPrimitiveT& operator=(const CollisionEllipsoidT<S>& ellipsoid) {
183 type = CollisionPrimitiveType::Ellipsoid;
184 transformation = ellipsoid.transformation;
185 radius.setZero();
186 ellipsoidRadii = ellipsoid.radii;
187 parent = ellipsoid.parent;
188 length = S(0);
189 return *this;
190 }
191
193 [[nodiscard]] bool isTaperedCapsule() const {
194 return type == CollisionPrimitiveType::TaperedCapsule;
195 }
196
198 [[nodiscard]] bool isEllipsoid() const {
199 return type == CollisionPrimitiveType::Ellipsoid;
200 }
201
206 [[nodiscard]] TaperedCapsuleT<S> toTaperedCapsule() const {
207 MT_CHECK(
208 isTaperedCapsule(),
209 "Collision primitive type {} does not store tapered capsule data",
210 static_cast<int>(type));
211
212 TaperedCapsuleT<S> capsule;
213 capsule.transformation = transformation;
214 capsule.radius = radius;
215 capsule.parent = parent;
216 capsule.length = length;
217 return capsule;
218 }
219
224 [[nodiscard]] CollisionEllipsoidT<S> toEllipsoid() const {
225 MT_CHECK(
226 isEllipsoid(),
227 "Collision primitive type {} does not store ellipsoid data",
228 static_cast<int>(type));
229
230 CollisionEllipsoidT<S> ellipsoid;
231 ellipsoid.transformation = transformation;
232 ellipsoid.radii = ellipsoidRadii;
233 ellipsoid.parent = parent;
234 return ellipsoid;
235 }
236
238 [[nodiscard]] bool isApprox(const CollisionPrimitiveT& other, const S& tol = Eps<S>(1e-4f, 1e-10))
239 const {
240 if (type != other.type || parent != other.parent) {
241 return false;
242 }
243
244 if (!transformation.isApprox(other.transformation, tol)) {
245 return false;
246 }
247
248 if (type == CollisionPrimitiveType::TaperedCapsule) {
249 return radius.isApprox(other.radius, tol) && ::momentum::isApprox(length, other.length, tol);
250 }
251
252 if (type == CollisionPrimitiveType::Ellipsoid) {
253 return ellipsoidRadii.isApprox(other.ellipsoidRadii, tol);
254 }
255
256 return false;
257 }
258};
