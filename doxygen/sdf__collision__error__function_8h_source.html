225inline size_t findCommonAncestorForVertex(
226 const SkinWeights& skinWeights,
227 const Skeleton& skeleton,
228 size_t vertexIndex,
229 size_t sdfParent) {
230 if (sdfParent == kInvalidIndex) {
231 return kInvalidIndex;
232 }
233 size_t minSkinnedJoint = kInvalidIndex;
234 for (uint32_t k = 0; k < kMaxSkinJoints; ++k) {
235 const auto w = skinWeights.weight(vertexIndex, k);
236 if (w > 0) {
237 const auto joint = skinWeights.index(vertexIndex, k);
238 if (minSkinnedJoint == kInvalidIndex || joint < minSkinnedJoint) {
239 minSkinnedJoint = joint;
240 }
241 }
242 }
243 if (minSkinnedJoint == kInvalidIndex) {
244 return kInvalidIndex;
245 }
246 return skeleton.commonAncestor(minSkinnedJoint, sdfParent);
247}

252SDFCollisionErrorFunctionT<T, SdfColliderType>::SDFCollisionErrorFunctionT(
253 const Character& character,

264 MT_CHECK_NOTNULL(

266 MT_CHECK(!sdfColliders_.empty(), "Must provide at least one SDF collider");
267 MT_CHECK(maxCollisionsPerVertex_ > 0, "maxCollisionsPerVertex must be positive");

273template <typename T, typename SdfColliderType>
274void SDFCollisionErrorFunctionT<T, SdfColliderType>::organizeVerticesByBone(
275 const std::vector<int>& participatingVertices,
276 const std::vector<T>& weights) {
277 std::vector<int> vertices;
278 if (participatingVertices.empty()) {
279 vertices.reserve(character_.mesh->vertices.size());
280 for (int i = 0; i < static_cast<int>(character_.mesh->vertices.size()); ++i) {
281 vertices.push_back(i);
282 }
283 } else {
284 vertices = participatingVertices;
285 }
286
287 for (int vertexIndex : vertices) {
288 MT_CHECK(
289 vertexIndex >= 0 && vertexIndex < static_cast<int>(character_.mesh->vertices.size()),
290 "Invalid vertex index: {} (mesh has {} vertices)",
291 vertexIndex,
292 character_.mesh->vertices.size());
293 }
294
295 std::vector<T> vertexWeights;
296 if (weights.empty()) {
297 vertexWeights.assign(vertices.size(), T(1));
298 } else {
299 MT_CHECK(
300 weights.size() == vertices.size(),
301 "Weight count ({}) must match vertex count ({})",
302 weights.size(),
303 vertices.size());
304 vertexWeights = weights;
305 }
306
307 struct VertexInfo {
308 size_t vertexIndex;
309 T weight;
310 size_t primaryBone;
311 };
312
313 std::vector<VertexInfo> vertexInfos;
314 vertexInfos.reserve(vertices.size());
315
316 MT_CHECK_NOTNULL(character_.skinWeights);
317 const auto& skinWeights = *character_.skinWeights;
318
319 for (size_t i = 0; i < vertices.size(); ++i) {
320 const int vertexIndex = vertices[i];
321 const T weight = vertexWeights[i];
322 const size_t primaryBone = skinWeights.index(vertexIndex, 0);
323 MT_CHECK(primaryBone < character_.skeleton.joints.size());
324 vertexInfos.push_back({static_cast<size_t>(vertexIndex), weight, primaryBone});
325 }
326
327 std::sort(vertexInfos.begin(), vertexInfos.end(), [](const VertexInfo& a, const VertexInfo& b) {
328 return a.primaryBone < b.primaryBone;
329 });
330
331 boneVertexRanges_.resize(character_.skeleton.joints.size());
332 activeVertexIndices_.reserve(vertices.size());
333 vertexWeights_.reserve(vertices.size());
334
335 size_t currentBone = 0;
336 for (size_t i = 0; i < vertexInfos.size(); ++i) {
337 const auto& info = vertexInfos[i];
338 while (currentBone < info.primaryBone) {
339 boneVertexRanges_[currentBone].end = i;
340 ++currentBone;
341 boneVertexRanges_[currentBone].start = i;
342 }
343 activeVertexIndices_.push_back(info.vertexIndex);
344 vertexWeights_.push_back(info.weight);
345 }
346
347 boneVertexRanges_[currentBone].end = activeVertexIndices_.size();
348 ++currentBone;
349 while (currentBone < boneVertexRanges_.size()) {
350 boneVertexRanges_[currentBone++] =
351 BoneVertexRange{activeVertexIndices_.size(), activeVertexIndices_.size()};
352 }
353}

355template <typename T, typename SdfColliderType>
356void SDFCollisionErrorFunctionT<T, SdfColliderType>::initializeBoneCollisionPairs(
357 bool filterRestPoseIntersections) {
358 std::unordered_set<size_t> activeBones;
359 for (size_t boneIndex = 0; boneIndex < boneVertexRanges_.size(); ++boneIndex) {
360 if (!boneVertexRanges_[boneIndex].empty()) {
361 activeBones.insert(boneIndex);
362 }
363 }
364
365 const SkeletonStateT<T> restState(
366 this->parameterTransform_.zero().template cast<T>(), this->skeleton_);
367 const ModelParametersT<T> restParams =
368 ModelParametersT<T>::Zero(this->parameterTransform_.numAllModelParameters());
369 updateColliders(restParams);
370
371 boneCollisionPairs_.clear();
372 boneCollisionPairs_.reserve(activeBones.size() * sdfColliders_.size());
373
374 for (size_t boneIndex : activeBones) {
375 if (boneVertexRanges_[boneIndex].empty()) {
376 continue;
377 }
378 for (size_t sdfIndex = 0; sdfIndex < sdfColliders_.size(); ++sdfIndex) {
379 const auto& collider = sdfColliders_[sdfIndex];
380 if (collider.parentJoint() != kInvalidIndex && filterRestPoseIntersections &&
381 testRestPoseIntersection(boneIndex, sdfIndex, restState)) {
382 continue;
383 }
384 boneCollisionPairs_.emplace_back(BoneCollisionPairT<T>{boneIndex, sdfIndex});
385 }
386 }
387}

389template <typename T, typename SdfColliderType>
390bool SDFCollisionErrorFunctionT<T, SdfColliderType>::testRestPoseIntersection(
391 size_t boneIndex,
392 size_t sdfColliderIndex,
393 const SkeletonStateT<T>& restState) const {
394 const auto& range = boneVertexRanges_[boneIndex];
395 if (range.empty()) {
396 return false;
397 }
398
399 const auto& collider = sdfColliders_[sdfColliderIndex];
400 if (!collider.isValid()) {
401 return false;
402 }
403
404 const auto worldToCollider = computeWorldToColliderTransform(sdfColliderIndex, restState);
405
406 for (size_t i = range.start; i < range.end; ++i) {
407 const size_t vertexIndex = activeVertexIndices_[i];
408 const auto& restPos = character_.mesh->vertices[vertexIndex].template cast<T>();
409 const Vector3<T> localPos = worldToCollider * restPos;
410 const T distance = collider.evaluate(localPos);
411 if (distance < T(0)) {
412 return true;
413 }
414 }
415 return false;
416}

426template <typename T, typename SdfColliderType>
427TransformT<T> SDFCollisionErrorFunctionT<T, SdfColliderType>::computeWorldToColliderTransform(
428 size_t sdfColliderIndex,
429 const SkeletonStateT<T>& state) const {
430 const auto& collider = sdfColliders_[sdfColliderIndex];
431 TransformT<T> parentTransform;
432 const auto parentJoint = collider.parentJoint();
433 if (parentJoint != kInvalidIndex) {
434 MT_CHECK(parentJoint < state.jointState.size(), "Invalid parent joint index: {}", parentJoint);
435 parentTransform = state.jointState[parentJoint].transform;
436 }
437 auto colliderWorldTransform =
438 parentTransform * collider.localToParentTransform().template cast<T>();
439 return colliderWorldTransform.inverse();
440}

442template <typename T, typename SdfColliderType>
443double SDFCollisionErrorFunctionT<T, SdfColliderType>::getError(
444 const ModelParametersT<T>& params,
445 const SkeletonStateT<T>& state,
446 const MeshStateT<T>& meshState) {
447 MT_PROFILE_FUNCTION();
448 MT_CHECK_NOTNULL(meshState.posedMesh_);
449 updateColliders(params);
450 computeActivePairs(meshState, state);
451
452 double totalError = 0.0;
453 collisionCounts_.assign(activeVertexIndices_.size(), 0);
454
455 for (size_t pairIndex = 0; pairIndex < boneCollisionPairs_.size(); ++pairIndex) {
456 if (!activePairs_[pairIndex]) {
457 continue;
458 }
459 const auto& pair = boneCollisionPairs_[pairIndex];
460 const auto& range = boneVertexRanges_[pair.boneIndex];
461 if (range.empty()) {
462 continue;
463 }
464 const auto& collider = sdfColliders_[pair.sdfColliderIndex];
465 if (!collider.isValid()) {
466 continue;
467 }
468 const auto worldToCollider = computeWorldToColliderTransform(pair.sdfColliderIndex, state);
469
470 for (size_t i = range.start; i < range.end; ++i) {
471 if (collisionCounts_[i] >= maxCollisionsPerVertex_) {
472 continue;
473 }
474 const size_t vertexIndex = activeVertexIndices_[i];
475 const T vertexWeight = vertexWeights_[i];
476 const auto& worldPos = meshState.posedMesh_->vertices[vertexIndex];
477 const Vector3<T> localPos = worldToCollider * worldPos;
478 const T distance = collider.evaluate(localPos);
479
480 if (distance < T(0)) {
481 collisionCounts_[i]++;
482 const T penetrationDepth = -distance;
483 totalError += static_cast<double>(
484 vertexWeight * penetrationDepth * penetrationDepth * kSDFCollisionWeight *
485 this->weight_);
486 }
487 }
488 }
489 return totalError;
490}

492template <typename T, typename SdfColliderType>
493void SDFCollisionErrorFunctionT<T, SdfColliderType>::accumulateJointGradient(
494 const JointStateT<T>& jointState,
495 size_t paramIndex,
496 const Eigen::Vector3<T>& posd,
497 const Eigen::Vector3<T>& direction,
498 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
499 for (size_t d = 0; d < 3; d++) {
500 if (this->activeJointParams_[paramIndex + d]) {
501 gradient_jointParams_to_modelParams(
502 direction.dot(jointState.getTranslationDerivative(d)),
503 paramIndex + d,
504 this->parameterTransform_,
505 gradient);
506 }
507 if (this->activeJointParams_[paramIndex + 3 + d]) {
508 gradient_jointParams_to_modelParams(
509 direction.dot(jointState.getRotationDerivative(d, posd)),
510 paramIndex + 3 + d,
511 this->parameterTransform_,
512 gradient);
513 }
514 }
515 if (this->activeJointParams_[paramIndex + 6]) {
516 gradient_jointParams_to_modelParams(
517 direction.dot(jointState.getScaleDerivative(posd)),
518 paramIndex + 6,
519 this->parameterTransform_,
520 gradient);
521 }
522}

556template <typename T, typename SdfColliderType>
557void SDFCollisionErrorFunctionT<T, SdfColliderType>::accumulateColliderHierarchyGradient(
558 const SkeletonStateT<T>& state,
559 size_t colliderParent,
560 size_t commonAncestor,
561 const Eigen::Vector3<T>& sdfSurfacePoint,
562 const Eigen::Vector3<T>& direction,
563 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
564 size_t jointIndex = colliderParent;
565 while (jointIndex != kInvalidIndex && jointIndex != commonAncestor) {
566 const auto& jointState = state.jointState[jointIndex];
567 const size_t paramIndex = jointIndex * kParametersPerJoint;
568 const Eigen::Vector3<T> posd = sdfSurfacePoint - jointState.translation();
569 accumulateJointGradient(jointState, paramIndex, posd, direction, gradient);
570 jointIndex = this->skeleton_.joints[jointIndex].parent;
571 }
572}

592template <typename T, typename SdfColliderType>
593double SDFCollisionErrorFunctionT<T, SdfColliderType>::getGradient(
594 const ModelParametersT<T>& params,
595 const SkeletonStateT<T>& state,
596 const MeshStateT<T>& meshState,
597 Eigen::Ref<Eigen::VectorX<T>> gradient) {
598 MT_PROFILE_FUNCTION();
599 MT_CHECK_NOTNULL(meshState.posedMesh_);
600 MT_CHECK_NOTNULL(meshState.restMesh_);
601 updateColliders(params);
602 computeActivePairs(meshState, state);
603
604 double totalError = 0.0;
605 collisionCounts_.assign(activeVertexIndices_.size(), 0);
606
607 for (size_t pairIndex = 0; pairIndex < boneCollisionPairs_.size(); ++pairIndex) {
608 if (!activePairs_[pairIndex]) {
609 continue;
610 }
611 const auto& pair = boneCollisionPairs_[pairIndex];
612 const auto& range = boneVertexRanges_[pair.boneIndex];
613 if (range.empty()) {
614 continue;
615 }
616 const auto& collider = sdfColliders_[pair.sdfColliderIndex];
617 if (!collider.isValid()) {
618 continue;
619 }
620 const auto worldToCollider = computeWorldToColliderTransform(pair.sdfColliderIndex, state);
621 const auto colliderToWorld = worldToCollider.inverse();
622
623 for (size_t i = range.start; i < range.end; ++i) {
624 if (collisionCounts_[i] >= maxCollisionsPerVertex_) {
625 continue;
626 }
627 const size_t vertexIndex = activeVertexIndices_[i];
628 const T vertexWeight = vertexWeights_[i];
629 const auto& worldPos = meshState.posedMesh_->vertices[vertexIndex];
630 const Vector3<T> localPos = worldToCollider * worldPos;
631 const auto [distance, localGradient] = collider.evaluateWithGradient(localPos);
632
633 if (distance < T(0)) {
634 const T penetrationDepth = -distance;
635 totalError += static_cast<double>(
636 vertexWeight * penetrationDepth * penetrationDepth * kSDFCollisionWeight *
637 this->weight_);
638 collisionCounts_[i]++;
639
640 // Transform gradient from collider local space to world space
641 const Eigen::Vector3<T> sdfGradient = colliderToWorld.rotation * localGradient;
642
643 const T wgt = vertexWeight * 2.0f * kSDFCollisionWeight * this->weight_;
644 const Eigen::Vector3<T> diff = -penetrationDepth * sdfGradient;
645 const Eigen::Vector3<T> sdfSurfacePoint = worldPos + penetrationDepth * sdfGradient;
646
647 const size_t commonAncestor = detail_sdf_collision::findCommonAncestorForVertex(
648 *character_.skinWeights, this->skeleton_, vertexIndex, collider.parentJoint());
649
650 SkinningWeightIteratorT<T> skinningIter(
651 character_, *meshState.restMesh_, state, vertexIndex);
652 while (!skinningIter.finished()) {
653 const auto [jointIndex, boneWeight, transformedVertex] = skinningIter.next();
654 if (jointIndex == commonAncestor) {
655 break;
656 }
657 if (std::abs(boneWeight) < std::numeric_limits<T>::epsilon()) {
658 continue;
659 }
660 MT_CHECK(jointIndex < this->skeleton_.joints.size());
661 const auto& jointState = state.jointState[jointIndex];
662 const size_t paramIndex = jointIndex * kParametersPerJoint;
663 const Eigen::Vector3<T> posd = transformedVertex - jointState.translation();
664 accumulateJointGradient(jointState, paramIndex, posd, boneWeight * wgt * diff, gradient);
665 }
666
667 if (collider.parentJoint() != kInvalidIndex) {
668 accumulateColliderHierarchyGradient(
669 state,
670 collider.parentJoint(),
671 commonAncestor,
672 sdfSurfacePoint,
673 -wgt * diff,
674 gradient);
675 }
676 }
677 }
678 }
679 return totalError;
680}

682template <typename T, typename SdfColliderType>
683double SDFCollisionErrorFunctionT<T, SdfColliderType>::getJacobian(
684 const ModelParametersT<T>& params,
685 const SkeletonStateT<T>& state,
686 const MeshStateT<T>& meshState,
687 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
688 Eigen::Ref<Eigen::VectorX<T>> residual,
689 int& usedRows) {
690 MT_PROFILE_FUNCTION();
691 MT_CHECK(
692 jacobian.cols() == static_cast<Eigen::Index>(this->parameterTransform_.transform.cols()),
693 "Jacobian column count mismatch");
694 MT_CHECK_NOTNULL(meshState.posedMesh_);
695 MT_CHECK_NOTNULL(meshState.restMesh_);
696 updateColliders(params);
697 computeActivePairs(meshState, state);
698
699 double totalError = 0.0;
700 int currentRow = 0;
701 collisionCounts_.assign(activeVertexIndices_.size(), 0);
702
703 for (size_t pairIndex = 0; pairIndex < boneCollisionPairs_.size(); ++pairIndex) {
704 if (!activePairs_[pairIndex]) {
705 continue;
706 }
707 const auto& pair = boneCollisionPairs_[pairIndex];
708 const auto& range = boneVertexRanges_[pair.boneIndex];
709 if (range.empty()) {
710 continue;
711 }
712 const auto& collider = sdfColliders_[pair.sdfColliderIndex];
713 if (!collider.isValid()) {
714 continue;
715 }
716 const auto worldToCollider = computeWorldToColliderTransform(pair.sdfColliderIndex, state);
717 const auto colliderToWorld = worldToCollider.inverse();
718
719 for (size_t i = range.start; i < range.end; ++i) {
720 if (collisionCounts_[i] >= maxCollisionsPerVertex_) {
721 continue;
722 }
723 const size_t vertexIndex = activeVertexIndices_[i];
724 const T vertexWeight = vertexWeights_[i];
725 const auto& worldPos = meshState.posedMesh_->vertices[vertexIndex];
726 const Vector3<T> localPos = worldToCollider * worldPos;
727 const auto [distance, localGradient] = collider.evaluateWithGradient(localPos);
728
729 if (distance < T(0)) {
730 const T penetrationDepth = -distance;
731 MT_CHECK(
732 currentRow < jacobian.rows() && currentRow < residual.rows(),
733 "Insufficient Jacobian/residual rows");
734
735 const T wgt = std::sqrt(vertexWeight * kSDFCollisionWeight * this->weight_);
736 residual[currentRow] = wgt * penetrationDepth;
737 totalError += static_cast<double>(
738 vertexWeight * penetrationDepth * penetrationDepth * kSDFCollisionWeight *
739 this->weight_);
740 ++collisionCounts_[i];
741
742 jacobian.row(currentRow).setZero();
743
744 // Transform gradient from collider local space to world space
745 const Eigen::Vector3<T> sdfGradient = colliderToWorld.rotation * localGradient;
746
747 const T jacobianMagnitude = -wgt;
748 const Eigen::Vector3<T> worldJacobianContribution = jacobianMagnitude * sdfGradient;
749 const Eigen::Vector3<T> sdfSurfacePoint = worldPos + penetrationDepth * sdfGradient;
750
751 const size_t commonAncestor = detail_sdf_collision::findCommonAncestorForVertex(
752 *character_.skinWeights, this->skeleton_, vertexIndex, collider.parentJoint());
753
754 SkinningWeightIteratorT<T> skinningIter(
755 character_, *meshState.restMesh_, state, vertexIndex);
756 while (!skinningIter.finished()) {
757 const auto [jointIndex, boneWeight, transformedVertex] = skinningIter.next();
758 if (jointIndex == commonAncestor) {
759 break;
760 }
761 if (std::abs(boneWeight) < std::numeric_limits<T>::epsilon()) {
762 continue;
763 }
764 MT_CHECK(jointIndex < this->skeleton_.joints.size());
765 const auto& jointState = state.jointState[jointIndex];
766 const size_t paramIndex = jointIndex * kParametersPerJoint;
767 const Eigen::Vector3<T> posd = transformedVertex - jointState.translation();
768 accumulateJointJacobian(
769 jointState,
770 paramIndex,
771 posd,
772 boneWeight * worldJacobianContribution,
773 jacobian.middleRows(currentRow, 1));
774 }
775
776 if (collider.parentJoint() != kInvalidIndex) {
777 accumulateColliderHierarchyJacobian(
778 state,
779 collider.parentJoint(),
780 commonAncestor,
781 sdfSurfacePoint,
782 -worldJacobianContribution,
783 jacobian.middleRows(currentRow, 1));
784 }
785 currentRow++;
786 }
787 }
788 }
789
790 usedRows = currentRow;
791 return totalError;
792}

799template <typename T, typename SdfColliderType>
800void SDFCollisionErrorFunctionT<T, SdfColliderType>::computeActivePairs(
801 const MeshStateT<T>& meshState,
802 const SkeletonStateT<T>& state) {
803 MT_PROFILE_FUNCTION();
804 activePairs_.assign(boneCollisionPairs_.size(), false);
805
806 boneBBoxCache_.resize(boneVertexRanges_.size());
807 boneBBoxValid_.assign(boneVertexRanges_.size(), false);
808 for (size_t boneIndex = 0; boneIndex < boneVertexRanges_.size(); ++boneIndex) {
809 if (!boneVertexRanges_[boneIndex].empty()) {
810 boneBBoxCache_[boneIndex] = computeBoneBoundingBox(boneIndex, *meshState.posedMesh_);
811 boneBBoxValid_[boneIndex] = true;
812 }
813 }
814
815 sdfBBoxCache_.resize(sdfColliders_.size());
816 for (size_t sdfIndex = 0; sdfIndex < sdfColliders_.size(); ++sdfIndex) {
817 sdfBBoxCache_[sdfIndex] = computeColliderWorldBoundingBox(sdfIndex, state);
818 }
819
820 for (size_t pairIndex = 0; pairIndex < boneCollisionPairs_.size(); ++pairIndex) {
821 const auto& pair = boneCollisionPairs_[pairIndex];
822 if (!boneBBoxValid_[pair.boneIndex]) {
823 continue;
824 }
825 activePairs_[pairIndex] =
826 boneBBoxCache_[pair.boneIndex].intersects(sdfBBoxCache_[pair.sdfColliderIndex]);
827 }
828}

831axel::BoundingBox<T> SDFCollisionErrorFunctionT<T, SdfColliderType>::computeBoneBoundingBox(
832 size_t boneIndex,
833 const MeshT<T>& posedMesh) const {
834 const auto& range = boneVertexRanges_[boneIndex];
835 if (range.empty()) {
836 return axel::BoundingBox<T>();

842 const size_t vertexIndex = activeVertexIndices_[i];
843 bbox.extend(posedMesh.vertices[vertexIndex]);
844 }

848template <typename T, typename SdfColliderType>
849axel::BoundingBox<T>
850SDFCollisionErrorFunctionT<T, SdfColliderType>::computeColliderWorldBoundingBox(
851 size_t sdfColliderIndex,
852 const SkeletonStateT<T>& state) const {
853 const auto& collider = sdfColliders_[sdfColliderIndex];
854 if (!collider.isValid()) {
855 return axel::BoundingBox<T>();
856 }
857
858 TransformT<T> colliderWorldTransform;
859 const auto parentJoint = collider.parentJoint();
860 if (parentJoint == kInvalidIndex) {
861 colliderWorldTransform = collider.localToParentTransform().template cast<T>();
862 } else {
863 MT_CHECK(parentJoint < state.jointState.size(), "Invalid parent joint index: {}", parentJoint);
864 colliderWorldTransform = state.jointState[parentJoint].transform *
865 collider.localToParentTransform().template cast<T>();
866 }
867
868 const auto& localBounds = collider.bounds();
869 const auto& minCorner = localBounds.min();
870 const auto& maxCorner = localBounds.max();
871
872 std::array<Eigen::Vector3<T>, 8> corners = {
873 {{minCorner.x(), minCorner.y(), minCorner.z()},
874 {maxCorner.x(), minCorner.y(), minCorner.z()},
875 {minCorner.x(), maxCorner.y(), minCorner.z()},
876 {maxCorner.x(), maxCorner.y(), minCorner.z()},
877 {minCorner.x(), minCorner.y(), maxCorner.z()},
878 {maxCorner.x(), minCorner.y(), maxCorner.z()},
879 {minCorner.x(), maxCorner.y(), maxCorner.z()},
880 {maxCorner.x(), maxCorner.y(), maxCorner.z()}}};
881
882 const auto firstWorldCorner = colliderWorldTransform * corners[0];
883 axel::BoundingBox<T> worldBounds(firstWorldCorner, T(0));
884 for (size_t i = 1; i < corners.size(); ++i) {
885 worldBounds.extend(colliderWorldTransform * corners[i]);
886 }
887 return worldBounds;
888}
