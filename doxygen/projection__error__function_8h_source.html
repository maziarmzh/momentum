35template <typename T>
36struct ProjectionConstraintDataT {
37 // Project a 3d point (after skinning) to (rxz, ryz, z) where
38 // (rx,ry) is the residual.
39 //
40 // This encodes the target 2d point, the view and camera transforms, and the
41 // camera distortion (if any).
42 //
43 // This projection model was copied from the hand IK FullProjectionConstraint.
44 // Ideally we would refactor somehow to make them share more code but
45 // the kinematic models are so different this would take some work.
46 Eigen::Matrix<T, 3, 4> projection; // Projection matrix
47
48 size_t parent{}; // parent joint of the constraint
49 Eigen::Vector3<T> offset = Eigen::Vector3<T>::Zero(); // relative offset to the parent
50 T weight = 1; // constraint weight
51
52 Eigen::Vector2<T> target = Eigen::Vector2<T>::Zero();
53
54 static ProjectionConstraintDataT<T> createFromLocator(const momentum::Locator& locator);
55};
