31Character addGloveBones(
32 Character character,
33 const GloveConfig& cfg,
34 const std::array<GloveOffset, 2>& offsets,
35 const std::string& prefix) {
36 auto& newSkel = character.skeleton;
37 auto& newTransform = character.parameterTransform;
38 auto& newInvBindPose = character.inverseBindPose;
39
40 for (size_t hand = 0; hand < 2; ++hand) {
41 const auto& wristName = cfg.wristJointNames[hand];
42
43 const size_t wristIdx = newSkel.getJointIdByName(wristName);
44 if (wristIdx == kInvalidIndex) {
45 MT_LOGW("addGloveBones: wrist joint '{}' not found, skipping", wristName);
46 continue;
47 }
48
49 // Skip if glove bone already exists
50 const std::string gloveBoneName = prefix + wristName;
51 if (newSkel.getJointIdByName(gloveBoneName) != kInvalidIndex) {
52 continue;
53 }
54
55 // Create a new joint as child of the wrist
56 Joint joint;
57 joint.name = gloveBoneName;
58 joint.parent = wristIdx;
59 joint.translationOffset = offsets[hand].translation;
60 joint.preRotation = eulerToQuaternion<float>(
61 offsets[hand].rotationEulerXYZ.cast<float>(), 0, 1, 2, EulerConvention::Intrinsic);
62
63 newSkel.joints.push_back(joint);
64 newInvBindPose.push_back(Affine3f::Identity());
65
66 // Resize offsets and activeJointParams for the new joint
67 newTransform.offsets.conservativeResize(newTransform.offsets.size() + kParametersPerJoint);
68 newTransform.activeJointParams.conservativeResize(
69 newTransform.activeJointParams.size() + kParametersPerJoint);
70 for (size_t j = newTransform.offsets.size() - kParametersPerJoint;
71 j < static_cast<size_t>(newTransform.offsets.size());
72 ++j) {
73 newTransform.offsets(j) = 0.0f;
74 }
75 for (size_t j = newTransform.activeJointParams.size() - kParametersPerJoint;
76 j < static_cast<size_t>(newTransform.activeJointParams.size());
77 ++j) {
78 newTransform.activeJointParams(j) = true;
79 }
80
81 // Resize the transform matrix rows for the new joint (no new columns)
82 const int newRows = static_cast<int>(newSkel.joints.size()) * kParametersPerJoint;
83 const int curCols = static_cast<int>(newTransform.name.size());
84 newTransform.transform.conservativeResize(newRows, curCols);
85 }
86
87 return character;
88}

90Character addGloveCalibrationParameters(
91 Character character,
92 const GloveConfig& cfg,
93 const std::string& prefix) {
94 auto& newSkel = character.skeleton;
95 auto& newTransform = character.parameterTransform;
96
97 ParameterSet gloveSet;
98 std::vector<Eigen::Triplet<float>> triplets;
99
100 // Parameter name suffixes for translation and rotation DOFs
101 static const std::array<std::string, 6> dofNames{"_tx", "_ty", "_tz", "_rx", "_ry", "_rz"};
102 // Corresponding joint parameter indices: TX=0, TY=1, TZ=2, RX=3, RY=4, RZ=5
103 static const std::array<size_t, 6> dofIndices{TX, TY, TZ, RX, RY, RZ};
104
105 for (size_t hand = 0; hand < 2; ++hand) {
106 const std::string gloveBoneName = prefix + cfg.wristJointNames[hand];
107 const size_t jointIdx = newSkel.getJointIdByName(gloveBoneName);
108 if (jointIdx == kInvalidIndex) {
109 MT_LOGW("addGloveCalibrationParameters: glove bone '{}' not found, skipping", gloveBoneName);
110 continue;
111 }
112
113 // Add 6 DOF parameters (TX, TY, TZ, RX, RY, RZ) for this glove bone
114 for (size_t d = 0; d < 6; ++d) {
115 const std::string paramName = gloveBoneName + dofNames[d];
116 const auto row =
117 static_cast<int>(jointIdx) * kParametersPerJoint + static_cast<int>(dofIndices[d]);
118 const auto col = static_cast<int>(newTransform.name.size());
119
120 triplets.emplace_back(row, col, 1.0f);
121
122 MT_CHECK(
123 newTransform.name.size() < kMaxModelParams,
124 "Number of model parameters reached the {} limit.",
125 kMaxModelParams);
126 gloveSet.set(newTransform.name.size());
127 newTransform.name.push_back(paramName);
128 }
129 }
130
131 newTransform.parameterSets["gloves"] = gloveSet;
132
133 // Update the parameter transform matrix
134 const int newRows = static_cast<int>(newSkel.joints.size()) * kParametersPerJoint;
135 const int newCols = static_cast<int>(newTransform.name.size());
136 newTransform.transform.conservativeResize(newRows, newCols);
137 SparseRowMatrixf additionalTransforms(newRows, newCols);
138 additionalTransforms.setFromTriplets(triplets.begin(), triplets.end());
139 newTransform.transform += additionalTransforms;
140
141 return character;
142}

149std::array<GloveOffset, 2> extractGloveOffsetsFromCharacter(
150 const Character& gloveChar,
151 const CharacterParameters& params,
152 const GloveConfig& cfg) {
153 std::array<GloveOffset, 2> result;
154
155 // Apply parameter transform to get joint parameters
156 const JointParameters jointParams = gloveChar.parameterTransform.apply(params);
157
158 for (size_t hand = 0; hand < 2; ++hand) {
159 const std::string gloveBoneName = "glove_" + cfg.wristJointNames[hand];
160 const size_t jointIdx = gloveChar.skeleton.getJointIdByName(gloveBoneName);
161 if (jointIdx == kInvalidIndex) {
162 continue;
163 }
164
165 const size_t base = jointIdx * kParametersPerJoint;
166 result[hand].translation =
167 Vector3f(jointParams.v(base + TX), jointParams.v(base + TY), jointParams.v(base + TZ));
168 result[hand].rotationEulerXYZ =
169 Vector3f(jointParams.v(base + RX), jointParams.v(base + RY), jointParams.v(base + RZ));
170 }
171
172 return result;
173}

175void bakeGloveOffsetsFromParams(
176 Character& character,
177 const ModelParameters& solvedParams,
178 const Character& solvingCharacter,
179 const std::optional<GloveConfig>& cfg) {
180 if (!cfg) {
181 return;
182 }
183 // Extract offsets from the solving character's calibration parameters
184 CharacterParameters params;
185 params.pose = solvedParams;
186 params.offsets =
187 JointParameters::Zero(solvingCharacter.skeleton.joints.size() * kParametersPerJoint);
188 const auto offsets = extractGloveOffsetsFromCharacter(solvingCharacter, params, *cfg);
189
190 for (size_t hand = 0; hand < 2; ++hand) {
191 MT_LOGI(
192 "Baking glove offset for '{}': translation=({:.4f}, {:.4f}, {:.4f}), "
193 "rotation_xyz=({:.4f}, {:.4f}, {:.4f})",
194 cfg->wristJointNames[hand],
195 offsets[hand].translation.x(),
196 offsets[hand].translation.y(),
197 offsets[hand].translation.z(),
198 offsets[hand].rotationEulerXYZ.x(),
199 offsets[hand].rotationEulerXYZ.y(),
200 offsets[hand].rotationEulerXYZ.z());
201 }
202
203 // Add glove bones with the solved offsets baked into the joint
204 character = addGloveBones(character, *cfg, offsets);
205}

207std::vector<std::vector<JointToJointPositionDataT<float>>> createGlovePositionConstraintData(
208 std::span<const GloveFrameData> gloveData,
209 const Character& character,
210 const GloveConfig& cfg,
211 size_t handIndex) {
212 if (gloveData.empty() || handIndex >= 2) {
213 return {};
214 }
215
216 std::vector<std::vector<JointToJointPositionDataT<float>>> results(gloveData.size());
217
218 // Find the glove bone for this hand; fall back to the wrist joint itself
219 const std::string gloveBoneName = "glove_" + cfg.wristJointNames[handIndex];
220 size_t referenceIdx = character.skeleton.getJointIdByName(gloveBoneName);
221 if (referenceIdx == kInvalidIndex) {
222 referenceIdx = character.skeleton.getJointIdByName(cfg.wristJointNames[handIndex]);
223 if (referenceIdx == kInvalidIndex) {
224 return results;
225 }
226 }
227
228 // Build name-to-index map for the skeleton
229 std::unordered_map<std::string, size_t> jointLookup;
230 jointLookup.reserve(character.skeleton.joints.size());
231 for (size_t i = 0; i < character.skeleton.joints.size(); ++i) {
232 jointLookup[character.skeleton.joints[i].name] = i;
233 }
234
235 for (size_t iFrame = 0; iFrame < results.size(); ++iFrame) {
236 results[iFrame].reserve(gloveData[iFrame].size());
237 for (const auto& obs : gloveData[iFrame]) {
238 if (!obs.valid) {
239 continue;
240 }
241
242 auto it = jointLookup.find(obs.jointName);
243 if (it == jointLookup.end()) {
244 continue;
245 }
246
247 JointToJointPositionDataT<float> constraint;
248 constraint.sourceJoint = it->second;
249 constraint.sourceOffset = Vector3f::Zero();
250 constraint.referenceJoint = referenceIdx;
251 constraint.referenceOffset = Vector3f::Zero();
252 constraint.target = obs.position;
253 constraint.weight = 1.0f;
254 constraint.name = obs.jointName;
255 results[iFrame].push_back(std::move(constraint));
256 }
257 }
258
259 return results;
260}

315void addGloveConstraintsToSequenceSolver(
316 SequenceSolverFunctionT<float>& solverFunc,
317 size_t solverFrame,
318 const Character& character,
319 const std::vector<std::vector<JointToJointPositionDataT<float>>>& posData,
320 const std::vector<std::vector<JointToJointOrientationDataT<float>>>& oriData,
321 size_t iFrame,
322 float posWeight,
323 float oriWeight) {
324 // Add position constraints for this frame
325 if (iFrame < posData.size() && !posData[iFrame].empty()) {
326 auto posFunc = std::make_shared<JointToJointPositionErrorFunctionT<float>>(character);
327 for (const auto& c : posData[iFrame]) {
328 posFunc->addConstraint(c);
329 }
330 posFunc->setWeight(PositionErrorFunction::kLegacyWeight * posWeight);
331 solverFunc.addErrorFunction(solverFrame, posFunc);
332 }
333
334 // Add orientation constraints for this frame
335 if (iFrame < oriData.size() && !oriData[iFrame].empty()) {
336 auto oriFunc = std::make_shared<JointToJointOrientationErrorFunctionT<float>>(character);
337 for (const auto& c : oriData[iFrame]) {
338 oriFunc->addConstraint(c);
339 }
340 oriFunc->setWeight(OrientationErrorFunction::kLegacyWeight * oriWeight);
341 solverFunc.addErrorFunction(solverFrame, oriFunc);
342 }

345std::optional<GloveErrorFunctions> setupGloveErrorFunctions(
346 SkeletonSolverFunctionT<float>& solverFunc,
347 const Character& character,
348 std::span<const GloveFrameData> gloveData,
349 const GloveConfig& cfg,
350 size_t handIndex) {
351 if (gloveData.empty()) {
352 return std::nullopt;
353 }
354
355 // Verify a reference joint exists (glove bone preferred, wrist as fallback)
356 if (handIndex >= 2) {
357 return std::nullopt;
358 }
359 const std::string gloveBoneName = "glove_" + cfg.wristJointNames[handIndex];
360 size_t referenceIdx = character.skeleton.getJointIdByName(gloveBoneName);
361 if (referenceIdx == kInvalidIndex) {
362 referenceIdx = character.skeleton.getJointIdByName(cfg.wristJointNames[handIndex]);
363 if (referenceIdx == kInvalidIndex) {
364 return std::nullopt;
365 }
366 }
367
368 GloveErrorFunctions funcs;
369
370 // Build constraint data
371 funcs.posData = createGlovePositionConstraintData(gloveData, character, cfg, handIndex);
372 funcs.oriData = createGloveOrientationConstraintData(gloveData, character, cfg, handIndex);
373
374 // Create and register error functions
375 funcs.posFunc = std::make_shared<JointToJointPositionErrorFunctionT<float>>(character);
376 funcs.posFunc->setWeight(PositionErrorFunction::kLegacyWeight * cfg.positionWeight);
377 solverFunc.addErrorFunction(funcs.posFunc);
378
379 funcs.oriFunc = std::make_shared<JointToJointOrientationErrorFunctionT<float>>(character);
380 funcs.oriFunc->setWeight(OrientationErrorFunction::kLegacyWeight * cfg.orientationWeight);
381 solverFunc.addErrorFunction(funcs.oriFunc);
382
383 return funcs;
384}

386void updateGloveConstraintsForFrame(GloveErrorFunctions& funcs, size_t iFrame) {
387 // Update position constraints
388 funcs.posFunc->clearConstraints();
389 if (iFrame < funcs.posData.size()) {
390 for (const auto& c : funcs.posData[iFrame]) {
391 funcs.posFunc->addConstraint(c);
392 }
393 }
394
395 // Update orientation constraints
396 funcs.oriFunc->clearConstraints();
397 if (iFrame < funcs.oriData.size()) {
398 for (const auto& c : funcs.oriData[iFrame]) {
399 funcs.oriFunc->addConstraint(c);
400 }
401 }
