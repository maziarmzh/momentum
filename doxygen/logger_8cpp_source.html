49void logMesh(
50 const rerun::RecordingStream& rec,
51 const std::string& streamName,
52 const Mesh& mesh,
53 std::optional<rerun::Color> color) {
54 auto rerunMesh = rerun::Mesh3D(mesh.vertices).with_triangle_indices(mesh.faces);
55 if (color.has_value()) {
56 rerunMesh = std::move(rerunMesh).with_albedo_factor(rerun::AlbedoFactor(color.value()));
57 } else if (mesh.colors.size() == mesh.vertices.size()) {
58 rerunMesh = std::move(rerunMesh).with_vertex_colors(mesh.colors);
59 }
60
61 if (mesh.normals.size() == mesh.vertices.size()) {
62 rerunMesh = std::move(rerunMesh).with_vertex_normals(mesh.normals);
63 }
64
65 safeLog(rec, streamName, rerunMesh);
66}

68void logJoints(
69 const rerun::RecordingStream& rec,
70 const std::string& streamName,
71 const Skeleton& skeleton,
72 const JointStateList& jointStates) {
73 const rerun::Color kGrey(180, 180, 180);
74 const auto names = skeleton.getJointNames();
75 std::vector<std::string> labels;
76 std::vector<std::vector<std::array<float, 3>>> lines;
77 labels.reserve(names.size());
78 lines.reserve(names.size());
79 for (size_t iJoint = 0; iJoint < jointStates.size(); ++iJoint) {
80 if (iJoint >= skeleton.joints.size()) {
81 break;
82 }
83 const size_t parentIdx = skeleton.joints[iJoint].parent;
84 if (parentIdx != kInvalidIndex && parentIdx < jointStates.size()) {
85 lines.push_back(
86 {toStdArray3f(jointStates[parentIdx].transform.translation),
87 toStdArray3f(jointStates[iJoint].transform.translation)});
88 labels.push_back(names[iJoint]);
89 }
90 safeLog(
91 rec,
92 streamName + "/" + names[iJoint],
93 rerun::Transform3D()
94 .with_mat3x3(
95 rerun::datatypes::Mat3x3(jointStates[iJoint].transform.toRotationMatrix().data()))
96 .with_translation(
97 rerun::datatypes::Vec3D(jointStates[iJoint].transform.translation.data()))
98#if !defined(RERUN_VERSION_GE) || !RERUN_VERSION_GE(0, 24, 0)
99 .with_axis_length(rerun::components::AxisLength(10))
100#endif
101 );
102 }
103 safeLog(
104 rec,
105 streamName,
106 rerun::LineStrips3D(lines).with_radii(0.2f).with_colors(kGrey).with_labels(labels));
107}

134void logLocators(
135 const rerun::RecordingStream& rec,
136 const std::string& streamName,
137 const LocatorList& locators,
138 const LocatorState& locatorState) {
139 if (locators.empty()) {
140 return;
141 }
142
143 const rerun::Color kGreen(100, 255, 100);
144 std::vector<std::string> labels;
145 std::vector<rerun::Position3D> points3d;
146 std::vector<rerun::Color> colors;
147 points3d.reserve(locatorState.position.size());
148 labels.reserve(locatorState.position.size());
149 colors.reserve(locatorState.position.size());
150
151 for (size_t i = 0; i < locatorState.position.size(); ++i) {
152 const auto& locator = locatorState.position[i];
153 points3d.push_back(toRerunPosition3D(locator));
154 labels.push_back(locators[i].name);
155 rerun::Color color{};
156 if (locators.at(i).name.find("Floor_") != std::string::npos) {
157 color = kGreen;
158 } else {
159 float intensity = 255.0f * locators[i].weight * 0.6f;
160 color = rerun::Color(
161 static_cast<uint8_t>(intensity),
162 static_cast<uint8_t>(intensity),
163 static_cast<uint8_t>(intensity * 0.5f));
164 }
165 colors.push_back(color);
166 }
167
168 // TODO: make radius and color configurable
169 safeLog(
170 rec,
171 streamName,
172 rerun::Points3D(points3d).with_radii(0.5f).with_colors(colors).with_labels(labels));
173}

175void logMarkerLocatorCorrespondence(
176 const rerun::RecordingStream& rec,
177 const std::string& streamName,
178 const std::map<std::string, size_t>& locatorLookup,
179 const LocatorState& locatorState,
180 std::span<const Marker> markers,
181 const float kPositionErrorThreshold) {
182 if (locatorLookup.empty()) {
183 // No correspondence provided.
184 return;
185 }
186 std::vector<std::string> labels;
187 std::vector<std::vector<std::array<float, 3>>> lines;
188 std::vector<rerun::components::Color> colors;
189 std::vector<rerun::components::Radius> radius;
190 lines.reserve(markers.size());
191 labels.reserve(markers.size());
192 colors.reserve(markers.size());
193 radius.reserve(markers.size());
194
195 const rerun::components::Color kGreenColor(50, 255, 128);
196 const rerun::components::Color kRedColor(255, 100, 100);
197 const rerun::components::Radius kDefaultRadius(0.1f);
198 const rerun::components::Radius kLargeRadius(0.5f);
199
200 for (const auto& marker : markers) {
201 if (!marker.occluded && (locatorLookup.count(marker.name) != 0)) {
202 const auto locatorIdx = locatorLookup.at(marker.name);
203 const auto locator = locatorState.position.at(locatorIdx);
204
205 lines.push_back({toStdArray3f(marker.pos), toStdArray3f(locator)});
206 labels.push_back(marker.name);
207
208 // TODO: expose marker error computation?
209 const auto error = (marker.pos.cast<float>() - locator).squaredNorm();
210 if (error > kPositionErrorThreshold) {
211 colors.push_back(kRedColor);
212 radius.push_back(kLargeRadius);
213 } else {
214 colors.push_back(kGreenColor);
215 radius.push_back(kDefaultRadius);
216 }
217 }
218
219 safeLog(
220 rec,
221 streamName,
222 rerun::LineStrips3D(lines).with_labels(labels).with_colors(colors).with_radii(radius));
223 }

303void logModelParamsColumns(
304 const rerun::RecordingStream& rec,
305 const std::string& worldPrefix,
306 const std::string& posePrefix,
307 std::span<const std::string> names,
308 const Eigen::MatrixXf& allParams,
309 std::span<const int64_t> frameIndices,
310 std::span<const double> times) {
311 const auto nFrames = static_cast<size_t>(allParams.cols());
312 if (nFrames == 0) {
313 return;
314 }
315
316 // Build time columns (reused for all parameters)
317 auto frameIndexColumn = makeSequenceTimeColumn(
318 "frame_index",
319 rerun::Collection<int64_t>::borrow(frameIndices.data(), frameIndices.size()),
320 rerun::SortingStatus::Sorted);
321 auto timeColumn = makeDurationSecondsTimeColumn(
322 "time",
323 rerun::Collection<double>::borrow(times.data(), times.size()),
324 rerun::SortingStatus::Sorted);
325 const std::vector<rerun::TimeColumn> timeColumns = {
326 std::move(frameIndexColumn), std::move(timeColumn)};
327
328 const size_t nParams = std::min(static_cast<size_t>(allParams.rows()), names.size());
329 for (size_t iParam = 0; iParam < nParams; ++iParam) {
330 // Build scalar values for this parameter across all frames
331 std::vector<double> scalarValues(nFrames);
332 for (size_t iFrame = 0; iFrame < nFrames; ++iFrame) {
333 scalarValues[iFrame] = static_cast<double>(allParams(iParam, iFrame));
334 }
335
336 auto scalarColumns = makeScalarColumns(std::move(scalarValues));
337
338 if (names[iParam].find("root") != std::string::npos) {
339 rec.send_columns(
340 fmt::format("{}/{}", worldPrefix, names[iParam]), timeColumns, scalarColumns);
341 } else {
342 rec.send_columns(fmt::format("{}/{}", posePrefix, names[iParam]), timeColumns, scalarColumns);
343 }
344 }

399void logBvh(
400 const rerun::RecordingStream& rec,
401 const std::string& streamName,
402 const CollisionGeometry& collisionGeometry,
403 const SkeletonState& skeletonState) {
404 // Compute collision state
405 CollisionGeometryState collisionState;
406 collisionState.update(skeletonState, collisionGeometry);
407
408 // Compute AABBs
409 const auto n = collisionGeometry.size();
410 std::vector<axel::BoundingBoxf> aabbs(n);
411 for (size_t i = 0; i < n; ++i) {
412 auto& aabb = aabbs[i];
413 aabb.id = static_cast<axel::Index>(i);
414 updateAabb(
415 aabb, collisionState.origin[i], collisionState.direction[i], collisionState.radius[i]);
416 }
417
418 // Construct BVH
419 axel::Bvhf bvh;
420 bvh.setBoundingBoxes(aabbs);
421 const auto& bvs = bvh.getPrimitives();
422
423 // Log BVH
424 std::vector<rerun::Position3D> centers;
425 std::vector<rerun::HalfSize3D> halfSizes;
426 centers.reserve(bvs.size());
427 halfSizes.reserve(bvs.size());
428 for (const auto& bv : bvs) {
429 centers.push_back(toRerunPosition3D(bv.center()));
430 halfSizes.push_back(toRerunHalfSizes3D((bv.max() - bv.min()) / 2.0f));
431 }
432
433 // TODO: make radius configurable
434 // TODO: use different colors by the depth of BVH nodes
435 safeLog(
436 rec,
437 streamName,
438 rerun::Boxes3D::from_centers_and_half_sizes(centers, halfSizes)
439 .with_radii(0.1f)
440 .with_colors(rerun::Color(64, 128, 64)));
441}

443void logCollisionGeometry(
444 const rerun::RecordingStream& rec,
445 const std::string& streamName,
446 const CollisionGeometry& collisionGeometry,
447 const SkeletonState& skeletonState) {
448 // TODO: update collision geometry representation from box to capsule
449
450 std::vector<rerun::Position3D> translations;
451 std::vector<rerun::Quaternion> quaternions;
452 std::vector<float> lengths;
453 std::vector<float> radii;
454
455 translations.reserve(collisionGeometry.size());
456 quaternions.reserve(collisionGeometry.size());
457 lengths.reserve(collisionGeometry.size());
458 radii.reserve(collisionGeometry.size());
459
460 for (const auto& cg : collisionGeometry) {
461 const auto& js = skeletonState.jointState.at(cg.parent);
462
463 const Transform tf = js.transform * cg.transformation;
464 const Quaternionf& q = tf.rotation * Eigen::AngleAxisf(0.5f * pi(), Vector3f::UnitY());
465
466 translations.push_back(toRerunPosition3D(tf.translation));
467 quaternions.emplace_back(rerun::Quaternion::from_xyzw(q.x(), q.y(), q.z(), q.w()));
468 lengths.emplace_back(cg.length);
469 // TODO: Rerun doesn't support capsules with different radii (i.e. tapered capsule) yet
470 radii.emplace_back(cg.radius.maxCoeff());
471 }
472
473 // TODO: make radius and color configurable
474 safeLog(
475 rec,
476 streamName,
477 rerun::Capsules3D::from_lengths_and_radii(lengths, radii)
478 .with_translations(translations)
479 .with_quaternions(quaternions)
480 .with_colors(rerun::Color(128, 64, 64)));
481 // TODO: Switch to wireframe once available in Rerun
482
483 logBvh(rec, streamName + "/bvh", collisionGeometry, skeletonState);
484}
