31template <typename T>
32CharacterParametersT<T> InverseParameterTransformT<T>::apply(
33 const JointParametersT<T>& jointParameters) const {
34 MT_CHECK(
35 jointParameters.size() == transform.rows(),
36 "{} is not {}",
37 jointParameters.size(),
38 transform.rows());
39 MT_CHECK(offsets.size() == transform.rows(), "{} is not {}", offsets.size(), transform.rows());
40 CharacterParametersT<T> result;
41 result.pose = inverseTransform.solve(jointParameters.v - offsets);
42 result.offsets = jointParameters.v - transform * result.pose.v;
43 return result;
44}

49ModelParameters applyModelParameterScales(
50 const ParameterTransform& parameterTransform,
51 MatrixXf& motion,
52 const JointParameters& jointIdentity) {
53 const auto scalingParams = parameterTransform.getScalingParameters();
54
55 // Check if motion already contains non-zero scale parameters
56 bool hasScaleParametersInMotion = false;
57 for (Eigen::Index iFrame = 0; iFrame < motion.cols() && !hasScaleParametersInMotion; ++iFrame) {
58 for (Eigen::Index iParam = 0; iParam < parameterTransform.numAllModelParameters(); ++iParam) {
59 if (scalingParams.test(iParam) && motion(iParam, iFrame) != 0) {
60 hasScaleParametersInMotion = true;
61 break;
62 }
63 }
64 }
65
66 // Convert joint identity parameters to model identity parameters
67 ModelParameters modelIdentity = ModelParameters::Zero(parameterTransform.numAllModelParameters());
68 if (jointIdentity.size() > 0 && !jointIdentity.v.isZero()) {
69 if (hasScaleParametersInMotion) {
70 MT_LOGW("Motion already contains scale parameters, not adding joint identity parameters.");
71 }
72 const auto scalingTransform = parameterTransform.simplify(scalingParams);
73 const ModelParameters scaleParametersSubset =
74 InverseParameterTransform(scalingTransform).apply(jointIdentity).pose;
75
76 MT_CHECK(
77 static_cast<Eigen::Index>(parameterTransform.name.size()) ==
78 parameterTransform.numAllModelParameters(),
79 "Parameter name count ({}) does not match model parameter count ({})",
80 parameterTransform.name.size(),
81 parameterTransform.numAllModelParameters());
82 for (Eigen::Index iParam = 0; iParam < parameterTransform.numAllModelParameters(); iParam++) {
83 if (scalingParams.test(iParam)) {
84 MT_THROW_IF(
85 static_cast<size_t>(iParam) >= parameterTransform.name.size(),
86 "Parameter index {} exceeds parameter name count {}",
87 iParam,
88 parameterTransform.name.size());
89 auto subsetParamIdx =
90 scalingTransform.getParameterIdByName(parameterTransform.name[iParam]);
91 MT_THROW_IF(
92 subsetParamIdx >= static_cast<size_t>(scaleParametersSubset.size()),
93 "Subset parameter index {} exceeds scaleParametersSubset size {}",
94 subsetParamIdx,
95 scaleParametersSubset.size());
96 modelIdentity[iParam] = scaleParametersSubset(subsetParamIdx);
97
98 // Add the identity parameters into the motion:
99 if (!hasScaleParametersInMotion) {
100 for (Eigen::Index jFrame = 0; jFrame < motion.cols(); ++jFrame) {
101 motion(iParam, jFrame) += scaleParametersSubset(subsetParamIdx);
102 }
103 }
104 }
105 }
106 }
107
108 return modelIdentity;
109}
