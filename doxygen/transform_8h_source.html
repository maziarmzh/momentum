124 [[nodiscard]] TransformT<T> operator*(const TransformT<T>& other) const {
125 const Vector3<T> trans = translation + rotation * (scale * other.translation);
126 const Quaternion<T> rot = rotation * other.rotation;
127 const T newScale = scale * other.scale;
128 return TransformT<T>(std::move(trans), std::move(rot), std::move(newScale));

173 [[nodiscard]] Vector3<T> getAxisX() const {
174 Vector3<T> axis;
175
176 const T ty = T(2) * rotation.y();
177 const T tz = T(2) * rotation.z();
178 const T twy = ty * rotation.w();
179 const T twz = tz * rotation.w();
180 const T txy = ty * rotation.x();
181 const T txz = tz * rotation.x();
182 const T tyy = ty * rotation.y();
183 const T tzz = tz * rotation.z();
184
185 axis[0] = T(1) - (tyy + tzz);
186 axis[1] = txy + twz;
187 axis[2] = txz - twy;
188
189 return axis;
190 }
