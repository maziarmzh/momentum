62template <typename T, class Data, size_t FuncDim>
63double VertexErrorFunctionT<T, Data, FuncDim>::getError(
64 const ModelParametersT<T>& /*params*/,
65 const SkeletonStateT<T>& state,
66 const MeshStateT<T>& meshState) {
67 const size_t numConstraints = constraints_.size();
68 if (numConstraints == 0) {
69 return 0.0;
70 }
71
72 double error = 0.0;
73 for (size_t i = 0; i < numConstraints; ++i) {
74 const T constrWeight = static_cast<T>(constraints_[i].weight);
75 if (constrWeight == T(0)) {
76 continue;
77 }
78
79 const size_t vertexIndex = constraints_[i].vertexIndex;
80 const Eigen::Vector3<T> worldVec =
81 meshState.posedMesh_->vertices[vertexIndex].template cast<T>();
82
83 FuncType f;
84 std::span<DfdvType> emptyDfdv;
85 evalFunction(
86 i, state, meshState, std::span<const Eigen::Vector3<T>>(&worldVec, 1), f, emptyDfdv);
87
88 const T sqrError = f.squaredNorm();
89 error += constrWeight * loss_.value(sqrError);
90 }
91 return this->weight_ * legacyWeight_ * error;
92}

94template <typename T, class Data, size_t FuncDim>
95double VertexErrorFunctionT<T, Data, FuncDim>::getGradient(
96 const ModelParametersT<T>& /*params*/,
97 const SkeletonStateT<T>& state,
98 const MeshStateT<T>& meshState,
99 Eigen::Ref<Eigen::VectorX<T>> gradient) {
100 const size_t numConstraints = constraints_.size();
101 if (numConstraints == 0) {
102 return 0.0;
103 }
104
105 auto dispensoOptions = dispenso::ParForOptions();
106 dispensoOptions.maxThreads = static_cast<uint32_t>(maxThreads_);
107
108 const SkeletonDerivativeT<T> skeletonDerivative(
109 this->skeleton_,
110 this->parameterTransform_,
111 this->activeJointParams_,
112 this->enabledParameters_);
113
114 const bool isL2 = loss_.isL2();
115 const T lossInvC2 = loss_.invC2();
116
117 std::vector<std::tuple<double, VectorX<T>>> errorGradThread;
118 dispenso::parallel_for(
119 errorGradThread,
120 [&]() -> std::tuple<double, VectorX<T>> {
121 return {0.0, VectorX<T>::Zero(this->parameterTransform_.numAllModelParameters())};
122 },
123 size_t(0),
124 numConstraints,
125 [&](std::tuple<double, VectorX<T>>& errorGradLocal, const size_t i) {
126 double& errorLocal = std::get<0>(errorGradLocal);
127 auto& gradLocal = std::get<1>(errorGradLocal);
128
129 const T constrWeight = static_cast<T>(constraints_[i].weight);
130 if (constrWeight == T(0)) {
131 return;
132 }
133
134 const size_t vertexIndex = constraints_[i].vertexIndex;
135 const Eigen::Vector3<T> worldVec =
136 meshState.posedMesh_->vertices[vertexIndex].template cast<T>();
137
138 FuncType f;
139 DfdvType dfdv;
140 dfdv.setZero();
141
142 evalFunction(
143 i,
144 state,
145 meshState,
146 std::span<const Eigen::Vector3<T>>(&worldVec, 1),
147 f,
148 std::span<DfdvType>(&dfdv, 1));
149
150 if (f.isZero()) {
151 return;
152 }
153
154 const T sqrError = f.squaredNorm();
155 const T w = constrWeight * this->weight_ * legacyWeight_;
156
157 FuncType weightedResidual;
158 if (isL2) {
159 errorLocal += w * sqrError * lossInvC2;
160 weightedResidual = T(2) * w * lossInvC2 * f;
161 } else {
162 errorLocal += w * loss_.value(sqrError);
163 weightedResidual = T(2) * w * loss_.deriv(sqrError) * f;
164 }
165
166 skeletonDerivative.template accumulateVertexGradient<FuncDim>(
167 vertexIndex, worldVec, dfdv, weightedResidual, state, meshState, character_, gradLocal);
168 },
169 dispensoOptions);
170
171 if (!errorGradThread.empty()) {
172 errorGradThread[0] = std::accumulate(
173 errorGradThread.begin() + 1,
174 errorGradThread.end(),
175 errorGradThread[0],
176 [](const auto& a, const auto& b) -> std::tuple<double, VectorX<T>> {
177 return {std::get<0>(a) + std::get<0>(b), std::get<1>(a) + std::get<1>(b)};
178 });
179
180 gradient += std::get<1>(errorGradThread[0]);
181 return std::get<0>(errorGradThread[0]);
182 }
183
184 return 0.0;
185}

187template <typename T, class Data, size_t FuncDim>
188double VertexErrorFunctionT<T, Data, FuncDim>::getJacobian(
189 const ModelParametersT<T>& /*params*/,
190 const SkeletonStateT<T>& state,
191 const MeshStateT<T>& meshState,
192 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
193 Eigen::Ref<Eigen::VectorX<T>> residual,
194 int& usedRows) {
195 const size_t numConstraints = constraints_.size();
196 usedRows = 0;
197 if (numConstraints == 0) {
198 return 0.0;
199 }
200
201 auto dispensoOptions = dispenso::ParForOptions();
202 dispensoOptions.maxThreads = static_cast<uint32_t>(maxThreads_);
203
204 const SkeletonDerivativeT<T> skeletonDerivative(
205 this->skeleton_,
206 this->parameterTransform_,
207 this->activeJointParams_,
208 this->enabledParameters_);
209
210 const bool isL2Jac = loss_.isL2();
211 const T lossInvC2Jac = loss_.invC2();
212
213 std::vector<double> errorThread;
214 dispenso::parallel_for(
215 errorThread,
216 [&]() -> double { return 0.0; },
217 size_t(0),
218 numConstraints,
219 [&](double& errorLocal, const size_t i) {
220 const T constrWeight = static_cast<T>(constraints_[i].weight);
221 if (constrWeight == T(0)) {
222 return;
223 }
224
225 const size_t vertexIndex = constraints_[i].vertexIndex;
226 const Eigen::Vector3<T> worldVec =
227 meshState.posedMesh_->vertices[vertexIndex].template cast<T>();
228
229 const size_t rowIndex = i * FuncDim;
230 FuncType f;
231 DfdvType dfdv;
232 dfdv.setZero();
233
234 evalFunction(
235 i,
236 state,
237 meshState,
238 std::span<const Eigen::Vector3<T>>(&worldVec, 1),
239 f,
240 std::span<DfdvType>(&dfdv, 1));
241
242 const T sqrError = f.squaredNorm();
243 const T w = constrWeight * this->weight_ * legacyWeight_;
244
245 T deriv;
246 if (isL2Jac) {
247 errorLocal += w * sqrError * lossInvC2Jac;
248 deriv = std::sqrt(w * lossInvC2Jac);
249 } else {
250 errorLocal += w * loss_.value(sqrError);
251 deriv = std::sqrt(w * loss_.deriv(sqrError));
252 }
253
254 residual.template segment<FuncDim>(rowIndex).noalias() = deriv * f;
255
256 if (deriv == T(0)) {
257 return;
258 }
259
260 skeletonDerivative.template accumulateVertexJacobian<FuncDim>(
261 vertexIndex, worldVec, dfdv, deriv, state, meshState, character_, jacobian, rowIndex);
262 },
263 dispensoOptions);
264
265 usedRows = static_cast<int>(numConstraints * FuncDim);
266
267 if (!errorThread.empty()) {
268 return std::accumulate(errorThread.begin() + 1, errorThread.end(), errorThread[0]);
269 }
270
271 return 0.0;
272}
