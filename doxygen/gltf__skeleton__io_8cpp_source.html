40[[nodiscard]] std::vector<uint32_t> getAncestorsIncludingSelf(
41 uint32_t nodeId,
42 const std::vector<uint32_t>& parents) {
43 auto ancestors = std::vector<uint32_t>();
44 auto curNode = nodeId;
45 while (curNode != kInvalidNodeId) {
46 ancestors.push_back(curNode);
47 curNode = parents[curNode];
48 }
49 std::reverse(ancestors.begin(), ancestors.end());
50 return ancestors;
51}

53[[nodiscard]] uint32_t findClosestCommonAncestor(
54 const uint32_t node1,
55 const uint32_t node2,
56 const std::vector<uint32_t>& parents) {
57 const auto ancestors1 = getAncestorsIncludingSelf(node1, parents);
58 const auto ancestors2 = getAncestorsIncludingSelf(node2, parents);
59 const auto maxNumIter = std::min(ancestors1.size(), ancestors2.size());
60 auto commonAncestor = kInvalidNodeId;
61 for (auto i = 0; i < maxNumIter; i++) {
62 if (ancestors1[i] == ancestors2[i]) {
63 commonAncestor = ancestors1[i];
64 } else {
65 break;
66 }
67 }
68 return commonAncestor;
69}

71// DFS the hierarchy graph to reconstruct it.
72// Along the way, save the joints, collision geometry and the locators, as well as a
73// nodeToObjectMap for query. nodeToObjectMap stores mapping from node index to joint index,
74// if the node is a joint. If it's a collision capsule or locator,
75// stores the index in the collision volume array or locator array.
76// If no extension is present in the glb file, right now we will load every not-skinned node as a
77// joint. #TODO: correctly load hierarchy from the skinning information if no momentum extension is
78// in the file.
79void loadHierarchyRecursive(
80 const fx::gltf::Document& model,
81 const int nodeId,
82 size_t parentJointId,
83 JointList& joints,
84 CollisionGeometry_u& collision,
85 LocatorList& locators,
86 std::vector<size_t>& nodeToObjectMap,
87 bool useExtension) {
88 MT_THROW_IF(
89 (nodeId < 0) || (nodeId >= model.nodes.size()),
90 "Invalid node id found in the gltf hierarchy: {}",
91 nodeId);
92 const auto& node = model.nodes[nodeId];
93 if (!isHierarchyNode(node)) {
94 return;
95 }
96
97 const auto& extension = getMomentumExtension(node.extensionsAndExtras);
98 const std::string type = extension.value("type", "");
99 MT_CHECK(nodeId < nodeToObjectMap.size(), "id: {}, size: {}", nodeId, nodeToObjectMap.size());
100 MT_CHECK(
101 nodeToObjectMap.size() == model.nodes.size(),
102 "NodeMap: {}, nodes: {}",
103 nodeToObjectMap.size(),
104 model.nodes.size());
105 auto newParentJointId = parentJointId;
106 if (type == "collision_capsule") {
107 // Found collision geometry, should be the end node
108 MT_THROW_IF(
109 parentJointId == kInvalidIndex,
110 "Invalid collision capsule without a parent joint: {}",
111 node.name);
112 auto capsule = createCollisionCapsule(node, extension);
113 capsule.parent = parentJointId;
114 collision->push_back(capsule);
115 nodeToObjectMap[nodeId] = collision->size() - 1;
116 } else if (type == "locator") {
117 // Found locator, should be the end node
118 MT_THROW_IF(
119 parentJointId == kInvalidIndex, "Invalid locator without a parent joint: {}", node.name);
120 Locator loc = createLocator(node, extension);
121 loc.parent = parentJointId;
122 locators.push_back(loc);
123 nodeToObjectMap[nodeId] = locators.size() - 1;
124 } else if ((!useExtension && (node.mesh < 0)) || type == "skeleton_joint") {
125 Joint joint = createJoint(node);
126 joint.parent = parentJointId;
127 joints.push_back(joint);
128 nodeToObjectMap[nodeId] = joints.size() - 1;
129 newParentJointId = joints.size() - 1;
130 }
131 // #TODO: log skipped nodes
132
133 MT_CHECK(!model.nodes.empty());
134 for (auto childId : model.nodes[nodeId].children) {
135 MT_THROW_IF(
136 (childId < 0) || (childId >= model.nodes.size()),
137 "Invalid node id found in the gltf hierarchy: {}",
138 childId);
139
140 loadHierarchyRecursive(
141 model,
142 childId,
143 newParentJointId,
144 joints,
145 collision,
146 locators,
147 nodeToObjectMap,
148 useExtension);
149 }
150}

154TaperedCapsule createCollisionCapsule(const fx::gltf::Node& node, const nlohmann::json& extension) {
155 TaperedCapsule tc;
156 tc.parent = kInvalidIndex;
157 tc.transformation = Transform();
158 tc.transformation.rotation = toMomentumQuaternionf(node.rotation);
159 tc.transformation.translation = toMomentumVec3f(node.translation);
160
161 try {
162 tc.length = extension["length"];
163 tc.radius = fromJson<Vector2f>(extension["radius"]);
164 } catch (const std::exception&) {
165 MT_THROW(
166 "Fail to parse json {} for collision capsule {} when loading character.",
167 node.name,
168 extension.dump());
169 }
170 return tc;
171}

173Locator createLocator(const fx::gltf::Node& node, const nlohmann::json& extension) {
174 Locator loc;
175 loc.parent = kInvalidIndex;
176 loc.name = node.name;
177 loc.offset = toMomentumVec3f(node.translation);
178
179 try {
180 loc.weight = extension["weight"];
181 if (extension.contains("limitOrigin")) {
182 loc.limitOrigin = fromJson<Vector3f>(extension["limitOrigin"]);
183 } else {
184 loc.limitOrigin = loc.offset;
185 }
186 loc.limitWeight = fromJson<Vector3f>(extension["limitWeight"]);
187 loc.locked = fromJson<Vector3i>(extension["locked"]);
188 if (extension.contains("attachedToSkin")) {
189 loc.attachedToSkin = extension["attachedToSkin"];
190 }
191 if (extension.contains("skinOffset")) {
192 loc.skinOffset = extension["skinOffset"];
193 }
194 } catch (const std::exception&) {
195 MT_THROW(
196 "Fail to parse json {} for locator {} when loading character.",
197 extension.dump(),
198 node.name);
199 }
200 return loc;
201}

216std::vector<std::vector<uint32_t>> gatherSkeletonRoots(const fx::gltf::Document& model) {
217 MT_CHECK(model.scene < model.scenes.size(), "{}: {}", model.scene, model.scenes.size());
218 // If no skinned mesh is present, consider the scene root nodes as skeleton roots.
219 if (model.meshes.empty() || model.skins.empty()) {
220 std::vector<uint32_t> roots = model.scenes[model.scene].nodes;
221 return {roots};
222 }
223
224 // Build a parent map for ancestor query
225 auto parentMap = std::vector<uint32_t>(model.nodes.size(), kInvalidNodeId);
226 auto nodeStack = model.scenes[model.scene].nodes;
227 while (!nodeStack.empty()) {
228 std::vector<uint32_t> nextStack;
229 for (const auto& nodeId : nodeStack) {
230 // NOLINTNEXTLINE(facebook-hte-ParameterUncheckedArrayBounds)
231 const auto& node = model.nodes[nodeId];
232 nextStack.reserve(nextStack.size() + node.children.size());
233 nextStack.insert(nextStack.end(), node.children.begin(), node.children.end());
234 for (auto child : model.nodes[nodeId].children) {
235 parentMap[child] = nodeId;
236 }
237 }
238 nodeStack.swap(nextStack);
239 }
240
241 // Assumption: all the joints in the skin are directly connected (i.e. no non-joint node in
242 // between two joints) Merge skeleton within the same skin. If they are in the same hierarchy,
243 // they can be merged under their closest common ancestor.
244 auto skinSkeletonRoots =
245 std::vector<std::vector<uint32_t>>(model.skins.size(), std::vector<uint32_t>());
246 for (auto skinId = 0; skinId < model.skins.size(); skinId++) {
247 const auto& skin = model.skins[skinId];
248 auto& roots = skinSkeletonRoots[skinId];
249 for (const auto jointId : skin.joints) {
250 // NOLINTNEXTLINE(facebook-hte-LocalUncheckedArrayBounds)
251 const auto parentId = parentMap[jointId];
252 auto parentIter = std::find(skin.joints.begin(), skin.joints.end(), parentId);
253 if (parentIter == skin.joints.end()) {
254 bool foundSibling = false;
255 for (auto& root : roots) {
256 auto commonAncestor = findClosestCommonAncestor(jointId, root, parentMap);
257 if (commonAncestor != kInvalidNodeId) {
258 root = commonAncestor;
259 foundSibling = true;
260 break;
261 }
262 }
263 if (!foundSibling) {
264 skinSkeletonRoots[skinId].push_back(jointId);
265 }
266 }
267 }
268 }
269
270 auto result = std::vector<std::vector<uint32_t>>(model.skins.size());
271 // Merge skeleton between different skins. If the skeleton root of one is an ancestor of another,
272 // the other one can be merged to the same root.
273 for (auto skinId = 0; skinId < model.skins.size(); skinId++) {
274 MT_CHECK(
275 !skinSkeletonRoots[skinId].empty(),
276 "id: {}, size: {}",
277 skinId,
278 skinSkeletonRoots[skinId].size());
279 if (skinSkeletonRoots[skinId].size() > 1) {
280 // NOLINTNEXTLINE(facebook-hte-LocalUncheckedArrayBounds)
281 auto& roots = result[skinId];
282 roots = skinSkeletonRoots[skinId];
283 } else {
284 auto ancestors = getAncestorsIncludingSelf(skinSkeletonRoots[skinId][0], parentMap);
285 bool isIndependent = true;
286 for (auto checkId = 0; checkId < model.skins.size(); checkId++) {
287 MT_CHECK(
288 !skinSkeletonRoots[checkId].empty(),
289 "id: {}, size: {}",
290 checkId,
291 skinSkeletonRoots[checkId].size());
292 if ((checkId == skinId) || skinSkeletonRoots[checkId].size() > 1) {
293 continue;
294 }
295 if (std::find(ancestors.begin(), ancestors.end(), skinSkeletonRoots[checkId][0]) !=
296 ancestors.end()) {
297 isIndependent = false;
298 break;
299 }
300 }
301 if (isIndependent) {
302 // NOLINTNEXTLINE(facebook-hte-LocalUncheckedArrayBounds)
303 auto& roots = result[skinId];
304 roots.push_back(skinSkeletonRoots[skinId][0]);
305 }
306 }
307 }
308 // Remove the empty skeletons that were skipped
309 result.erase(
310 std::remove_if(
311 result.begin(),
312 result.end(),
313 [](const std::vector<uint32_t>& roots) { return roots.empty(); }),
314 result.end());
315 return result;
316}

318std::tuple<std::string, JointList, CollisionGeometry_u, LocatorList, std::vector<size_t>>
319loadHierarchy(const fx::gltf::Document& model) {
320 JointList joints;
321 auto collision = std::make_unique<CollisionGeometry>();
322 LocatorList locators;
323 auto nodeToObjectMap = std::vector<size_t>(model.nodes.size(), kInvalidIndex);
324 const bool useExtension = hasMomentumExtension(model);
325
326 // Currently only the first found skeleton is used
327 // #TODO: support multiple characters in the file, and use the skin information stored in
328 // CharacterInfo.
329 auto skeletonRoots = gatherSkeletonRoots(model);
330 MT_CHECK(skeletonRoots.size() == 1, "{}", skeletonRoots.size());
331 for (const auto& rootNode : skeletonRoots[0]) {
332 loadHierarchyRecursive(
333 model, rootNode, kInvalidIndex, joints, collision, locators, nodeToObjectMap, useExtension);
334 if (!joints.empty()) {
335 break;
336 }
337 }
338 MT_THROW_IF(model.nodes.empty(), "No valid node found in the gltf file.");
339 std::string name = model.nodes.at(0).name;
340
341 std::sort(
342 collision->begin(), collision->end(), [](const TaperedCapsule& c1, const TaperedCapsule& c2) {
343 int c1Parent = c1.parent == kInvalidIndex ? -1 : static_cast<int>(c1.parent);
344 int c2Parent = c2.parent == kInvalidIndex ? -1 : static_cast<int>(c2.parent);
345 return c1Parent < c2Parent;
346 });
347 std::sort(locators.begin(), locators.end(), [](const Locator& l1, const Locator& l2) {
348 int l1Parent = l1.parent == kInvalidIndex ? -1 : static_cast<int>(l1.parent);
349 int l2Parent = l2.parent == kInvalidIndex ? -1 : static_cast<int>(l2.parent);
350 return l1Parent < l2Parent;
351 });
352 // remove the collision geometry if it was empty to begin with
353 if (collision->empty()) {
354 collision.reset();
355 }
356 return std::make_tuple(name, joints, std::move(collision), locators, nodeToObjectMap);
357}
