14template <typename T>
15void LowRankCovarianceMatrixT<T>::reset(T sigma, Eigen::Ref<const Eigen::MatrixX<T>> A) {
16 A_ = A;
17 sigma_ = sigma;
18
19 const Eigen::Index n = A_.cols();
20 const Eigen::Index m = A_.rows();
21
22 // Initialize the online QR with a sigma*I diagonal, then stream in the rows of A.
23 // The resulting R is the upper-triangular factor of the augmented matrix
24 // A_aug = [ sigma*I ; A ] so that A_aug^T * A_aug = sigma^2*I + A^T*A = C.
25 // Equivalently, R is (up to sign of diagonals) the Cholesky factor of C.
26 // The dummy zero rhs is required by the online QR API but unused here.
27 qrFactorization_.reset(n, sigma);
28 Eigen::VectorX<T> b = Eigen::VectorX<T>::Zero(m);
29 qrFactorization_.add(A_, b);
30}

47template <typename T>
48Eigen::VectorX<T> LowRankCovarianceMatrixT<T>::inverse_times_vec(
49 Eigen::Ref<const Eigen::VectorX<T>> rhs) const {
50 const auto& R = qrFactorization_.R();
51 MT_CHECK(R.rows() == rhs.size());
52
53 // C^{-1} * rhs = (R^T * R)^{-1} * rhs = R^{-1} * R^{-T} * rhs.
54 // Implemented as two triangular back-solves: first R^T y = rhs, then R x = y.
55 return R.template triangularView<Eigen::Upper>().solve(
56 R.template triangularView<Eigen::Upper>().transpose().solve(rhs));
57}

59template <typename T>
60Eigen::VectorX<T> LowRankCovarianceMatrixT<T>::times_vec(
61 Eigen::Ref<const Eigen::VectorX<T>> rhs) const {
62 // C * rhs = (sigma^2 * I + A^T * A) * rhs computed directly from the stored
63 // basis A_; avoids forming the dense [n x n] matrix and exploits the low-rank
64 // structure (cost O(m*n) instead of O(n^2)).
65 Eigen::VectorX<T> result = (sigma_ * sigma_) * rhs;
66 result.noalias() += A_.transpose() * (A_ * rhs);
67 return result;
68}

70template <typename T>
71Eigen::MatrixX<T> LowRankCovarianceMatrixT<T>::inverse_times_mat(
72 Eigen::Ref<const Eigen::MatrixX<T>> rhs) const {
73 const auto& R = qrFactorization_.R();
74 MT_CHECK(rhs.rows() == dimension());
75
76 // Same C^{-1} = R^{-1} * R^{-T} solve as inverse_times_vec, applied column-wise.
77 return R.template triangularView<Eigen::Upper>().solve(
78 R.template triangularView<Eigen::Upper>().transpose().solve(rhs));
79}

94template <typename T>
95T LowRankCovarianceMatrixT<T>::logDeterminant() const {
96 const auto& R = qrFactorization_.R();
97
98 // log|R| = sum of log of diagonal entries (R is upper triangular).
99 // The MT_CHECK enforces strictly positive diagonals, which the
100 // OnlineHouseholderQR is expected to produce because the augmented matrix
101 // [sigma*I; A] has full column rank for any sigma > 0 (so R is non-singular)
102 // and the implementation chooses positive diagonal signs.
103 T result = 0;
104 for (Eigen::Index i = 0; i < dimension(); ++i) {
105 MT_CHECK(R(i, i) > 0);
106 result += std::log(R(i, i));
107 }
108
109 // C = R^T * R, so log|C| = log|R^T| + log|R| = 2 * log|R|.
110 return T(2) * result;
111}

113template <typename T>
114T LowRankCovarianceMatrixT<T>::inverse_logDeterminant() const {
115 return -logDeterminant();
116}
