51template <typename T>
52double VertexVertexDistanceErrorFunctionT<T>::getError(
53 const ModelParametersT<T>& /* modelParameters */,
54 const SkeletonStateT<T>& /* state */,
55 const MeshStateT<T>& meshState) {
56 MT_PROFILE_FUNCTION();
57
58 MT_CHECK_NOTNULL(meshState.posedMesh_);
59
60 double error = 0.0;
61
62 for (const auto& constraint : constraints_) {
63 const auto& pos1 = meshState.posedMesh_->vertices[constraint.vertexIndex1];
64 const auto& pos2 = meshState.posedMesh_->vertices[constraint.vertexIndex2];
65
66 const T actualDistance = (pos1 - pos2).norm();
67 const T distanceDiff = actualDistance - constraint.targetDistance;
68
69 error += constraint.weight * distanceDiff * distanceDiff;
70 }
71
72 return error * this->weight_;
73}

95template <typename T>
96double VertexVertexDistanceErrorFunctionT<T>::getJacobian(
97 const ModelParametersT<T>& modelParameters,
98 const SkeletonStateT<T>& state,
99 const MeshStateT<T>& meshState,
100 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
101 Eigen::Ref<Eigen::VectorX<T>> residual,
102 int& usedRows) {
103 MT_PROFILE_FUNCTION();
104
105 MT_CHECK(
106 jacobian.cols() == static_cast<Eigen::Index>(this->parameterTransform_.transform.cols()));
107 MT_CHECK(jacobian.rows() >= static_cast<Eigen::Index>(constraints_.size()));
108 MT_CHECK(residual.rows() >= static_cast<Eigen::Index>(constraints_.size()));
109
110 MT_CHECK_NOTNULL(meshState.posedMesh_);
111
112 double error = 0.0;
113
114 for (size_t i = 0; i < constraints_.size(); i++) {
115 T residualValue;
116 error += calculateJacobian(
117 modelParameters,
118 state,
119 meshState,
120 constraints_[i],
121 jacobian.block(i, 0, 1, modelParameters.size()),
122 residualValue);
123 residual(i) = residualValue;
124 }
125
126 usedRows = static_cast<int>(constraints_.size());
127 return error;
128}

135template <typename T>
136double VertexVertexDistanceErrorFunctionT<T>::calculateJacobian(
137 const ModelParametersT<T>& /* modelParameters */,
138 const SkeletonStateT<T>& state,
139 const MeshStateT<T>& meshState,
140 const VertexVertexDistanceConstraintT<T>& constraint,
141 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
142 T& residual) const {
143 MT_PROFILE_FUNCTION();
144
145 const auto& pos1 = meshState.posedMesh_->vertices[constraint.vertexIndex1];
146 const auto& pos2 = meshState.posedMesh_->vertices[constraint.vertexIndex2];
147
148 const Eigen::Vector3<T> diff = pos1 - pos2;
149 const T actualDistance = diff.norm();
150
151 // Handle degenerate case where vertices are at the same position
152 if (actualDistance == T(0)) {
153 residual = T(0);
154 return T(0); // No meaningful jacobian when distance is zero
155 }
156
157 const T distanceDiff = actualDistance - constraint.targetDistance;
158 const Eigen::Vector3<T> distanceGradient = diff / actualDistance; // normalized difference vector
159
160 // Weight for the jacobian: sqrt(weight * this->weight_)
161 const T wgt = std::sqrt(constraint.weight * this->weight_);
162
163 // Set residual: wgt * distanceDiff
164 residual = wgt * distanceDiff;
165
166 // df/dv for distance function is the normalized direction (1×3 matrix)
167 // For vertex1: positive contribution, for vertex2: negative contribution
168 const Eigen::Matrix<T, 1, 3> dfdv1 = distanceGradient.transpose();
169 const Eigen::Matrix<T, 1, 3> dfdv2 = -distanceGradient.transpose();
170
171 SkeletonDerivativeT<T> deriv(
172 this->skeleton_,
173 this->parameterTransform_,
174 this->activeJointParams_,
175 this->enabledParameters_);
176
177 // Calculate jacobian contribution from vertex1 (positive contribution)
178 deriv.template accumulateVertexJacobian<1>(
179 constraint.vertexIndex1, pos1, dfdv1, wgt, state, meshState, character_, jacobian, 0);
180
181 // Calculate jacobian contribution from vertex2 (negative contribution)
182 deriv.template accumulateVertexJacobian<1>(
183 constraint.vertexIndex2, pos2, dfdv2, wgt, state, meshState, character_, jacobian, 0);
184
185 return wgt * wgt * distanceDiff * distanceDiff;
186}

188template <typename T>
189double VertexVertexDistanceErrorFunctionT<T>::calculateGradient(
190 const ModelParametersT<T>& /* modelParameters */,
191 const SkeletonStateT<T>& state,
192 const MeshStateT<T>& meshState,
193 const VertexVertexDistanceConstraintT<T>& constraint,
194 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
195 MT_PROFILE_FUNCTION();
196
197 const auto& pos1 = meshState.posedMesh_->vertices[constraint.vertexIndex1];
198 const auto& pos2 = meshState.posedMesh_->vertices[constraint.vertexIndex2];
199
200 const Eigen::Vector3<T> diff = pos1 - pos2;
201 const T actualDistance = diff.norm();
202
203 // Handle degenerate case where vertices are at the same position
204 if (actualDistance == T(0)) {
205 return T(0); // No meaningful gradient when distance is zero
206 }
207
208 const T distanceDiff = actualDistance - constraint.targetDistance;
209 const Eigen::Vector3<T> distanceGradient = diff / actualDistance; // normalized difference vector
210
211 // df/dv for distance function is the normalized direction (1×3 matrix)
212 // For vertex1: positive contribution, for vertex2: negative contribution
213 const Eigen::Matrix<T, 1, 3> dfdv1 = distanceGradient.transpose();
214 const Eigen::Matrix<T, 1, 3> dfdv2 = -distanceGradient.transpose();
215
216 // weightedResidual = 2 * weight * loss_deriv * f
217 // For distance: f = distanceDiff, loss = 0.5 * f^2, loss_deriv = 1
218 // So weightedResidual = 2 * weight * distanceDiff
219 const Eigen::Matrix<T, 1, 1> weightedResidual =
220 Eigen::Matrix<T, 1, 1>::Constant(T(2) * constraint.weight * this->weight_ * distanceDiff);
221
222 SkeletonDerivativeT<T> deriv(
223 this->skeleton_,
224 this->parameterTransform_,
225 this->activeJointParams_,
226 this->enabledParameters_);
227
228 // Calculate gradient contribution from vertex1 (positive contribution)
229 deriv.template accumulateVertexGradient<1>(
230 constraint.vertexIndex1,
231 pos1,
232 dfdv1,
233 weightedResidual,
234 state,
235 meshState,
236 character_,
237 gradient);
238
239 // Calculate gradient contribution from vertex2 (negative contribution)
240 deriv.template accumulateVertexGradient<1>(
241 constraint.vertexIndex2,
242 pos2,
243 dfdv2,
244 weightedResidual,
245 state,
246 meshState,
247 character_,
248 gradient);
249
250 return constraint.weight * distanceDiff * distanceDiff;
251}
