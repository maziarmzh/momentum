15// Rerun SDK version compatibility layer
16// This header provides compatibility between rerun 0.23.x (legacy) and 0.29.x+ (latest) APIs.
17//
18// Minimum supported versions:
19// - rerun-latest: >=0.29.0
20// - rerun-legacy: >=0.23.3,<0.24
21//
22// Key API changes between versions:
23// - 0.24+: Scalar -> Scalars (takes batch of values)
24// - 0.24+: SeriesLine -> SeriesLines
25// - 0.24+: SeriesPoint -> SeriesPoints
26// - 0.26+: set_time_seconds deprecated -> use set_time_duration_secs
27// - 0.28+: Transform3D::with_axis_length removed (use Axes3D archetype instead)
28// - 0.29+: No breaking API changes affecting this codebase
29//
30// The RERUN_VERSION_GE macro is available since rerun 0.18.
31

34// Helper function to log a single scalar value
35// In rerun 0.24+, Scalar was renamed to Scalars and takes a batch of values
36// This wrapper provides a consistent API across versions
37template <typename T>
38inline auto makeScalar([[maybe_unused]] T value) {
39#if defined(RERUN_VERSION_GE) && RERUN_VERSION_GE(0, 24, 0)
40 // Rerun 0.24+: Use Scalars with a single-element batch
41 return rerun::Scalars({static_cast<double>(value)});
42#else
43 // Rerun 0.23.x: Use Scalar with a single value
44 return rerun::Scalar(value);
45#endif
46}

145template <typename T>
146inline bool hasComponentData(const T& archetype) {
147 const auto result = rerun::AsComponents<T>().as_batches(archetype);
148 return result.is_ok() && !result.value.empty();

156template <typename T>
157inline void
158safeLog(const rerun::RecordingStream& rec, std::string_view entityPath, const T& archetype) {
159 auto result = rerun::AsComponents<T>().as_batches(archetype);
160 if (result.is_err() || result.value.empty()) {
161 return;
162 }
163 rec.try_log_serialized_batches(entityPath, false, std::move(result.value).to_vector()).handle();
164}
