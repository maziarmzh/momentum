31template <typename T>
32void VertexPlaneErrorFunctionT<T>::evalFunction(
33 size_t constrIndex,
34 const SkeletonStateT<T>& /*state*/,
35 const MeshStateT<T>& meshState,
36 std::span<const Eigen::Vector3<T>> worldVecs,
37 FuncType& f,
38 std::span<DfdvType> dfdv) const {
39 MT_CHECK(constrIndex < this->constraints_.size(), "Constraint index out of range");
40 MT_CHECK(worldVecs.size() == 1, "Expected exactly one world vector for vertex plane");
41 MT_CHECK_NOTNULL(meshState.posedMesh_);
42
43 const auto& constraint = this->constraints_[constrIndex];
44
45 // Match legacy sign convention: flip the plane normal if it faces away from
46 // the posed mesh normal at this vertex. This ensures that the signed distance
47 // is consistent regardless of whether the caller provided a normal pointing
48 // toward or away from the mesh surface.
49 const Eigen::Vector3<T> sourceNormal =
50 meshState.posedMesh_->normals[constraint.vertexIndex].template cast<T>();
51 Eigen::Vector3<T> normal = constraint.normal;
52 if (sourceNormal.dot(normal) < T(0)) {
53 normal *= T(-1);
54 }
55
56 // Signed distance from vertex to plane: d = (v - point) · normal
57 const T distance = (worldVecs[0] - constraint.point).dot(normal);
58
59 // Half-space constraint: clamp to zero when above the plane
60 if (constraint.above && distance > T(0)) {
61 f(0) = T(0);
62 if (!dfdv.empty()) {
63 dfdv[0].setZero();
64 }
65 } else {
66 f(0) = distance;
67 if (!dfdv.empty()) {
68 MT_CHECK(dfdv.size() == 1, "Expected exactly one dfdv for vertex plane");
69 dfdv[0] = normal.transpose();
70 }
71 }
72}
