31size_t
32addMesh(const fx::gltf::Document& model, const fx::gltf::Primitive& primitive, Mesh_u& mesh) {
33 if (primitive.mode != fx::gltf::Primitive::Mode::Triangles) {
34 MT_LOGW("Mesh not loaded because it's not a triangle mesh");
35 return 0;
36 }
37
38 // load index buffer
39 auto idxDense = copyAccessorBuffer<uint32_t>(model, primitive.indices);
40 if (idxDense.empty()) {
41 // Try fallback with short indices.
42 auto a = copyAccessorBuffer<uint16_t>(model, primitive.indices);
43 for (const auto& ae : a) {
44 idxDense.push_back(ae);
45 }
46 }
47 MT_CHECK(idxDense.size() % 3 == 0, "{} % 3 = {}", idxDense.size(), idxDense.size() % 3);
48 std::vector<Vector3i> idx(idxDense.size() / 3);
49 if (!idx.empty()) {
50 std::copy_n(idxDense.data(), idxDense.size(), &idx[0][0]);
51 }
52
53 // load vertex position buffer
54 auto pos = copyAccessorBuffer<Vector3f>(model, primitive.attributes.at("POSITION"));
55 toMomentumVec3f(pos);
56
57 // if we have no points or indices, skip loading
58 if (idx.empty() || pos.empty()) {
59 return 0;
60 }
61
62 // load optional normal buffer
63 std::vector<Vector3f> nml;
64 const auto normId = primitive.attributes.find("NORMAL");
65 if (normId != primitive.attributes.end()) {
66 nml = copyAccessorBuffer<Vector3f>(model, normId->second);
67 }
68 MT_CHECK(nml.empty() || nml.size() == pos.size(), "nml: {}, pos: {}", nml.size(), pos.size());
69 MT_LOGT_IF(nml.empty(), "no vertex normal found");
70
71 // load optional color buffer
72 std::vector<Vector3b> col;
73 const auto colorId = primitive.attributes.find("COLOR_0");
74 if (colorId != primitive.attributes.end()) {
75 col = copyAccessorBuffer<Vector3b>(model, colorId->second);
76 }
77 MT_CHECK(col.empty() || col.size() == pos.size(), "col: {}, pos: {}", col.size(), pos.size());
78 MT_LOGT_IF(col.empty(), "no vertex color found");
79
80 // NOTE: gltf does not support multiple texcoords per vertex, so we only load the GLTF texcoords
81 // if we don't have our own custom extension
82 const auto& extension = getMomentumExtension(primitive);
83 std::vector<Vector2f> texcoord;
84 std::vector<Vector3i> texfaces;
85 if (extension.count("texcoords") > 0 && extension.count("texfaces") > 0) {
86 texcoord = copyAccessorBuffer<Vector2f>(model, extension.at("texcoords"));
87
88 // load "face index buffer
89 auto fidxDense = copyAccessorBuffer<uint32_t>(model, extension.at("texfaces"));
90 MT_CHECK(fidxDense.size() % 3 == 0, "{} % 3 = {}", fidxDense.size(), fidxDense.size() % 3);
91 texfaces.resize(fidxDense.size() / 3);
92 if (!fidxDense.empty()) {
93 MT_THROW_IF(texfaces.empty(), "texfaces is empty but fidxDense is not empty");
94 std::copy_n(fidxDense.data(), fidxDense.size(), &texfaces[0][0]);
95 }
96 } else {
97 // load optional standard GLTF texcoord buffer
98 auto texcoordId = primitive.attributes.find("TEXCOORD_0");
99 if (texcoordId != primitive.attributes.end()) {
100 texcoord = copyAccessorBuffer<Vector2f>(model, texcoordId->second);
101 }
102 const auto kVertexOffset = mesh->vertices.size();
103 // Update vertex indices of the faces!!!
104 if (kVertexOffset > 0) {
105 for (auto&& iFace : idx) {
106 iFace[0] += kVertexOffset;
107 iFace[1] += kVertexOffset;
108 iFace[2] += kVertexOffset;
109 }
110 }
111 if (texcoord.size() == pos.size()) {
112 texfaces = idx;
113 }
114 MT_CHECK(
115 texcoord.empty() || texcoord.size() == pos.size(),
116 "texcoord: {}, pos: {}",
117 texcoord.size(),
118 pos.size());
119 }
120 MT_LOGT_IF(texcoord.empty(), "no texture coords found");
121
122 // append new faces
123 mesh->faces.insert(mesh->faces.end(), idx.begin(), idx.end());
124 mesh->vertices.insert(mesh->vertices.end(), pos.begin(), pos.end());
125 mesh->normals.insert(mesh->normals.end(), nml.begin(), nml.end());
126 mesh->colors.insert(mesh->colors.end(), col.begin(), col.end());
127 mesh->texcoords.insert(mesh->texcoords.end(), texcoord.begin(), texcoord.end());
128 mesh->texcoord_faces.insert(mesh->texcoord_faces.end(), texfaces.begin(), texfaces.end());
129
130 // make sure we have enough normals and colors
131 mesh->normals.resize(mesh->vertices.size(), Vector3f::Zero());
132 mesh->colors.resize(mesh->vertices.size(), Vector3b::Zero());
133 mesh->confidence.resize(mesh->vertices.size(), 1.0f);
134 return pos.size();
135}

137size_t addBlendShapes(
138 const fx::gltf::Document& model,
139 const fx::gltf::Primitive& primitive,
140 BlendShape_u& blendShape) {
141 if (primitive.targets.empty()) {
142 return 0;
143 }
144
145 const size_t numNewTargets = primitive.targets.size();
146
147 // Read the base mesh vertices from the primitive
148 auto baseVertices = copyAccessorBuffer<Vector3f>(model, primitive.attributes.at("POSITION"));
149 toMomentumVec3f(baseVertices);
150 const size_t kNumNewVertices = baseVertices.size();
151
152 // Create the BlendShape if it doesn't exist yet
153 if (blendShape == nullptr) {
154 // Initialize with the base mesh vertices
155 blendShape =
156 std::make_unique<BlendShape>(std::span<const Vector3f>(baseVertices), numNewTargets);
157
158 // load blendshape names from extension
159 const auto& extension = getMomentumExtension(primitive.extensionsAndExtras);
160 std::vector<std::string> blendShapeNames =
161 extension.value("shapeNames", std::vector<std::string>());
162
163 // Load each morph target
164 for (size_t iTarget = 0; iTarget < numNewTargets; ++iTarget) {
165 const auto& target = primitive.targets[iTarget];
166
167 // Look for POSITION attribute in the morph target
168 auto posId = target.find("POSITION");
169 if (posId == target.end()) {
170 MT_LOGW("Morph target {} has no POSITION attribute", iTarget);
171 continue;
172 }
173
174 // Load the position deltas
175 auto deltas = copyAccessorBuffer<Vector3f>(model, posId->second);
176 toMomentumVec3f(deltas);
177
178 MT_CHECK(
179 deltas.size() == kNumNewVertices,
180 "Morph target {} has {} vertices but mesh has {} vertices",
181 iTarget,
182 deltas.size(),
183 kNumNewVertices);
184
185 // Set the shape vector for this target
186 if (iTarget < blendShapeNames.size()) {
187 blendShape->setShapeVector(
188 iTarget, std::span<const Vector3f>(deltas), blendShapeNames[iTarget]);
189 } else {
190 blendShape->setShapeVector(iTarget, std::span<const Vector3f>(deltas));
191 }
192 }
193 } else {
194 // Append to existing BlendShape
195 const size_t kNumExistingVertices = blendShape->modelSize();
196 const size_t kNumExistingTargets = blendShape->shapeSize();
197 const size_t kTotalVertices = kNumExistingVertices + kNumNewVertices;
198 const size_t kMaxTargets = std::max(kNumExistingTargets, numNewTargets);
199
200 // Get the existing data
201 auto existingBaseShape = blendShape->getBaseShape();
202 const auto& existingShapeVectors = blendShape->getShapeVectors();
203
204 // Create new base shape by appending new vertices to existing ones
205 std::vector<Vector3f> newBaseShape;
206 newBaseShape.reserve(kTotalVertices);
207 newBaseShape.insert(newBaseShape.end(), existingBaseShape.begin(), existingBaseShape.end());
208 newBaseShape.insert(newBaseShape.end(), baseVertices.begin(), baseVertices.end());
209
210 // Create new shape vectors matrix
211 MatrixXf newShapeVectors(kTotalVertices * 3, kMaxTargets);
212 newShapeVectors.setZero();
213
214 // Copy existing shape vectors to the top rows
215 newShapeVectors.topRows(kNumExistingVertices * 3).leftCols(kNumExistingTargets) =
216 existingShapeVectors;
217
218 // Load and append new morph target deltas
219 for (size_t iTarget = 0; iTarget < numNewTargets; ++iTarget) {
220 const auto& target = primitive.targets[iTarget];
221
222 // Look for POSITION attribute in the morph target
223 auto posId = target.find("POSITION");
224 if (posId == target.end()) {
225 MT_LOGW("Morph target {} has no POSITION attribute", iTarget);
226 continue;
227 }
228
229 // Load the position deltas
230 auto deltas = copyAccessorBuffer<Vector3f>(model, posId->second);
231 toMomentumVec3f(deltas);
232
233 MT_CHECK(
234 deltas.size() == kNumNewVertices,
235 "Morph target {} has {} vertices but mesh has {} vertices",
236 iTarget,
237 deltas.size(),
238 kNumNewVertices);
239
240 // Copy deltas into the new shape vectors matrix
241 for (size_t iVert = 0; iVert < kNumNewVertices; ++iVert) {
242 const size_t rowOffset = (kNumExistingVertices + iVert) * 3;
243 newShapeVectors(rowOffset + 0, iTarget) = deltas[iVert].x();
244 newShapeVectors(rowOffset + 1, iTarget) = deltas[iVert].y();
245 newShapeVectors(rowOffset + 2, iTarget) = deltas[iVert].z();
246 }
247 }
248
249 // Create a new BlendShape with the combined data
250 blendShape = std::make_unique<BlendShape>(std::span<const Vector3f>(newBaseShape), kMaxTargets);
251 blendShape->setShapeVectors(newShapeVectors);
252 }
253
254 return numNewTargets;
255}

257std::vector<Vector4f> loadWeightsFromAccessor(
258 const fx::gltf::Document& model,
259 int32_t accessorIdx) {
260 std::vector<Vector4f> weightsData = copyAlignedAccessorBuffer<Vector4f>(model, accessorIdx);
261 if (weightsData.empty()) {
262 // Try fallback with normalized unsigned short weights (per glTF 2.0 spec, WEIGHTS_n can be
263 // FLOAT, UNSIGNED_BYTE normalized, or UNSIGNED_SHORT normalized).
264 auto weightsShort = copyAccessorBuffer<Vector4s>(model, accessorIdx);
265 if (!weightsShort.empty()) {
266 weightsData.reserve(weightsShort.size());
267 for (const auto& value : weightsShort) {
268 weightsData.emplace_back(value.cast<float>() / 65535.0f);
269 }
270 }
271 }
272 if (weightsData.empty()) {
273 // Try fallback with normalized unsigned byte weights.
274 auto weightsByte = copyAccessorBuffer<Vector4b>(model, accessorIdx);
275 if (!weightsByte.empty()) {
276 weightsData.reserve(weightsByte.size());
277 for (const auto& value : weightsByte) {
278 weightsData.emplace_back(value.cast<float>() / 255.0f);
279 }
280 }
281 }
282 return weightsData;
283}

285void addSkinWeights(
286 const fx::gltf::Document& model,
287 const fx::gltf::Skin& skin,
288 const fx::gltf::Primitive& primitive,
289 const std::vector<size_t>& nodeToObjectMap,
290 const size_t kNumVertices,
291 SkinWeights_u& skinWeights) {
292 const auto kVertexOffset = skinWeights->index.rows();
293 skinWeights->index.conservativeResize(kVertexOffset + kNumVertices, Eigen::NoChange);
294 skinWeights->weight.conservativeResize(kVertexOffset + kNumVertices, Eigen::NoChange);
295 skinWeights->index.bottomRows(kNumVertices).setZero();
296 skinWeights->weight.bottomRows(kNumVertices).setZero();
297
298 // load skinning in batches of 4 indices/weights at a time (up to 8)
299 for (size_t i = 0; i < 2; i++) {
300 // load skinning index buffer
301 auto jointAttribute = primitive.attributes.find(std::string("JOINTS_") + std::to_string(i));
302 std::vector<Vector4s> jointIndices;
303 if (jointAttribute != primitive.attributes.end()) {
304 jointIndices = copyAccessorBuffer<Vector4s>(model, jointAttribute->second);
305 if (jointIndices.empty()) {
306 // Try fallback with short indices.
307 auto jointIndicesShort = copyAccessorBuffer<Vector4b>(model, jointAttribute->second);
308 for (const auto& value : jointIndicesShort) {
309 jointIndices.emplace_back(value.cast<uint16_t>());
310 }
311 }
312 }
313
314 // load skinning weight buffer
315 auto weightsAttribute = primitive.attributes.find(std::string("WEIGHTS_") + std::to_string(i));
316 std::vector<Vector4f> weightsData;
317 if (weightsAttribute != primitive.attributes.end()) {
318 weightsData = loadWeightsFromAccessor(model, weightsAttribute->second);
319 if (weightsData.empty()) {
320 MT_LOGW("No skinning weights read");
321 return;
322 }
323 } else {
324 MT_LOGW_IF(i == 0, "No skinning weights stored on primitive");
325 return;
326 }
327
328 if (jointIndices.empty() || weightsData.empty() || jointIndices.size() != weightsData.size() ||
329 jointIndices.size() != kNumVertices) {
330 MT_LOGW_IF(jointIndices.empty() || weightsData.empty(), "Mesh is not skinned to any joint");
331 MT_LOGW_IF(
332 !jointIndices.empty() && !weightsData.empty() &&
333 jointIndices.size() != weightsData.size(),
334 "Inconsistent data: {} vertices are skinned but {} weights found",
335 jointIndices.size(),
336 weightsData.size());
337 MT_LOGW_IF(
338 jointIndices.size() != kNumVertices,
339 "Inconsistent data: {} vertices are skinned but mesh has {} vertices",
340 jointIndices.size(),
341 kNumVertices);
342 return;
343 }
344
345 // copy indices/vertices into our buffer
346 for (Eigen::Index v = 0; v < static_cast<int>(kNumVertices); v++) {
347 for (size_t d = 0; d < 4; d++) {
348 const auto& skinWeight = weightsData[v][d];
349 const auto& jointIdxInSkin = jointIndices[v][d];
350 MT_CHECK(
351 jointIdxInSkin >= 0 && jointIdxInSkin < skin.joints.size(),
352 "{}: {}",
353 jointIdxInSkin,
354 skin.joints.size());
355 const auto& nodeIdx = skin.joints[jointIdxInSkin];
356 MT_CHECK(nodeIdx < nodeToObjectMap.size(), "{}: {}", nodeIdx, nodeToObjectMap.size());
357 const auto& jointIndex = nodeToObjectMap[nodeIdx];
358 if (jointIndex != kInvalidIndex) {
359 skinWeights->index(kVertexOffset + v, i * 4 + d) = (uint32_t)jointIndex;
360 skinWeights->weight(kVertexOffset + v, i * 4 + d) = skinWeight;
361 } else {
362 MT_LOGW("Invalid joint index encountered when reading skinning weights");
363 }
364 }
365 }
366 }

369std::tuple<Mesh_u, SkinWeights_u, BlendShape_u> loadSkinnedMesh(
370 const fx::gltf::Document& model,
371 const std::vector<size_t>& meshNodes,
372 const std::vector<size_t>& nodeToObjectMap) {
373 std::vector<size_t> skinnedNodes;
374 skinnedNodes.reserve(meshNodes.size());
375 std::vector<size_t> unskinnedNodes;
376 unskinnedNodes.reserve(meshNodes.size());
377 for (auto nodeId : meshNodes) {
378 MT_CHECK(nodeId >= 0 && nodeId < model.nodes.size(), "{}: {}", nodeId, model.nodes.size());
379 const auto& node = model.nodes[nodeId];
380 if ((node.mesh >= 0) && (node.skin >= 0)) {
381 MT_CHECK(node.mesh < model.meshes.size(), "{}: {}", node.mesh, model.meshes.size());
382 MT_CHECK(node.skin < model.skins.size(), "{}: {}", node.skin, model.skins.size());
383 skinnedNodes.push_back(nodeId);
384 } else if (node.mesh >= 0) {
385 MT_CHECK(node.mesh < model.meshes.size(), "{}: {}", node.mesh, model.meshes.size());
386 unskinnedNodes.push_back(nodeId);
387 }
388 }
389
390 MT_LOGW_IF(
391 !skinnedNodes.empty() && !unskinnedNodes.empty(),
392 "Found both skinned meshes {} and meshes without skinning {}. Unskinned meshes will be ignored.",
393 skinnedNodes.size(),
394 unskinnedNodes.size());
395 auto resultMesh = std::make_unique<Mesh>();
396 auto skinWeights = std::make_unique<SkinWeights>();
397 BlendShape_u blendShape = nullptr;
398
399 if (!skinnedNodes.empty()) {
400 for (auto nodeId : skinnedNodes) {
401 // NOLINTBEGIN(facebook-hte-ParameterUncheckedArrayBounds)
402 const auto& node = model.nodes[nodeId];
403 const auto& mesh = model.meshes[node.mesh];
404 const auto& skin = model.skins[node.skin];
405 // NOLINTEND(facebook-hte-ParameterUncheckedArrayBounds)
406 for (const auto& primitive : mesh.primitives) {
407 const auto kNumVertices = addMesh(model, primitive, resultMesh);
408 addSkinWeights(model, skin, primitive, nodeToObjectMap, kNumVertices, skinWeights);
409 addBlendShapes(model, primitive, blendShape);
410 MT_CHECK(
411 resultMesh->vertices.size() == skinWeights->index.rows(),
412 "vertices: {}, skinWeights: {}",
413 resultMesh->vertices.size(),
414 skinWeights->index.rows());
415 }
416 }
417 } else if (!unskinnedNodes.empty()) {
418 for (auto nodeId : unskinnedNodes) {
419 // NOLINTBEGIN(facebook-hte-ParameterUncheckedArrayBounds)
420 const auto& node = model.nodes[nodeId];
421 const auto& mesh = model.meshes[node.mesh];
422 // NOLINTEND(facebook-hte-ParameterUncheckedArrayBounds)
423 for (const auto& primitive : mesh.primitives) {
424 addMesh(model, primitive, resultMesh);
425 addBlendShapes(model, primitive, blendShape);
426 }
427 }
428 return {std::move(resultMesh), nullptr, std::move(blendShape)};
429 } else {
430 return {};
431 }
432 return {std::move(resultMesh), std::move(skinWeights), std::move(blendShape)};
433}

435SkinWeights_u bindMeshToJoint(const Mesh_u& mesh, size_t jointId) {
436 auto skinWeights = std::make_unique<SkinWeights>();
437 const auto kNumVertices = mesh->vertices.size();
438 skinWeights->index.conservativeResize(kNumVertices, Eigen::NoChange);
439 skinWeights->weight.conservativeResize(kNumVertices, Eigen::NoChange);
440 skinWeights->index.bottomRows(kNumVertices).setZero();
441 skinWeights->weight.bottomRows(kNumVertices).setZero();
442 for (auto vertId = 0; vertId < kNumVertices; vertId++) {
443 skinWeights->index(vertId, 0) = static_cast<uint32_t>(jointId);
444 skinWeights->weight(vertId, 0) = 1.0f;
445 }
446
447 return skinWeights;
448}
